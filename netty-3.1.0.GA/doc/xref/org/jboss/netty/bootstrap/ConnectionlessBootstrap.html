48  /**
49   * A helper class which creates a new server-side {@link Channel} for a
50   * connectionless transport.
51   *
52   * <h3>Only for connectionless transports</h3>
53   *
54   * This bootstrap is for connectionless transports only such as UDP/IP.
55   * Use {@link ServerBootstrap} instead for connection oriented transports.
56   * Do not use this helper if you are using a connection oriented transport such
57   * as TCP/IP and local transport which accepts an incoming connection and lets
58   * the accepted child channels handle received messages.
59   *
60   * <h3>Configuring channels</h3>
61   *
62   * {@link #setOption(String, Object) Options} are used to configure a channel:
63   *
64   * <pre>
65   * ConnectionlessBootstrap b = ...;
66   *
67   * // Options for a new channel
68   * b.setOption("localAddress", new {@link InetSocketAddress}(8080));
69   * b.setOption("tcpNoDelay", true);
70   * b.setOption("receiveBufferSize", 1048576);
71   * </pre>
72   *
73   * For the detailed list of available options, please refer to
74   * {@link ChannelConfig} and its sub-types
75   *
76   * <h3>Configuring a channel pipeline</h3>
77   *
78   * Every channel has its own {@link ChannelPipeline} and you can configure it
79   * in two ways.
80   * <p>
81   * {@linkplain #setPipeline(ChannelPipeline) The first approach} is to use
82   * the default pipeline and let the bootstrap to shallow-copy the default
83   * pipeline for each new channel:
84   *
85   * <pre>
86   * ConnectionlessBootstrap b = ...;
87   * {@link ChannelPipeline} p = b.getPipeline();
88   *
89   * // Add handlers to the pipeline.
90   * p.addLast("encoder", new EncodingHandler());
91   * p.addLast("decoder", new DecodingHandler());
92   * p.addLast("logic",   new LogicHandler());
93   * </pre>
94   *
95   * Please note 'shallow-copy' here means that the added {@link ChannelHandler}s
96   * are not cloned but only their references are added to the new pipeline.
97   * Therefore, you have to choose the second approach if you are going to open
98   * more than one {@link Channel} whose {@link ChannelPipeline} contains any
99   * {@link ChannelHandler} whose {@link ChannelPipelineCoverage} is {@code "one"}.
100  *
101  * <p>
102  * {@linkplain #setPipelineFactory(ChannelPipelineFactory) The second approach}
103  * is to specify a {@link ChannelPipelineFactory} by yourself and have full
104  * control over how a new pipeline is created.  This approach is more complex
105  * than the first approach while it is much more flexible:
106  *
107  * <pre>
108  * ConnectionlessBootstrap b = ...;
109  * b.setPipelineFactory(new MyPipelineFactory());
110  *
111  * public class MyPipelineFactory implements {@link ChannelPipelineFactory} {
112  *   // Create a new pipeline for a new channel and configure it here ...
113  * }
114  * </pre>
115  *
116  * <h3>Applying different settings for different {@link Channel}s</h3>
117  *
118  * {@link ConnectionlessBootstrap} is just a helper class.  It neither
119  * allocates nor manages any resources.  What manages the resources is the
120  * {@link ChannelFactory} implementation you specified in the constructor of
121  * {@link ConnectionlessBootstrap}.  Therefore, it is OK to create as
122  * many {@link ConnectionlessBootstrap} instances as you want to apply
123  * different settings for different {@link Channel}s.
124  *
125  * @author The Netty Project (netty-dev@lists.jboss.org)
126  * @author Trustin Lee (tlee@redhat.com)
127  *
128  * @version $Rev: 1393 $, $Date: 2009-06-17 14:09:45 +0900 (Wed, 17 Jun 2009) $
129  *
130  * @apiviz.landmark
131  */

150     /**
151      * Creates a new channel which is bound to the local address which was
152      * specified in the current {@code "localAddress"} option.  This method is
153      * similar to the following code:
154      *
155      * <pre>
156      * ServerBootstrap b = ...;
157      * b.connect(b.getOption("localAddress"));
158      * </pre>
159      *
160      * @return a new bound channel which accepts incoming connections
161      *
162      * @throws IllegalStateException
163      *         if {@code "localAddress"} option was not set
164      * @throws ClassCastException
165      *         if {@code "localAddress"} option's value is
166      *         neither a {@link SocketAddress} nor {@code null}
167      * @throws ChannelException
168      *         if failed to create a new channel and
169      *                      bind it to the local address
170      */
171     public Channel bind() {
172         SocketAddress localAddress = (SocketAddress) getOption("localAddress");
173         if (localAddress == null) {
174             throw new IllegalStateException("localAddress option is not set.");
175         }
176         return bind(localAddress);
177     }

179     /**
180      * Creates a new channel which is bound to the specified local address.
181      *
182      * @return a new bound channel which accepts incoming connections
183      *
184      * @throws ChannelException
185      *         if failed to create a new channel and
186      *                      bind it to the local address
187      */
188     public Channel bind(final SocketAddress localAddress) {
189         if (localAddress == null) {
190             throw new NullPointerException("localAddress");
191         }
192 
193         final BlockingQueue<ChannelFuture> futureQueue =
194             new LinkedBlockingQueue<ChannelFuture>();
195 
196         ChannelPipeline pipeline;
197         try {
198             pipeline = getPipelineFactory().getPipeline();
199         } catch (Exception e) {
200             throw new ChannelPipelineException("Failed to initialize a pipeline.", e);
201         }
202 
203         pipeline.addFirst("binder", new ConnectionlessBinder(localAddress, futureQueue));
204 
205         Channel channel = getFactory().newChannel(pipeline);
206 
207         // Wait until the future is available.
208         ChannelFuture future = null;
209         do {
210             try {
211                 future = futureQueue.poll(Integer.MAX_VALUE, TimeUnit.SECONDS);
212             } catch (InterruptedException e) {
213                 // Ignore
214             }
215         } while (future == null);
216 
217         pipeline.remove("binder");
218 
219         // Wait for the future.
220         future.awaitUninterruptibly();
221         if (!future.isSuccess()) {
222             future.getChannel().close().awaitUninterruptibly();
223             throw new ChannelException("Failed to bind to: " + localAddress, future.getCause());
224         }
225 
226         return channel;
227     }

290     /**
291      * Creates a new connected channel with the specified
292      * {@code "remoteAddress"} and the specified {@code "localAddress"}.
293      * If the specified local address is {@code null}, the local address of a
294      * new channel is determined automatically.
295      *
296      * @return a future object which notifies when the creation of the connected
297      *         channel succeeds or fails
298      *
299      * @throws ChannelPipelineException
300      *         if this bootstrap's {@link #setPipelineFactory(ChannelPipelineFactory) pipelineFactory}
301      *            failed to create a new {@link ChannelPipeline}
302      */
303     public ChannelFuture connect(final SocketAddress remoteAddress, final SocketAddress localAddress) {
304 
305         if (remoteAddress == null) {
306             throw new NullPointerException("remoteAddress");
307         }
308 
309         final BlockingQueue<ChannelFuture> futureQueue =
310             new LinkedBlockingQueue<ChannelFuture>();
311 
312         ChannelPipeline pipeline;
313         try {
314             pipeline = getPipelineFactory().getPipeline();
315         } catch (Exception e) {
316             throw new ChannelPipelineException("Failed to initialize a pipeline.", e);
317         }
318 
319         pipeline.addFirst(
320                 "connector", new ClientBootstrap.Connector(
321                         this, remoteAddress, localAddress, futureQueue));
322 
323         getFactory().newChannel(pipeline);
324 
325         // Wait until the future is available.
326         ChannelFuture future = null;
327         do {
328             try {
329                 future = futureQueue.poll(Integer.MAX_VALUE, TimeUnit.SECONDS);
330             } catch (InterruptedException e) {
331                 // Ignore
332             }
333         } while (future == null);
334 
335         pipeline.remove("connector");
336 
337         return future;
338     }

340     @ChannelPipelineCoverage("one")
341     private final class ConnectionlessBinder extends SimpleChannelUpstreamHandler {
342 
343         private final SocketAddress localAddress;
344         private final BlockingQueue<ChannelFuture> futureQueue;
345 
346         ConnectionlessBinder(SocketAddress localAddress, BlockingQueue<ChannelFuture> futureQueue) {
347             this.localAddress = localAddress;
348             this.futureQueue = futureQueue;
349         }
350 
351         @Override
352         public void channelOpen(
353                 ChannelHandlerContext ctx,
354                 ChannelStateEvent evt) {
355 
356             try {
357                 evt.getChannel().getConfig().setPipelineFactory(getPipelineFactory());
358 
359                 // Apply options.
360                 evt.getChannel().getConfig().setOptions(getOptions());
361             } finally {
362                 ctx.sendUpstream(evt);
363             }
364 
365             boolean finished = futureQueue.offer(evt.getChannel().bind(localAddress));
366             assert finished;
367         }
368 
369         @Override
370         public void exceptionCaught(
371                 ChannelHandlerContext ctx, ExceptionEvent e)
372                 throws Exception {
373             boolean finished = futureQueue.offer(failedFuture(e.getChannel(), e.getCause()));
374             assert finished;
375             ctx.sendUpstream(e);
376         }
377     }
