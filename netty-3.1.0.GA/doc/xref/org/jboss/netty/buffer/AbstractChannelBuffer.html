115     public void discardReadBytes() {

178     public void setZero(int index, int length) {
179         if (length == 0) {
180             return;
181         }
182         if (length < 0) {
183             throw new IllegalArgumentException(
184                     "length must be 0 or greater than 0.");
185         }
186 
187         int nLong = length >>> 3;
188         int nBytes = length & 7;
189         for (int i = nLong; i > 0; i --) {
190             setLong(index, 0);
191             index += 8;
192         }
193         if (nBytes == 4) {
194             setInt(index, 0);
195         } else if (nBytes < 4) {
196             for (int i = nBytes; i > 0; i --) {
197                 setByte(index, (byte) 0);
198                 index ++;
199             }
200         } else {
201             setInt(index, 0);
202             index += 4;
203             for (int i = nBytes - 4; i > 0; i --) {
204                 setByte(index, (byte) 0);
205                 index ++;
206             }
207         }
208     }

439     public void writeZero(int length) {
440         if (length == 0) {
441             return;
442         }
443         if (length < 0) {
444             throw new IllegalArgumentException(
445                     "length must be 0 or greater than 0.");
446         }
447         int nLong = length >>> 3;
448         int nBytes = length & 7;
449         for (int i = nLong; i > 0; i --) {
450             writeLong(0);
451         }
452         if (nBytes == 4) {
453             writeInt(0);
454         } else if (nBytes < 4) {
455             for (int i = nBytes; i > 0; i --) {
456                 writeByte((byte) 0);
457             }
458         } else {
459             writeInt(0);
460             for (int i = nBytes - 4; i > 0; i --) {
461                 writeByte((byte) 0);
462             }
463         }
464     }
