56      public CompositeChannelBuffer(ChannelBuffer... buffers) {
57          if (buffers.length == 0) {
58              throw new IllegalArgumentException("buffers should not be empty.");
59          }
60  
61          ByteOrder expectedEndianness = null;
62          for (ChannelBuffer buffer : buffers) {
63              if (buffer.capacity() != 0) {
64                  expectedEndianness = buffer.order();
65              }
66          }
67  
68          if (expectedEndianness == null) {
69              throw new IllegalArgumentException("buffers have only empty buffers.");
70          }
71  
72          order = expectedEndianness;
73          slices = new ChannelBuffer[buffers.length];
74          for (int i = 0; i < buffers.length; i ++) {
75              if (buffers[i].capacity() != 0 && buffers[i].order() != expectedEndianness) {
76                  throw new IllegalArgumentException(
77                          "All buffers must have the same endianness.");
78              }
79              slices[i] = buffers[i].slice();
80          }
81          indices = new int[buffers.length + 1];
82          for (int i = 1; i <= buffers.length; i ++) {
83              indices[i] = indices[i - 1] + slices[i - 1].capacity();
84          }
85          writerIndex(capacity());
86      }
87  
88      private CompositeChannelBuffer(CompositeChannelBuffer buffer) {
89          order = buffer.order;
90          slices = buffer.slices.clone();
91          indices = buffer.indices.clone();
92          setIndex(buffer.readerIndex(), buffer.writerIndex());
93      }

112     public short getShort(int index) {
113         int sliceId = sliceId(index);
114         if (index + 2 <= indices[sliceId + 1]) {
115             return slices[sliceId].getShort(index - indices[sliceId]);
116         } else if (order() == ByteOrder.BIG_ENDIAN) {
117             return (short) ((getByte(index) & 0xff) << 8 | getByte(index + 1) & 0xff);
118         } else {
119             return (short) (getByte(index) & 0xff | (getByte(index + 1) & 0xff) << 8);
120         }
121     }

156     public void getBytes(int index, byte[] dst, int dstIndex, int length) {
157         int sliceId = sliceId(index);
158         if (index > capacity() - length || dstIndex > dst.length - length) {
159             throw new IndexOutOfBoundsException();
160         }
161 
162         int i = sliceId;
163         while (length > 0) {
164             ChannelBuffer s = slices[i];
165             int adjustment = indices[i];
166             int localLength = Math.min(length, s.capacity() - (index - adjustment));
167             s.getBytes(index - adjustment, dst, dstIndex, localLength);
168             index += localLength;
169             dstIndex += localLength;
170             length -= localLength;
171             i ++;
172         }
173     }

175     public void getBytes(int index, ByteBuffer dst) {
176         int sliceId = sliceId(index);
177         int limit = dst.limit();
178         int length = dst.remaining();
179         if (index > capacity() - length) {
180             throw new IndexOutOfBoundsException();
181         }
182 
183         int i = sliceId;
184         try {
185             while (length > 0) {
186                 ChannelBuffer s = slices[i];
187                 int adjustment = indices[i];
188                 int localLength = Math.min(length, s.capacity() - (index - adjustment));
189                 dst.limit(dst.position() + localLength);
190                 s.getBytes(index - adjustment, dst);

193                 i ++;
194             }
195         } finally {
196             dst.limit(limit);
197         }

366     public int setBytes(int index, InputStream in, int length)
367             throws IOException {
368         int sliceId = sliceId(index);
369         if (index > capacity() - length) {
370             throw new IndexOutOfBoundsException();
371         }
372 
373         int i = sliceId;
374         int readBytes = 0;
375 
376         do {
377             ChannelBuffer s = slices[i];
378             int adjustment = indices[i];
379             int localLength = Math.min(length, s.capacity() - (index - adjustment));
380             int localReadBytes = s.setBytes(index - adjustment, in, localLength);
381             if (localReadBytes < 0) {
382                 if (readBytes == 0) {
383                     return -1;
384                 } else {
385                     break;
386                 }
387             }
388 
389             if (localReadBytes == localLength) {
390                 index += localLength;
391                 length -= localLength;
392                 readBytes += localLength;
393                 i ++;
394             } else {
395                 index += localReadBytes;
396                 length -= localReadBytes;
397                 readBytes += localReadBytes;
398             }
399         } while (length > 0);
400 
401         return readBytes;
402     }

451     private void copyTo(int index, int length, int sliceId, ChannelBuffer dst) {
452         int dstIndex = 0;
453         int i = sliceId;

455         while (length > 0) {

457             int adjustment = indices[i];

469     public ChannelBuffer slice(int index, int length) {
470         if (index == 0) {
471             if (length == 0) {
472                 return ChannelBuffers.EMPTY_BUFFER;
473             } else {
474                 return new TruncatedChannelBuffer(this, length);
475             }
476         } else if (index < 0 || index > capacity() - length) {
477             throw new IndexOutOfBoundsException();
478         } else if (length == 0) {
479             return ChannelBuffers.EMPTY_BUFFER;
480         } else {
481             return new SlicedChannelBuffer(this, index, length);
482         }
483     }

485     public ByteBuffer toByteBuffer(int index, int length) {
486         if (slices.length == 1) {
487             return slices[0].toByteBuffer(index, length);
488         }
489 
490         ByteBuffer[] buffers = toByteBuffers(index, length);
491         ByteBuffer merged = ByteBuffer.allocate(length).order(order());
492         for (ByteBuffer b: buffers) {
493             merged.put(b);
494         }
495         merged.flip();
496         return merged;
497     }

499     @Override
500     public ByteBuffer[] toByteBuffers(int index, int length) {
501         int sliceId = sliceId(index);
502         if (index + length > capacity()) {
503             throw new IndexOutOfBoundsException();
504         }
505 
506         List<ByteBuffer> buffers = new ArrayList<ByteBuffer>(slices.length);
507 
508         int i = sliceId;
509         while (length > 0) {
510             ChannelBuffer s = slices[i];
511             int adjustment = indices[i];
512             int localLength = Math.min(length, s.capacity() - (index - adjustment));
513             buffers.add(s.toByteBuffer(index - adjustment, localLength));
514             index += localLength;
515             length -= localLength;
516             i ++;
517         }
518 
519         return buffers.toArray(new ByteBuffer[buffers.size()]);
520     }

522     public String toString(int index, int length, String charsetName) {
523         int sliceId = sliceId(index);
524         if (index + length <= indices[sliceId + 1]) {
525             return slices[sliceId].toString(
526                     index - indices[sliceId], length, charsetName);
527         }
528 
529         byte[] data = new byte[length];
530         int dataIndex = 0;
531         int i = sliceId;
532 
533         while (length > 0) {
534             ChannelBuffer s = slices[i];
535             int adjustment = indices[i];
536             int localLength = Math.min(length, s.capacity() - (index - adjustment));
537             s.getBytes(index - adjustment, data, dataIndex, localLength);
538             index += localLength;
539             dataIndex += localLength;
540             length -= localLength;
541             i ++;
542         }
543 
544         try {
545             return new String(data, charsetName);
546         } catch (UnsupportedEncodingException e) {
547             throw new UnsupportedCharsetException(charsetName);
548         }
549     }

551     private int sliceId(int index) {
552         int lastSliceId = this.lastSliceId;
553         if (index >= indices[lastSliceId]) {
554             if (index < indices[lastSliceId + 1]) {
555                 return lastSliceId;
556             }
557 
558             // Search right
559             for (int i = lastSliceId + 1; i < slices.length; i ++) {
560                 if (index < indices[i + 1]) {
561                     this.lastSliceId = i;
562                     return i;
563                 }
564             }
565         } else {
566             // Search left
567             for (int i = lastSliceId - 1; i >= 0; i --) {
568                 if (index >= indices[i]) {
569                     this.lastSliceId = i;
570                     return i;
571                 }
572             }
573         }
574 
575         throw new IndexOutOfBoundsException();
576     }
