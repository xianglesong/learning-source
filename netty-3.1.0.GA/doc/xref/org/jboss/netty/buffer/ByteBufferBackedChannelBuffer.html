37   * A NIO {@link ByteBuffer} based buffer.  It is recommended to use {@link ChannelBuffers#directBuffer(int)}
38   * and {@link ChannelBuffers#wrappedBuffer(ByteBuffer)} instead of calling the
39   * constructor explicitly.

52      /**
53       * Creates a new buffer which wraps the specified buffer's slice.
54       */
55      public ByteBufferBackedChannelBuffer(ByteBuffer buffer) {
56          if (buffer == null) {
57              throw new NullPointerException("buffer");
58          }
59  
60          this.buffer = buffer.slice().order(buffer.order());
61          capacity = buffer.remaining();
62          writerIndex(capacity);
63      }
64  
65      private ByteBufferBackedChannelBuffer(ByteBufferBackedChannelBuffer buffer) {
66          this.buffer = buffer.buffer;
67          capacity = buffer.capacity;
68          setIndex(buffer.readerIndex(), buffer.writerIndex());
69      }

109     public void getBytes(int index, ChannelBuffer dst, int dstIndex, int length) {
110         if (dst instanceof ByteBufferBackedChannelBuffer) {
111             ByteBufferBackedChannelBuffer bbdst = (ByteBufferBackedChannelBuffer) dst;
112             ByteBuffer data = bbdst.buffer.duplicate();
113 
114             data.limit(dstIndex + length).position(dstIndex);
115             getBytes(index, data);
116         } else if (buffer.hasArray()) {
117             dst.setBytes(dstIndex, buffer.array(), index + buffer.arrayOffset(), length);
118         } else {
119             dst.setBytes(dstIndex, this, index, length);
120         }

192     public void getBytes(int index, OutputStream out, int length) throws IOException {
193         if (length == 0) {
194             return;
195         }
196 
197         if (!buffer.isReadOnly() && buffer.hasArray()) {
198             out.write(
199                     buffer.array(),
200                     index + buffer.arrayOffset(),
201                     length);
202         } else {
203             byte[] tmp = new byte[length];
204             ((ByteBuffer) buffer.duplicate().position(index)).get(tmp);
205             out.write(tmp);

217     public int setBytes(int index, InputStream in, int length)
218             throws IOException {
219 
220         int readBytes = 0;
221 
222         if (!buffer.isReadOnly() && buffer.hasArray()) {
223             index += buffer.arrayOffset();
224             do {
225                 int localReadBytes = in.read(buffer.array(), index, length);
226                 if (localReadBytes < 0) {
227                     if (readBytes == 0) {
228                         return -1;
229                     } else {
230                         break;
231                     }
232                 }
233                 readBytes += localReadBytes;
234                 index += localReadBytes;
235                 length -= localReadBytes;
236             } while (length > 0);
237         } else {
238             byte[] tmp = new byte[length];
239             int i = 0;
240             do {
241                 int localReadBytes = in.read(tmp, i, tmp.length - i);
242                 if (localReadBytes < 0) {
243                     if (readBytes == 0) {
244                         return -1;
245                     } else {
246                         break;
247                     }
248                 }
249                 readBytes += localReadBytes;
250                 i += readBytes;
251             } while (i < tmp.length);
252             ((ByteBuffer) buffer.duplicate().position(index)).put(tmp);
253         }
254 
255         return readBytes;
256     }

258     public int setBytes(int index, ScatteringByteChannel in, int length)
259             throws IOException {
260 
261         ByteBuffer slice = (ByteBuffer) buffer.duplicate().limit(index + length).position(index);
262         int readBytes = 0;
263 
264         while (readBytes < length) {
265             int localReadBytes;
266             try {
267                 localReadBytes = in.read(slice);
268             } catch (ClosedChannelException e) {
269                 localReadBytes = -1;
270             }
271             if (localReadBytes < 0) {
272                 if (readBytes == 0) {
273                     return -1;
274                 } else {
275                     return readBytes;
276                 }
277             } else if (localReadBytes == 0) {
278                 break;
279             }
280             readBytes += localReadBytes;
281         }
282 
283         return readBytes;
284     }

286     public ByteBuffer toByteBuffer(int index, int length) {
287         if (index == 0 && length == capacity()) {
288             return buffer.duplicate().order(order());
289         } else {
290             return ((ByteBuffer) buffer.duplicate().position(
291                     index).limit(index + length)).slice().order(order());
292         }
293     }

295     public String toString(int index, int length, String charsetName) {
296         if (!buffer.isReadOnly() && buffer.hasArray()) {
297             try {
298                 return new String(
299                         buffer.array(), index + buffer.arrayOffset(), length,
300                         charsetName);
301             } catch (UnsupportedEncodingException e) {
302                 throw new UnsupportedCharsetException(charsetName);
303             }
304         } else {
305             byte[] tmp = new byte[length];
306             ((ByteBuffer) buffer.duplicate().position(index)).get(tmp);
307             try {
308                 return new String(tmp, charsetName);
309             } catch (UnsupportedEncodingException e) {
310                 throw new UnsupportedCharsetException(charsetName);
311             }
312         }
313     }

315     public ChannelBuffer slice(int index, int length) {
316         if (index == 0 && length == capacity()) {
317             return duplicate();
318         } else {
319             if (index >= 0 && length == 0) {
320                 return ChannelBuffers.EMPTY_BUFFER;
321             }
322             return new ByteBufferBackedChannelBuffer(
323                     ((ByteBuffer) buffer.duplicate().position(
324                             index).limit(index + length)).order(order()));
325         }
326     }
327 
328     public ChannelBuffer duplicate() {
329         return new ByteBufferBackedChannelBuffer(this);
330     }

332     public ChannelBuffer copy(int index, int length) {
333         ByteBuffer src;
334         try {
335             src = (ByteBuffer) buffer.duplicate().position(index).limit(index + length);
336         } catch (IllegalArgumentException e) {
337             throw new IndexOutOfBoundsException();
338         }
339 
340         ByteBuffer dst = buffer.isDirect() ? ByteBuffer.allocateDirect(length) : ByteBuffer.allocate(length);
341         dst.put(src);
342         dst.order(order());
343         dst.clear();
344         return new ByteBufferBackedChannelBuffer(dst);
345     }
