44  /**
45   * A {@link Timer} optimized for approximated I/O timeout scheduling.
46   *
47   * <h3>Tick Duration</h3>
48   *
49   * As described with 'approximated', this timer does not execute the scheduled
50   * {@link TimerTask} on time.  {@link HashedWheelTimer}, on every tick, will
51   * check if there are any {@link TimerTask}s behind the schedule and execute
52   * them.
53   * <p>
54   * You can increase or decrease the accuracy of the execution timing by
55   * specifying smaller or larger tick duration in the constructor.  In most
56   * network applications, I/O timeout does not need to be accurate.  Therefore,
57   * the default tick duration is 100 milliseconds and you will not need to try
58   * different configurations in most cases.
59   *
60   * <h3>Ticks per Wheel (Wheel Size)</h3>
61   *
62   * {@link HashedWheelTimer} maintains a data structure called 'wheel'.
63   * To put simply, a wheel is a hash table of {@link TimerTask}s whose hash
64   * function is 'dead line of the task'.  The default number of ticks per wheel
65   * (i.e. the size of the wheel) is 512.  You could specify a larger value
66   * if you are going to schedule a lot of timeouts.
67   *
68   * <h3>Implementation Details</h3>
69   *
70   * {@link HashedWheelTimer} is based on
71   * <a href="http://cseweb.ucsd.edu/users/varghese/>George Varghese</a> and
72   * Tony Lauck's paper,
73   * <a href="http://www-cse.ucsd.edu/users/varghese/PAPERS/twheel.ps.Z">'Hashed
74   * and Hierarchical Timing Wheels: data structures to efficiently implement a
75   * timer facility'</a>.  More comprehensive slides are located
76   * <a href="http://www.cse.wustl.edu/~cdgill/courses/cs6874/TimingWheels.ppt">here</a>.
77   *
78   * @author The Netty Project (netty-dev@lists.jboss.org)
79   * @author Trustin Lee (tlee@redhat.com)
80   * @version $Rev: 1397 $, $Date: 2009-06-17 15:37:36 +0900 (Wed, 17 Jun 2009) $
81   */

164     /**
165      * Creates a new timer.
166      *
167      * @param threadFactory  a {@link ThreadFactory} that creates a
168      *                       background {@link Thread} which is dedicated to
169      *                       {@link TimerTask} execution.
170      * @param tickDuration   the duration between tick
171      * @param unit           the time unit of the {@code tickDuration}
172      * @param ticksPerWheel  the size of the wheel
173      */
174     public HashedWheelTimer(
175             ThreadFactory threadFactory,
176             long tickDuration, TimeUnit unit, int ticksPerWheel) {
177 
178         if (threadFactory == null) {
179             throw new NullPointerException("threadFactory");
180         }
181         if (unit == null) {
182             throw new NullPointerException("unit");
183         }
184         if (tickDuration <= 0) {
185             throw new IllegalArgumentException(
186                     "tickDuration must be greater than 0: " + tickDuration);
187         }
188         if (ticksPerWheel <= 0) {
189             throw new IllegalArgumentException(
190                     "ticksPerWheel must be greater than 0: " + ticksPerWheel);
191         }
192 
193         // Normalize ticksPerWheel to power of two and initialize the wheel.
194         wheel = createWheel(ticksPerWheel);
195         iterators = createIterators(wheel);
196         mask = wheel.length - 1;
197 
198         // Convert tickDuration to milliseconds.
199         this.tickDuration = tickDuration = unit.toMillis(tickDuration);
200 
201         // Prevent overflow.
202         if (tickDuration == Long.MAX_VALUE ||
203                 tickDuration >= Long.MAX_VALUE / wheel.length) {
204             throw new IllegalArgumentException(
205                     "tickDuration is too long: " +
206                     tickDuration +  ' ' + unit);
207         }
208 
209         roundDuration = tickDuration * wheel.length;
210 
211         workerThread = threadFactory.newThread(new ThreadRenamingRunnable(
212                         worker, "Hashed wheel timer #" + id.incrementAndGet()));
213 
214         // Misuse check
215         int activeInstances = HashedWheelTimer.activeInstances.incrementAndGet();
216         if (activeInstances >= MISUSE_WARNING_THRESHOLD &&
217             loggedMisuseWarning.compareAndSet(false, true)) {
218             logger.debug(
219                     "There are too many active " +
220                     HashedWheelTimer.class.getSimpleName() + " instances (" +
221                     activeInstances + ") - you should share the small number " +
222                     "of instances to avoid excessive resource consumption.");
223         }
224     }

226     @SuppressWarnings("unchecked")
227     private static Set<HashedWheelTimeout>[] createWheel(int ticksPerWheel) {
228         if (ticksPerWheel <= 0) {
229             throw new IllegalArgumentException(
230                     "ticksPerWheel must be greater than 0: " + ticksPerWheel);
231         }
232         if (ticksPerWheel > 1073741824) {
233             throw new IllegalArgumentException(
234                     "ticksPerWheel may not be greater than 2^30: " + ticksPerWheel);
235         }
236 
237         ticksPerWheel = normalizeTicksPerWheel(ticksPerWheel);
238         Set<HashedWheelTimeout>[] wheel = new Set[ticksPerWheel];
239         for (int i = 0; i < wheel.length; i ++) {
240             wheel[i] = new MapBackedSet<HashedWheelTimeout>(
241                     new ConcurrentIdentityHashMap<HashedWheelTimeout, Boolean>(16, 0.95f, 4));
242         }
243         return wheel;
244     }
245 
246     @SuppressWarnings("unchecked")
247     private static ReusableIterator<HashedWheelTimeout>[] createIterators(Set<HashedWheelTimeout>[] wheel) {
248         ReusableIterator<HashedWheelTimeout>[] iterators = new ReusableIterator[wheel.length];
249         for (int i = 0; i < wheel.length; i ++) {
250             iterators[i] = (ReusableIterator<HashedWheelTimeout>) wheel[i].iterator();
251         }
252         return iterators;
253     }

255     private static int normalizeTicksPerWheel(int ticksPerWheel) {
256         int normalizedTicksPerWheel = 1;
257         while (normalizedTicksPerWheel < ticksPerWheel) {
258             normalizedTicksPerWheel <<= 1;
259         }
260         return normalizedTicksPerWheel;
261     }

276             workerThread.start();
277         }

280     public synchronized Set<Timeout> stop() {
281         if (!shutdown.compareAndSet(false, true)) {
282             return Collections.emptySet();
283         }
284 
285         while (workerThread.isAlive()) {
286             workerThread.interrupt();
287             try {
288                 workerThread.join(100);
289             } catch (InterruptedException e) {
290                 // Ignore
291             }
292         }
293 
294         activeInstances.decrementAndGet();
295 
296         Set<Timeout> unprocessedTimeouts = new HashSet<Timeout>();
297         for (Set<HashedWheelTimeout> bucket: wheel) {
298             unprocessedTimeouts.addAll(bucket);
299             bucket.clear();
300         }
301 
302         return Collections.unmodifiableSet(unprocessedTimeouts);
303     }

305     public Timeout newTimeout(TimerTask task, long delay, TimeUnit unit) {
306         final long currentTime = System.currentTimeMillis();
307 
308         if (task == null) {
309             throw new NullPointerException("task");
310         }
311         if (unit == null) {
312             throw new NullPointerException("unit");
313         }
314 
315         delay = unit.toMillis(delay);
316         if (delay < tickDuration) {
317             delay = tickDuration;
318         }
319 
320         if (!workerThread.isAlive()) {
321             start();
322         }
323 
324         // Prepare the required parameters to create the timeout object.
325         HashedWheelTimeout timeout;
326         final long lastRoundDelay = delay % roundDuration;
327         final long lastTickDelay = delay % tickDuration;
328         final long relativeIndex =
329             lastRoundDelay / tickDuration + (lastTickDelay != 0? 1 : 0);
330         final long deadline = currentTime + delay;
331 
332         final long remainingRounds =
333             delay / roundDuration - (delay % roundDuration == 0? 1 : 0);
334 
335         // Add the timeout to the wheel.
336         lock.readLock().lock();
337         try {
338             timeout =
339                 new HashedWheelTimeout(
340                         task, deadline,
341                         (int) (wheelCursor + relativeIndex & mask),
342                         remainingRounds);
343 
344             wheel[timeout.stopIndex].add(timeout);
345         } finally {
346             lock.readLock().unlock();
347         }
348 
349         return timeout;
350     }

352     private final class Worker implements Runnable {
353 
354         private long startTime;
355         private long tick;
356 
357         Worker() {
358             super();
359         }
360 
361         public void run() {
362             List<HashedWheelTimeout> expiredTimeouts =
363                 new ArrayList<HashedWheelTimeout>();
364 
365             startTime = System.currentTimeMillis();
366             tick = 1;
367 
368             while (!shutdown.get()) {
369                 waitForNextTick();
370                 fetchExpiredTimeouts(expiredTimeouts);
371                 notifyExpiredTimeouts(expiredTimeouts);
372             }
373         }
374 
375         private void fetchExpiredTimeouts(
376                 List<HashedWheelTimeout> expiredTimeouts) {
377 
378             // Find the expired timeouts and decrease the round counter
379             // if necessary.  Note that we don't send the notification
380             // immediately to make sure the listeners are called without
381             // an exclusive lock.
382             lock.writeLock().lock();
383             try {
384                 int oldBucketHead = wheelCursor;
385                 int newBucketHead = oldBucketHead + 1 & mask;
386                 wheelCursor = newBucketHead;
387 
388                 ReusableIterator<HashedWheelTimeout> i = iterators[oldBucketHead];
389                 fetchExpiredTimeouts(expiredTimeouts, i);
390             } finally {
391                 lock.writeLock().unlock();
392             }
393         }
394 
395         private void fetchExpiredTimeouts(
396                 List<HashedWheelTimeout> expiredTimeouts,
397                 ReusableIterator<HashedWheelTimeout> i) {
398 
399             long currentDeadline = System.currentTimeMillis() + tickDuration;
400             i.rewind();
401             while (i.hasNext()) {
402                 HashedWheelTimeout timeout = i.next();
403                 if (timeout.remainingRounds <= 0) {
404                     if (timeout.deadline < currentDeadline) {
405                         i.remove();
406                         expiredTimeouts.add(timeout);
407                     } else {
408                         // A rare case where a timeout is put for the next
409                         // round: just wait for the next round.
410                     }
411                 } else {
412                     timeout.remainingRounds --;
413                 }
414             }
415         }
416 
417         private void notifyExpiredTimeouts(
418                 List<HashedWheelTimeout> expiredTimeouts) {
419             // Notify the expired timeouts.
420             for (int i = expiredTimeouts.size() - 1; i >= 0; i --) {
421                 expiredTimeouts.get(i).expire();
422             }
423 
424             // Clean up the temporary list.
425             expiredTimeouts.clear();
426         }
427 
428         private void waitForNextTick() {
429             for (;;) {
430                 final long currentTime = System.currentTimeMillis();
431                 final long sleepTime = tickDuration * tick - (currentTime - startTime);
432 
433                 if (sleepTime <= 0) {
434                     break;
435                 }
436 
437                 try {
438                     Thread.sleep(sleepTime);
439                 } catch (InterruptedException e) {
440                     if (shutdown.get()) {
441                         return;
442                     }
443                 }
444             }
445 
446             // Reset the tick if overflow is expected.
447             if (tickDuration * tick > Long.MAX_VALUE - tickDuration) {
448                 startTime = System.currentTimeMillis();
449                 tick = 1;
450             } else {
451                 // Increase the tick if overflow is not likely to happen.
452                 tick ++;
453             }
454         }
455     }

457     private final class HashedWheelTimeout implements Timeout {
458 
459         private final TimerTask task;
460         final int stopIndex;
461         final long deadline;
462         volatile long remainingRounds;
463         private volatile boolean cancelled;
464 
465         HashedWheelTimeout(
466                 TimerTask task, long deadline, int stopIndex, long remainingRounds) {
467             this.task = task;
468             this.deadline = deadline;
469             this.stopIndex = stopIndex;
470             this.remainingRounds = remainingRounds;
471         }
472 
473         public Timer getTimer() {
474             return HashedWheelTimer.this;
475         }
476 
477         public TimerTask getTask() {
478             return task;
479         }
480 
481         public void cancel() {
482             if (isExpired()) {
483                 return;
484             }
485 
486             cancelled = true;
487 
488             // Might be called more than once, but doesn't matter.
489             wheel[stopIndex].remove(this);
490         }
491 
492         public boolean isCancelled() {
493             return cancelled;
494         }
495 
496         public boolean isExpired() {
497             return cancelled || System.currentTimeMillis() > deadline;
498         }
499 
500         public void expire() {
501             if (cancelled) {
502                 return;
503             }
504 
505             try {
506                 task.run(this);
507             } catch (Throwable t) {
508                 logger.warn(
509                         "An exception was thrown by " +
510                         TimerTask.class.getSimpleName() + ".", t);
511             }
512         }
513 
514         @Override
515         public String toString() {
516             long currentTime = System.currentTimeMillis();
517             long remaining = deadline - currentTime;
518 
519             StringBuilder buf = new StringBuilder(192);
520             buf.append(getClass().getSimpleName());
521             buf.append('(');
522 
523             buf.append("deadline: ");
524             if (remaining > 0) {
525                 buf.append(remaining);
526                 buf.append(" ms later, ");
527             } else if (remaining < 0) {
528                 buf.append(-remaining);
529                 buf.append(" ms ago, ");
530             } else {
531                 buf.append("now, ");
532             }
533 
534             if (isCancelled()) {
535                 buf.append (", cancelled");
536             }
537 
538             return buf.append(')').toString();
539         }
