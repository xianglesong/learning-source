42  /**
43   * An unbounded <tt>TransferQueue</tt> based on linked nodes.
44   * This queue orders elements FIFO (first-in-first-out) with respect
45   * to any given producer.  The <em>head</em> of the queue is that
46   * element that has been on the queue the longest time for some
47   * producer.  The <em>tail</em> of the queue is that element that has
48   * been on the queue the shortest time for some producer.
49   *
50   * <p>Beware that, unlike in most collections, the {@code size}
51   * method is <em>NOT</em> a constant-time operation. Because of the
52   * asynchronous nature of these queues, determining the current number
53   * of elements requires a traversal of the elements.
54   *
55   * <p>This class and its iterator implement all of the
56   * <em>optional</em> methods of the {@link Collection} and {@link
57   * Iterator} interfaces.
58   *
59   * <p>Memory consistency effects: As with other concurrent
60   * collections, actions in a thread prior to placing an object into a
61   * {@code LinkedTransferQueue}
62   * <a href="package-summary.html#MemoryVisibility"><i>happen-before</i></a>
63   * actions subsequent to the access or removal of that element from
64   * the {@code LinkedTransferQueue} in another thread.
65   *
66   * @author Doug Lea
67   * @author The Netty Project (netty-dev@lists.jboss.org)
68   * @author Trustin Lee (tlee@redhat.com)
69   *
70   * @param <E> the type of elements held in this collection
71   *
72   */
73  public class LinkedTransferQueue<E> extends AbstractQueue<E> implements BlockingQueue<E> {
74  
75      /*
76       * This class extends the approach used in FIFO-mode
77       * SynchronousQueues. See the internal documentation, as well as
78       * the PPoPP 2006 paper "Scalable Synchronous Queues" by Scherer,
79       * Lea & Scott
80       * (http://www.cs.rice.edu/~wns1/papers/2006-PPoPP-SQ.pdf)
81       *
82       * The main extension is to provide different Wait modes for the
83       * main "xfer" method that puts or takes items.  These don't
84       * impact the basic dual-queue logic, but instead control whether
85       * or how threads block upon insertion of request or data nodes
86       * into the dual queue. It also uses slightly different
87       * conventions for tracking whether nodes are off-list or
88       * cancelled.
89       */
90  
91      // Wait modes for xfer method
92      private static final int NOWAIT  = 0;
93      private static final int TIMEOUT = 1;
94      private static final int WAIT    = 2;
95  
96      /** The number of CPUs, for spin control */
97      private static final int NCPUS = Runtime.getRuntime().availableProcessors();
98  
99      /**
100      * The number of times to spin before blocking in timed waits.
101      * The value is empirically derived -- it works well across a
102      * variety of processors and OSes. Empirically, the best value
103      * seems not to vary with number of CPUs (beyond 2) so is just
104      * a constant.
105      */
106     private static final int maxTimedSpins = NCPUS < 2? 0 : 32;
107 
108     /**
109      * The number of times to spin before blocking in untimed waits.
110      * This is greater than timed value because untimed waits spin
111      * faster since they don't need to check times on each spin.
112      */
113     private static final int maxUntimedSpins = maxTimedSpins * 16;
114 
115     /**
116      * The number of nanoseconds for which it is faster to spin
117      * rather than to use timed park. A rough estimate suffices.
118      */
119     private static final long spinForTimeoutThreshold = 1000L;

121     /**
122      * Node class for LinkedTransferQueue. Opportunistically
123      * subclasses from AtomicReference to represent item. Uses Object,
124      * not E, to allow setting item to "this" after use, to avoid
125      * garbage retention. Similarly, setting the next field to this is
126      * used as sentinel that node is off list.
127      */
128     private static final class QNode extends AtomicReference<Object> {
129         private static final long serialVersionUID = 5925596372370723938L;
130 
131         transient volatile QNode next;
132         transient volatile Thread waiter;       // to control park/unpark
133         final boolean isData;
134         QNode(Object item, boolean isData) {
135             super(item);
136             this.isData = isData;
137         }
138 
139         private static final AtomicReferenceFieldUpdater<QNode, QNode> nextUpdater;
140         static {
141             AtomicReferenceFieldUpdater<QNode, QNode> tmp = null;
142             try {
143                 tmp = AtomicReferenceFieldUpdater.newUpdater(
144                         QNode.class, QNode.class, "next");
145 
146                 // Test if AtomicReferenceFieldUpdater is really working.
147                 QNode testNode = new QNode(null, false);
148                 tmp.set(testNode, testNode);
149                 if (testNode.next != testNode) {
150                     // Not set as expected - fall back to the safe mode.
151                     throw new Exception();
152                 }
153             } catch (Throwable t) {
154                 // Running in a restricted environment with a security manager.
155                 tmp = null;
156             }
157             nextUpdater = tmp;
158         }
159 
160         final boolean casNext(QNode cmp, QNode val) {
161             if (nextUpdater != null) {
162                 return nextUpdater.compareAndSet(this, cmp, val);
163             } else {
164                 return alternativeCasNext(cmp, val);
165             }
166         }
167 
168         private synchronized final boolean alternativeCasNext(QNode cmp, QNode val) {
169             if (next == cmp) {
170                 next = val;
171                 return true;
172             } else {
173                 return false;
174             }
175         }
176 
177         final void clearNext() {
178             // nextUpdater.lazySet(this, this);
179             next = this; // allows run on java5
180         }
181     }

183     /**
184      * Padded version of AtomicReference used for head, tail and
185      * cleanMe, to alleviate contention across threads CASing one vs
186      * the other.
187      */
188     private static final class PaddedAtomicReference<T> extends AtomicReference<T> {
189         private static final long serialVersionUID = 4684288940772921317L;
190 
191         // enough padding for 64bytes with 4byte refs
192         @SuppressWarnings("unused")
193         Object p0, p1, p2, p3, p4, p5, p6, p7, p8, p9, pa, pb, pc, pd, pe;
194         PaddedAtomicReference(T r) { super(r); }
195     }

210     /**
211      * Tries to cas nh as new head; if successful, unlink
212      * old head's next node to avoid garbage retention.
213      */
214     private boolean advanceHead(QNode h, QNode nh) {
215         if (h == head.get() && head.compareAndSet(h, nh)) {
216             h.clearNext(); // forget old next
217             return true;
218         }
219         return false;
220     }

222     /**
223      * Puts or takes an item. Used for most queue operations (except
224      * poll() and tryTransfer()). See the similar code in
225      * SynchronousQueue for detailed explanation.
226      *
227      * @param e the item or if null, signifies that this is a take
228      * @param mode the wait mode: NOWAIT, TIMEOUT, WAIT
229      * @param nanos timeout in nanosecs, used only if mode is TIMEOUT
230      * @return an item, or null on failure
231      */
232     private Object xfer(Object e, int mode, long nanos) {
233         boolean isData = e != null;
234         QNode s = null;
235         final PaddedAtomicReference<QNode> head = this.head;
236         final PaddedAtomicReference<QNode> tail = this.tail;
237 
238         for (;;) {
239             QNode t = tail.get();
240             QNode h = head.get();
241 
242             if (t != null && (t == h || t.isData == isData)) {
243                 if (s == null) {
244                     s = new QNode(e, isData);
245                 }
246                 QNode last = t.next;
247                 if (last != null) {
248                     if (t == tail.get()) {
249                         tail.compareAndSet(t, last);
250                     }
251                 }
252                 else if (t.casNext(null, s)) {
253                     tail.compareAndSet(t, s);
254                     return awaitFulfill(t, s, e, mode, nanos);
255                 }
256             }
257 
258             else if (h != null) {
259                 QNode first = h.next;
260                 if (t == tail.get() && first != null &&
261                     advanceHead(h, first)) {
262                     Object x = first.get();
263                     if (x != first && first.compareAndSet(x, e)) {
264                         LockSupport.unpark(first.waiter);
265                         return isData? e : x;
266                     }
267                 }
268             }
269         }
270     }

273     /**
274      * Version of xfer for poll() and tryTransfer, which
275      * simplifies control paths both here and in xfer.
276      */
277     private Object fulfill(Object e) {
278         boolean isData = e != null;
279         final PaddedAtomicReference<QNode> head = this.head;
280         final PaddedAtomicReference<QNode> tail = this.tail;
281 
282         for (;;) {
283             QNode t = tail.get();
284             QNode h = head.get();
285 
286             if (t != null && (t == h || t.isData == isData)) {
287                 QNode last = t.next;
288                 if (t == tail.get()) {
289                     if (last != null) {
290                         tail.compareAndSet(t, last);
291                     } else {
292                         return null;
293                     }
294                 }
295             }
296             else if (h != null) {
297                 QNode first = h.next;
298                 if (t == tail.get() &&
299                     first != null &&
300                     advanceHead(h, first)) {
301                     Object x = first.get();
302                     if (x != first && first.compareAndSet(x, e)) {
303                         LockSupport.unpark(first.waiter);
304                         return isData? e : x;
305                     }
306                 }
307             }
308         }
309     }

311     /**
312      * Spins/blocks until node s is fulfilled or caller gives up,
313      * depending on wait mode.
314      *
315      * @param pred the predecessor of waiting node
316      * @param s the waiting node
317      * @param e the comparison value for checking match
318      * @param mode mode
319      * @param nanos timeout value
320      * @return matched item, or s if cancelled
321      */
322     private Object awaitFulfill(QNode pred, QNode s, Object e,
323                                 int mode, long nanos) {
324         if (mode == NOWAIT) {
325             return null;
326         }
327 
328         long lastTime = mode == TIMEOUT? System.nanoTime() : 0;
329         Thread w = Thread.currentThread();
330         int spins = -1; // set to desired spin count below
331         for (;;) {
332             if (w.isInterrupted()) {
333                 s.compareAndSet(e, s);
334             }
335             Object x = s.get();
336             if (x != e) {                 // Node was matched or cancelled
337                 advanceHead(pred, s);     // unlink if head
338                 if (x == s) {              // was cancelled
339                     clean(pred, s);
340                     return null;
341                 }
342                 else if (x != null) {
343                     s.set(s);             // avoid garbage retention
344                     return x;
345                 } else {
346                     return e;
347                 }
348             }
349             if (mode == TIMEOUT) {
350                 long now = System.nanoTime();
351                 nanos -= now - lastTime;
352                 lastTime = now;
353                 if (nanos <= 0) {
354                     s.compareAndSet(e, s); // try to cancel
355                     continue;
356                 }
357             }
358             if (spins < 0) {
359                 QNode h = head.get(); // only spin if at head
360                 spins = h != null && h.next == s ?
361                          (mode == TIMEOUT?
362                           maxTimedSpins : maxUntimedSpins) : 0;
363             }
364             if (spins > 0) {
365                 --spins;
366             } else if (s.waiter == null) {
367                 s.waiter = w;
368             } else if (mode != TIMEOUT) {
369                 //                LockSupport.park(this);
370                 LockSupport.park(); // allows run on java5
371                 s.waiter = null;
372                 spins = -1;
373             }
374             else if (nanos > spinForTimeoutThreshold) {
375                 //                LockSupport.parkNanos(this, nanos);
376                 LockSupport.parkNanos(nanos);
377                 s.waiter = null;
378                 spins = -1;
379             }
380         }
381     }

383     /**
384      * Returns validated tail for use in cleaning methods.
385      */
386     private QNode getValidatedTail() {
387         for (;;) {
388             QNode h = head.get();
389             QNode first = h.next;
390             if (first != null && first.next == first) { // help advance
391                 advanceHead(h, first);
392                 continue;
393             }
394             QNode t = tail.get();
395             QNode last = t.next;
396             if (t == tail.get()) {
397                 if (last != null) {
398                     tail.compareAndSet(t, last); // help advance
399                 } else {
400                     return t;
401                 }
402             }
403         }
404     }

406     /**
407      * Gets rid of cancelled node s with original predecessor pred.
408      *
409      * @param pred predecessor of cancelled node
410      * @param s the cancelled node
411      */
412     void clean(QNode pred, QNode s) {
413         Thread w = s.waiter;
414         if (w != null) {             // Wake up thread
415             s.waiter = null;
416             if (w != Thread.currentThread()) {
417                 LockSupport.unpark(w);
418             }
419         }
420 
421         if (pred == null) {
422             return;
423         }
424 
425         /*
426          * At any given time, exactly one node on list cannot be
427          * deleted -- the last inserted node. To accommodate this, if
428          * we cannot delete s, we save its predecessor as "cleanMe",
429          * processing the previously saved version first. At least one
430          * of node s or the node previously saved can always be
431          * processed, so this always terminates.
432          */
433         while (pred.next == s) {
434             QNode oldpred = reclean();  // First, help get rid of cleanMe
435             QNode t = getValidatedTail();
436             if (s != t) {               // If not tail, try to unsplice
437                 QNode sn = s.next;      // s.next == s means s already off list
438                 if (sn == s || pred.casNext(s, sn)) {
439                     break;
440                 }
441             }
442             else if (oldpred == pred || // Already saved
443                      oldpred == null && cleanMe.compareAndSet(null, pred)) {
444                 break;                  // Postpone cleaning
445             }
446         }
447     }

449     /**
450      * Tries to unsplice the cancelled node held in cleanMe that was
451      * previously uncleanable because it was at tail.
452      *
453      * @return current cleanMe node (or null)
454      */
455     private QNode reclean() {
456         /*
457          * cleanMe is, or at one time was, predecessor of cancelled
458          * node s that was the tail so could not be unspliced.  If s
459          * is no longer the tail, try to unsplice if necessary and
460          * make cleanMe slot available.  This differs from similar
461          * code in clean() because we must check that pred still
462          * points to a cancelled node that must be unspliced -- if
463          * not, we can (must) clear cleanMe without unsplicing.
464          * This can loop only due to contention on casNext or
465          * clearing cleanMe.
466          */
467         QNode pred;
468         while ((pred = cleanMe.get()) != null) {
469             QNode t = getValidatedTail();
470             QNode s = pred.next;
471             if (s != t) {
472                 QNode sn;
473                 if (s == null || s == pred || s.get() != s ||
474                     (sn = s.next) == s || pred.casNext(s, sn)) {
475                     cleanMe.compareAndSet(pred, null);
476                 }
477             } else {
478                 break;
479             }
480         }
481         return pred;
482     }

604     public int drainTo(Collection<? super E> c) {
605         if (c == null) {
606             throw new NullPointerException();
607         }
608         if (c == this) {
609             throw new IllegalArgumentException();
610         }

638     /**
639      * Returns head after performing any outstanding helping steps.
640      */
641     QNode traversalHead() {
642         for (;;) {
643             QNode t = tail.get();
644             QNode h = head.get();
645             if (h != null && t != null) {
646                 QNode last = t.next;
647                 QNode first = h.next;
648                 if (t == tail.get()) {
649                     if (last != null) {
650                         tail.compareAndSet(t, last);
651                     } else if (first != null) {
652                         Object x = first.get();
653                         if (x == first) {
654                             advanceHead(h, first);
655                         } else {
656                             return h;
657                         }
658                     } else {
659                         return h;
660                     }
661                 }
662             }
663             reclean();
664         }
665     }

669         return new Itr();

672     /**
673      * Iterators. Basic strategy is to traverse list, treating
674      * non-data (i.e., request) nodes as terminating list.
675      * Once a valid data node is found, the item is cached
676      * so that the next call to next() will return it even
677      * if subsequently removed.
678      */
679     class Itr implements Iterator<E> {
680         QNode next;        // node to return next
681         QNode pnext;       // predecessor of next
682         QNode snext;       // successor of next
683         QNode curr;        // last returned node, for remove()
684         QNode pcurr;       // predecessor of curr, for remove()
685         E nextItem;        // Cache of next item, once commited to in next
686 
687         Itr() {
688             findNext();
689         }
690 
691         /**
692          * Ensure next points to next valid node, or null if none.
693          */
694         void findNext() {
695             for (;;) {
696                 QNode pred = pnext;
697                 QNode q = next;
698                 if (pred == null || pred == q) {
699                     pred = traversalHead();
700                     q = pred.next;
701                 }
702                 if (q == null || !q.isData) {
703                     next = null;
704                     return;
705                 }
706                 Object x = q.get();
707                 QNode s = q.next;
708                 if (x != null && q != x && q != s) {
709                     nextItem = cast(x);
710                     snext = s;
711                     pnext = pred;
712                     next = q;
713                     return;
714                 }
715                 pnext = q;
716                 next = s;
717             }
718         }
719 
720         public boolean hasNext() {
721             return next != null;
722         }
723 
724         public E next() {
725             if (next == null) {
726                 throw new NoSuchElementException();
727             }
728             pcurr = pnext;
729             curr = next;
730             pnext = next;
731             next = snext;
732             E x = nextItem;
733             findNext();
734             return x;
735         }
736 
737         public void remove() {
738             QNode p = curr;
739             if (p == null) {
740                 throw new IllegalStateException();
741             }
742             Object x = p.get();
743             if (x != null && x != p && p.compareAndSet(x, p)) {
744                 clean(pcurr, p);
745             }
746         }
747     }

749     public E peek() {
750         for (;;) {
751             QNode h = traversalHead();
