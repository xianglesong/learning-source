23  package org.jboss.netty.util.internal;
24  
25  import org.jboss.netty.channel.ChannelFuture;
26  
27  /**
28   * @author Trustin Heui-seung Lee (trustin@gmail.com)
29   * @version $Rev: 1483 $, $Date: 2009-06-20 02:54:42 +0900 (Sat, 20 Jun 2009) $
30   */
31  public class IoWorkerRunnable implements Runnable {
32  
33      /**
34       * An <em>internal use only</em> thread-local variable that determines if
35       * the caller is running on an I/O worker thread, which is the case where
36       * the caller enters a dead lock if the caller calls
37       * {@link ChannelFuture#await()} or {@link ChannelFuture#awaitUninterruptibly()}.
38       */
39      public static final ThreadLocal<Boolean> IN_IO_THREAD = new ThreadLocalBoolean();
40  
41      private final Runnable runnable;
42  
43      public IoWorkerRunnable(Runnable runnable) {
44          if (runnable == null) {
45              throw new NullPointerException("runnable");
46          }
47          this.runnable = runnable;
48      }
49  
50      public void run() {
51          IN_IO_THREAD.set(Boolean.TRUE);
52          try {
53              runnable.run();
54          } finally {
55              IN_IO_THREAD.remove();
56          }
57      }
58  }
