23  package org.jboss.netty.example.proxy;
24  
25  import java.net.InetSocketAddress;
26  import java.util.concurrent.Executor;
27  import java.util.concurrent.Executors;
28  
29  import org.jboss.netty.bootstrap.ServerBootstrap;
30  import org.jboss.netty.channel.socket.ClientSocketChannelFactory;
31  import org.jboss.netty.channel.socket.nio.NioClientSocketChannelFactory;
32  import org.jboss.netty.channel.socket.nio.NioServerSocketChannelFactory;
33  
34  /**
35   * @author The Netty Project (netty-dev@lists.jboss.org)
36   * @author Trustin Lee (tlee@redhat.com)
37   * @version $Rev: 1605 $, $Date: 2009-07-23 18:05:53 +0900 (Thu, 23 Jul 2009) $
38   */
39  public class HexDumpProxy {
40  
41      public static void main(String[] args) throws Exception {
42          // Validate command line options.
43          if (args.length != 3) {
44              System.err.println(
45                      "Usage: " + HexDumpProxy.class.getSimpleName() +
46                      " <local port> <remote host> <remote port>");
47              return;
48          }
49  
50          // Parse command line options.
51          int localPort = Integer.parseInt(args[0]);
52          String remoteHost = args[1];
53          int remotePort = Integer.parseInt(args[2]);
54  
55          System.err.println(
56                  "Proxying *:" + localPort + " to " +
57                  remoteHost + ':' + remotePort + " ...");
58  
59          // Configure the bootstrap.
60          Executor executor = Executors.newCachedThreadPool();
61          ServerBootstrap sb = new ServerBootstrap(
62                  new NioServerSocketChannelFactory(executor, executor));
63  
64          // Set up the event pipeline factory.
65          ClientSocketChannelFactory cf =
66                  new NioClientSocketChannelFactory(executor, executor);
67  
68          sb.setPipelineFactory(
69                  new HexDumpProxyPipelineFactory(cf, remoteHost, remotePort));
70  
71          // Start up the server.
72          sb.bind(new InetSocketAddress(localPort));
73      }
74  }
