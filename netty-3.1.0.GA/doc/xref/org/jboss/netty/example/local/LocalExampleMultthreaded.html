23  package org.jboss.netty.example.local;
24  
25  import java.util.concurrent.TimeUnit;
26  
27  import org.jboss.netty.bootstrap.ClientBootstrap;
28  import org.jboss.netty.bootstrap.ServerBootstrap;
29  import org.jboss.netty.channel.ChannelFuture;
30  import org.jboss.netty.channel.local.DefaultLocalClientChannelFactory;
31  import org.jboss.netty.channel.local.DefaultLocalServerChannelFactory;
32  import org.jboss.netty.channel.local.LocalAddress;
33  import org.jboss.netty.handler.codec.string.StringDecoder;
34  import org.jboss.netty.handler.codec.string.StringEncoder;
35  import org.jboss.netty.handler.execution.OrderedMemoryAwareThreadPoolExecutor;
36  import org.jboss.netty.handler.logging.LoggingHandler;
37  import org.jboss.netty.logging.InternalLogLevel;
38  
39  /**
40   * @author The Netty Project (netty-dev@lists.jboss.org)
41   * @author Andy Taylor (andy.taylor@jboss.org)
42   * @author Frederic Bregier (fredbregier@free.fr)
43   * @version $Rev: 1605 $, $Date: 2009-07-23 18:05:53 +0900 (Thu, 23 Jul 2009) $
44   */
45  public class LocalExampleMultthreaded {
46  
47      public static void main(String[] args) throws Exception {
48          LocalAddress socketAddress = new LocalAddress("1");
49  
50          OrderedMemoryAwareThreadPoolExecutor eventExecutor =
51              new OrderedMemoryAwareThreadPoolExecutor(
52                      5, 1000000, 10000000, 100,
53                      TimeUnit.MILLISECONDS);
54  
55          ServerBootstrap sb = new ServerBootstrap(
56                  new DefaultLocalServerChannelFactory());
57  
58          sb.setPipelineFactory(new LocalServerPipelineFactory(eventExecutor));
59          sb.bind(socketAddress);
60  
61          ClientBootstrap cb = new ClientBootstrap(
62                  new DefaultLocalClientChannelFactory());
63  
64          cb.getPipeline().addLast("decoder", new StringDecoder());
65          cb.getPipeline().addLast("encoder", new StringEncoder());
66          cb.getPipeline().addLast("handler", new LoggingHandler(InternalLogLevel.INFO));
67  
68          // Read commands from array
69          String[] commands = { "First", "Second", "Third", "quit" };
70          for (int j = 0; j < 5 ; j++) {
71              System.err.println("Start "+j);
72              ChannelFuture channelFuture = cb.connect(socketAddress);
73              channelFuture.awaitUninterruptibly();
74              if (! channelFuture.isSuccess()) {
75                  System.err.println("CANNOT CONNECT");
76                  channelFuture.getCause().printStackTrace();
77                  break;
78              }
79              ChannelFuture lastWriteFuture = null;
80              for (String line: commands) {
81                  // Sends the received line to the server.
82                  lastWriteFuture = channelFuture.getChannel().write(line);
83              }
84  
85              // Wait until all messages are flushed before closing the channel.
86              if (lastWriteFuture != null) {
87                  lastWriteFuture.awaitUninterruptibly();
88              }
89              channelFuture.getChannel().close();
90              // Wait until the connection is closed or the connection attempt fails.
91              channelFuture.getChannel().getCloseFuture().awaitUninterruptibly();
92              System.err.println("End "+j);
93          }
94  
95          // Release all resources
96          cb.releaseExternalResources();
97          sb.releaseExternalResources();
98          eventExecutor.shutdownNow();
99      }
