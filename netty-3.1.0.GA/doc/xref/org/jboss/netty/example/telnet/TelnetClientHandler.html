23  package org.jboss.netty.example.telnet;
24  
25  import java.util.logging.Level;
26  import java.util.logging.Logger;
27  
28  import org.jboss.netty.channel.ChannelEvent;
29  import org.jboss.netty.channel.ChannelHandlerContext;
30  import org.jboss.netty.channel.ChannelPipelineCoverage;
31  import org.jboss.netty.channel.ChannelStateEvent;
32  import org.jboss.netty.channel.ExceptionEvent;
33  import org.jboss.netty.channel.MessageEvent;
34  import org.jboss.netty.channel.SimpleChannelUpstreamHandler;
35  
36  /**
37   * Handles a client-side channel.
38   *
39   * @author The Netty Project (netty-dev@lists.jboss.org)
40   * @author Trustin Lee (tlee@redhat.com)
41   *
42   * @version $Rev: 1211 $, $Date: 2009-04-17 16:33:32 +0900 (Fri, 17 Apr 2009) $
43   */
44  @ChannelPipelineCoverage("all")
45  public class TelnetClientHandler extends SimpleChannelUpstreamHandler {
46  
47      private static final Logger logger = Logger.getLogger(
48              TelnetClientHandler.class.getName());
49  
50      @Override
51      public void handleUpstream(
52              ChannelHandlerContext ctx, ChannelEvent e) throws Exception {
53          if (e instanceof ChannelStateEvent) {
54              logger.info(e.toString());
55          }
56          super.handleUpstream(ctx, e);
57      }
58  
59      @Override
60      public void messageReceived(
61              ChannelHandlerContext ctx, MessageEvent e) {
62          // Print out the line received from the server.
63          System.err.println(e.getMessage());
64      }
65  
66      @Override
67      public void exceptionCaught(
68              ChannelHandlerContext ctx, ExceptionEvent e) {
69          logger.log(
70                  Level.WARNING,
71                  "Unexpected exception from downstream.",
72                  e.getCause());
73          e.getChannel().close();
74      }
75  }
