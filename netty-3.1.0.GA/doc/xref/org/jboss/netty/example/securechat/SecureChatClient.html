23  package org.jboss.netty.example.securechat;
24  
25  import java.io.BufferedReader;
26  import java.io.InputStreamReader;
27  import java.net.InetSocketAddress;
28  import java.util.concurrent.Executors;
29  
30  import org.jboss.netty.bootstrap.ClientBootstrap;
31  import org.jboss.netty.channel.Channel;
32  import org.jboss.netty.channel.ChannelFuture;
33  import org.jboss.netty.channel.socket.nio.NioClientSocketChannelFactory;
34  import org.jboss.netty.example.telnet.TelnetClient;
35  
36  /**
37   * Simple SSL chat client modified from {@link TelnetClient}.
38   *
39   * @author The Netty Project (netty-dev@lists.jboss.org)
40   * @author Trustin Lee (tlee@redhat.com)
41   *
42   * @version $Rev: 1605 $, $Date: 2009-07-23 18:05:53 +0900 (Thu, 23 Jul 2009) $
43   *
44   */
45  public class SecureChatClient {
46  
47      public static void main(String[] args) throws Exception {
48          // Print usage if no argument is specified.
49          if (args.length != 2) {
50              System.err.println(
51                      "Usage: " + SecureChatClient.class.getSimpleName() +
52                      " <host> <port>");
53              return;
54          }
55  
56          // Parse options.
57          String host = args[0];
58          int port = Integer.parseInt(args[1]);
59  
60          // Configure the client.
61          ClientBootstrap bootstrap = new ClientBootstrap(
62                  new NioClientSocketChannelFactory(
63                          Executors.newCachedThreadPool(),
64                          Executors.newCachedThreadPool()));
65  
66          SecureChatClientHandler handler = new SecureChatClientHandler();
67          bootstrap.setPipelineFactory(new SecureChatPipelineFactory(handler));
68  
69          // Start the connection attempt.
70          ChannelFuture future = bootstrap.connect(new InetSocketAddress(host, port));
71  
72          // Wait until the connection attempt succeeds or fails.
73          Channel channel = future.awaitUninterruptibly().getChannel();
74          if (!future.isSuccess()) {
75              future.getCause().printStackTrace();
76              bootstrap.releaseExternalResources();
77              return;
78          }
79  
80          // Read commands from the stdin.
81          ChannelFuture lastWriteFuture = null;
82          BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
83          for (;;) {
84              String line = in.readLine();
85              if (line == null) {
86                  break;
87              }
88  
89              // Sends the received line to the server.
90              lastWriteFuture = channel.write(line + '\n');
91  
92              // If user typed the 'bye' command, wait until the server closes
93              // the connection.
94              if (line.toLowerCase().equals("bye")) {
95                  channel.getCloseFuture().awaitUninterruptibly();
96                  break;
97              }
98          }
99  
100         // Wait until all messages are flushed before closing the channel.
101         if (lastWriteFuture != null) {
102             lastWriteFuture.awaitUninterruptibly();
103         }
104 
105         // Close the connection.  Make sure the close operation ends because
106         // all I/O operations are asynchronous in Netty.
107         channel.close().awaitUninterruptibly();
108 
109         // Shut down all thread pools to exit.
110         bootstrap.releaseExternalResources();
111     }
112 }
