23  package org.jboss.netty.example.factorial;
24  
25  import static org.jboss.netty.channel.Channels.*;
26  
27  import org.jboss.netty.channel.ChannelPipeline;
28  import org.jboss.netty.channel.ChannelPipelineFactory;
29  
30  /**
31   * Creates a newly configured {@link ChannelPipeline} for a client-side channel.
32   *
33   * @author The Netty Project (netty-dev@lists.jboss.org)
34   * @author Trustin Lee (tlee@redhat.com)
35   *
36   * @version $Rev: 55 $, $Date: 2008-08-12 20:58:15 +0900 (Tue, 12 Aug 2008) $
37   */
38  public class FactorialClientPipelineFactory implements
39          ChannelPipelineFactory {
40  
41      private final int count;
42  
43      public FactorialClientPipelineFactory(int count) {
44          this.count = count;
45      }
46  
47      public ChannelPipeline getPipeline() throws Exception {
48          ChannelPipeline pipeline = pipeline();
49  
50          // Add the number codec first,
51          pipeline.addLast("decoder", new BigIntegerDecoder());
52          pipeline.addLast("encoder", new NumberEncoder());
53  
54          // and then business logic.
55          pipeline.addLast("handler", new FactorialClientHandler(count));
56  
57          return pipeline;
58      }
59  }
