23  package org.jboss.netty.channel.socket.http;

25  import org.jboss.netty.channel.ChannelPipeline;
26  import org.jboss.netty.channel.ChannelSink;
27  import org.jboss.netty.channel.socket.ClientSocketChannelFactory;
28  import org.jboss.netty.channel.socket.SocketChannel;

30  /**
31   * Creates a client-side {@link SocketChannel} which connects to an
32   * {@link HttpTunnelingServlet} to communicate with the server application
33   * behind the {@link HttpTunnelingServlet}.  Please refer to the
34   * <a href="package-summary.html#package_description">package summary</a> for
35   * the detailed usage.
36   *
37   * @author The Netty Project (netty-dev@lists.jboss.org)
38   * @author Andy Taylor (andy.taylor@jboss.org)
39   * @author Trustin Lee (tlee@redhat.com)
40   * @version $Rev: 1588 $, $Date: 2009-07-20 14:05:18 +0900 (Mon, 20 Jul 2009) $
41   *
42   * @apiviz.landmark
43   */
44  public class HttpTunnelingClientSocketChannelFactory implements ClientSocketChannelFactory {
45  
46      private final ChannelSink sink = new HttpTunnelingClientSocketPipelineSink();
47      private final ClientSocketChannelFactory clientSocketChannelFactory;
48  
49      /**
50       * Creates a new instance.
51       */
52      public HttpTunnelingClientSocketChannelFactory(ClientSocketChannelFactory clientSocketChannelFactory) {
53          this.clientSocketChannelFactory = clientSocketChannelFactory;
54      }
55  
56      public SocketChannel newChannel(ChannelPipeline pipeline) {
57          return new HttpTunnelingClientSocketChannel(this, pipeline, sink, clientSocketChannelFactory);
58      }
59  
60      public void releaseExternalResources() {
61          clientSocketChannelFactory.releaseExternalResources();
62      }
63  }
