56  class NioProviderMetadata {
57      static final InternalLogger logger =
58          InternalLoggerFactory.getInstance(NioProviderMetadata.class);

60      private static final String CONSTRAINT_LEVEL_PROPERTY =
61          "java.nio.channels.spi.constraintLevel";
62  
63      /**
64       * 0 - no need to wake up to get / set interestOps (most cases)
65       * 1 - no need to wake up to get interestOps, but need to wake up to set.
66       * 2 - need to wake up to get / set interestOps    (old providers)
67       */
68      static final int CONSTRAINT_LEVEL;
69  
70      static {
71          int constraintLevel = -1;
72  
73          // Use the system property if possible.
74          try {
75              String value = SystemPropertyUtil.get(CONSTRAINT_LEVEL_PROPERTY, "-1");
76              constraintLevel = Integer.parseInt(value);
77              if (constraintLevel < 0 || constraintLevel > 2) {
78                  constraintLevel = -1;
79              } else {
80                  logger.debug(
81                          "Using the specified NIO constraint level: " +
82                          constraintLevel);
83              }
84          } catch (Exception e) {
85              // format error
86          }
87  
88          if (constraintLevel < 0) {
89              constraintLevel = detectConstraintLevelFromSystemProperties();
90  
91              if (constraintLevel < 0) {
92                  constraintLevel = 2;
93                  logger.debug(
94                          "Couldn't determine the NIO constraint level from " +
95                          "the system properties; using the safest level (2)");
96              } else if (constraintLevel != 0) {
97                  logger.info(
98                          "Using the autodetected NIO constraint level: " +
99                          constraintLevel +
100                         " (Use better NIO provider for better performance)");
101             } else {
102                 logger.debug(
103                         "Using the autodetected NIO constraint level: " +
104                         constraintLevel);
105             }
106         }
107 
108         CONSTRAINT_LEVEL = constraintLevel;
109 
110         if (CONSTRAINT_LEVEL < 0 || CONSTRAINT_LEVEL > 2) {
111             throw new Error(
112                     "Unexpected NIO constraint level: " +
113                     CONSTRAINT_LEVEL + ", please report this error.");
114         }
115     }

117     private static int detectConstraintLevelFromSystemProperties() {
118         String version = SystemPropertyUtil.get("java.specification.version");
119         String vminfo = SystemPropertyUtil.get("java.vm.info", "");
120         String os = SystemPropertyUtil.get("os.name");
121         String vendor = SystemPropertyUtil.get("java.vm.vendor");
122         String provider;
123         try {
124             provider = SelectorProvider.provider().getClass().getName();
125         } catch (Exception e) {
126             // Perhaps security exception.
127             provider = null;
128         }
129 
130         if (version == null || os == null || vendor == null || provider == null) {
131             return -1;
132         }
133 
134         os = os.toLowerCase();
135         vendor = vendor.toLowerCase();
136 
137 //        System.out.println(version);
138 //        System.out.println(vminfo);
139 //        System.out.println(os);
140 //        System.out.println(vendor);
141 //        System.out.println(provider);
142 
143         // Sun JVM
144         if (vendor.indexOf("sun") >= 0) {
145             // Linux
146             if (os.indexOf("linux") >= 0) {
147                 if (provider.equals("sun.nio.ch.EPollSelectorProvider") ||
148                     provider.equals("sun.nio.ch.PollSelectorProvider")) {
149                     return 0;
150                 }
151 
152             // Windows
153             } else if (os.indexOf("windows") >= 0) {
154                 if (provider.equals("sun.nio.ch.WindowsSelectorProvider")) {
155                     return 0;
156                 }
157 
158             // Solaris
159             } else if (os.indexOf("sun") >= 0 || os.indexOf("solaris") >= 0) {
160                 if (provider.equals("sun.nio.ch.DevPollSelectorProvider")) {
161                     return 0;
162                 }
163             }
164         // Apple JVM
165         } else if (vendor.indexOf("apple") >= 0) {
166             // Mac OS
167             if (os.indexOf("mac") >= 0 && os.indexOf("os") >= 0) {
168                 if (provider.equals("sun.nio.ch.KQueueSelectorProvider")) {
169                     return 0;
170                 }
171             }
172         // IBM
173         } else if (vendor.indexOf("ibm") >= 0) {
174             // Linux or AIX
175             if (os.indexOf("linux") >= 0 || os.indexOf("aix") >= 0) {
176                 if (version.equals("1.5") || version.matches("^1\\.5\\D.*$")) {
177                     if (provider.equals("sun.nio.ch.PollSelectorProvider")) {
178                         return 1;
179                     }
180                 } else if (version.equals("1.6") || version.matches("^1\\.6\\D.*$")) {
181                     // IBM JDK 1.6 has different constraint level for different
182                     // version.  The exact version can be determined only by its
183                     // build date.
184                     Pattern datePattern = Pattern.compile(
185                             "(?:^|[^0-9])(" +
186                             "[2-9][0-9]{3}" +              // year
187                             "(?:0[1-9]|1[0-2])" +          // month
188                             "(?:0[1-9]|[12][0-9]|3[01])" + // day of month
189                             ")(?:$|[^0-9])");
190 
191                     Matcher dateMatcher = datePattern.matcher(vminfo);
192                     if (dateMatcher.find()) {
193                         long dateValue = Long.parseLong(dateMatcher.group(1));
194                         if (dateValue < 20081105L) {
195                             // SR0, 1, and 2
196                             return 2;
197                         } else {
198                             // SR3 and later
199                             if (provider.equals("sun.nio.ch.EPollSelectorProvider")) {
200                                 return 0;
201                             } else if (provider.equals("sun.nio.ch.PollSelectorProvider")) {
202                                 return 1;
203                             }
204                         }
205                     }
206                 }
207             }
208         // BEA
209         } else if (vendor.indexOf("bea") >= 0 || vendor.indexOf("oracle") >= 0) {
210             // Linux
211             if (os.indexOf("linux") >= 0) {
212                 if (provider.equals("sun.nio.ch.EPollSelectorProvider") ||
213                     provider.equals("sun.nio.ch.PollSelectorProvider")) {
214                     return 0;
215                 }
216 
217             // Windows
218             } else if (os.indexOf("windows") >= 0) {
219                 if (provider.equals("sun.nio.ch.WindowsSelectorProvider")) {
220                     return 0;
221                 }
222             }
223         // Apache Software Foundation
224         } else if (vendor.indexOf("apache") >= 0) {
225             if (provider.equals("org.apache.harmony.nio.internal.SelectorProviderImpl")) {
226                 return 1;
227             }
228         }
229 
230         // Others (untested)
231         return -1;
232     }

234     private static final class ConstraintLevelAutodetector {
235 
236         ConstraintLevelAutodetector() {
237             super();
238         }
239 
240         int autodetectWithTimeout() {
241             final BlockingQueue<Integer> resultQueue = new LinkedBlockingQueue<Integer>();
242             Runnable detector = new ThreadRenamingRunnable(new Runnable() {
243                 public void run() {
244                     int level = -1;
245                     try {
246                         level = autodetectWithoutTimeout();
247                     } finally {
248                         boolean finished = resultQueue.offer(Integer.valueOf(level));
249                         assert finished;
250                     }
251                 }
252             }, "NIO constraint level detector");
253 
254             Thread detectorThread = new Thread(detector);
255             try {
256                 detectorThread.setDaemon(true);
257             } catch (Exception e) {
258                 logger.warn(
259                         "Failed to set the daemon flag of the autodetector thread.", e);
260             }
261             detectorThread.start();
262 
263             for (;;) {
264                 try {
265                     Integer result = resultQueue.poll(10000, TimeUnit.MILLISECONDS);
266                     if (result == null) {
267                         logger.warn("NIO constraint level autodetection timed out.");
268                         return -1;
269                     } else {
270                         return result.intValue();
271                     }
272                 } catch (InterruptedException e) {
273                     // Ignored
274                 }
275             }
276         }
277 
278         int autodetectWithoutTimeout() {
279             final int constraintLevel;
280             ExecutorService executor = Executors.newCachedThreadPool();
281             boolean success;
282             long startTime;
283             int interestOps;
284 
285             ServerSocketChannel ch = null;
286             SelectorLoop loop = null;
287 
288             try {
289                 // Open a channel.
290                 ch = ServerSocketChannel.open();
291 
292                 // Configure the channel
293                 try {
294                     ch.socket().bind(new InetSocketAddress(0));
295                     ch.configureBlocking(false);
296                 } catch (Throwable e) {
297                     logger.warn("Failed to configure a temporary socket.", e);
298                     return -1;
299                 }
300 
301                 // Prepare the selector loop.
302                 try {
303                     loop = new SelectorLoop();
304                 } catch (Throwable e) {
305                     logger.warn("Failed to open a temporary selector.", e);
306                     return -1;
307                 }
308 
309                 // Register the channel
310                 try {
311                     ch.register(loop.selector, 0);
312                 } catch (Throwable e) {
313                     logger.warn("Failed to register a temporary selector.", e);
314                     return -1;
315                 }
316 
317                 SelectionKey key = ch.keyFor(loop.selector);
318 
319                 // Start the selector loop.
320                 executor.execute(loop);
321 
322                 // Level 0
323                 success = true;
324                 for (int i = 0; i < 10; i ++) {
325 
326                     // Increase the probability of calling interestOps
327                     // while select() is running.
328                     do {
329                         while (!loop.selecting) {
330                             Thread.yield();
331                         }
332 
333                         // Wait a little bit more.
334                         try {
335                             Thread.sleep(50);
336                         } catch (InterruptedException e) {
337                             // Ignore
338                         }
339                     } while (!loop.selecting);
340 
341                     startTime = System.nanoTime();
342                     key.interestOps(key.interestOps() | SelectionKey.OP_ACCEPT);
343                     key.interestOps(key.interestOps() & ~SelectionKey.OP_ACCEPT);
344 
345                     if (System.nanoTime() - startTime >= 500000000L) {
346                         success = false;
347                         break;
348                     }
349                 }
350 
351                 if (success) {
352                     constraintLevel = 0;
353                 } else {
354                     // Level 1
355                     success = true;
356                     for (int i = 0; i < 10; i ++) {
357 
358                         // Increase the probability of calling interestOps
359                         // while select() is running.
360                         do {
361                             while (!loop.selecting) {
362                                 Thread.yield();
363                             }
364 
365                             // Wait a little bit more.
366                             try {
367                                 Thread.sleep(50);
368                             } catch (InterruptedException e) {
369                                 // Ignore
370                             }
371                         } while (!loop.selecting);
372 
373                         startTime = System.nanoTime();
374                         interestOps = key.interestOps();
375                         synchronized (loop) {
376                             loop.selector.wakeup();
377                             key.interestOps(interestOps | SelectionKey.OP_ACCEPT);
378                             key.interestOps(interestOps & ~SelectionKey.OP_ACCEPT);
379                         }
380 
381                         if (System.nanoTime() - startTime >= 500000000L) {
382                             success = false;
383                             break;
384                         }
385                     }
386                     if (success) {
387                         constraintLevel = 1;
388                     } else {
389                         constraintLevel = 2;
390                     }
391                 }
392             } catch (Throwable e) {
393                 return -1;
394             } finally {
395                 if (ch != null) {
396                     try {
397                         ch.close();
398                     } catch (Throwable e) {
399                         logger.warn("Failed to close a temporary socket.", e);
400                     }
401                 }
402 
403                 if (loop != null) {
404                     loop.done = true;
405                     try {
406                         executor.shutdownNow();
407                     } catch (NullPointerException ex) {
408                         // Some JDK throws NPE here, but shouldn't.
409                     }
410 
411                     try {
412                         for (;;) {
413                             loop.selector.wakeup();
414                             try {
415                                 if (executor.awaitTermination(1, TimeUnit.SECONDS)) {
416                                     break;
417                                 }
418                             } catch (InterruptedException e) {
419                                 // Ignore
420                             }
421                         }
422                     } catch (Throwable e) {
423                         // Perhaps security exception.
424                     }
425 
426                     try {
427                         loop.selector.close();
428                     } catch (Throwable e) {
429                         logger.warn("Failed to close a temporary selector.", e);
430                     }
431                 }
432             }
433 
434             return constraintLevel;
435         }

438     private static final class SelectorLoop implements Runnable {
439         final Selector selector;
440         volatile boolean done;
441         volatile boolean selecting; // Just an approximation
442 
443         SelectorLoop() throws IOException {
444             selector = Selector.open();
445         }
446 
447         public void run() {
448             while (!done) {
449                 synchronized (this) {
450                     // Guard
451                 }
452                 try {
453                     selecting = true;
454                     try {
455                         selector.select(1000);
456                     } finally {
457                         selecting = false;
458                     }
459 
460                     Set<SelectionKey> keys = selector.selectedKeys();
461                     for (SelectionKey k: keys) {
462                         k.interestOps(0);
463                     }
464                     keys.clear();
465                 } catch (IOException e) {
466                     logger.warn("Failed to wait for a temporary selector.", e);
467                 }
468             }
469         }
470     }
