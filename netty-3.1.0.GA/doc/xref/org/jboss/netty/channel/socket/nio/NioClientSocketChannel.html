46  final class NioClientSocketChannel extends NioSocketChannel {
47  
48      private static final InternalLogger logger =
49          InternalLoggerFactory.getInstance(NioClientSocketChannel.class);
50  
51      private static SocketChannel newSocket() {
52          SocketChannel socket;
53          try {
54              socket = SocketChannel.open();
55          } catch (IOException e) {
56              throw new ChannelException("Failed to open a socket.", e);
57          }
58  
59          boolean success = false;
60          try {
61              socket.configureBlocking(false);
62              success = true;
63          } catch (IOException e) {
64              throw new ChannelException("Failed to enter non-blocking mode.", e);
65          } finally {
66              if (!success) {
67                  try {
68                      socket.close();
69                  } catch (IOException e) {
70                      logger.warn(
71                              "Failed to close a partially initialized socket.",
72                              e);
73                  }
74              }
75          }
76  
77          return socket;
78      }
79  
80      volatile ChannelFuture connectFuture;
81      volatile boolean boundManually;
82  
83      NioClientSocketChannel(
84              ChannelFactory factory, ChannelPipeline pipeline,
85              ChannelSink sink, NioWorker worker) {
86  
87          super(null, factory, pipeline, sink, newSocket(), worker);
88          fireChannelOpen(this);
89      }
