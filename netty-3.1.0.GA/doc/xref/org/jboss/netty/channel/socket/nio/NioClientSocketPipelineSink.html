65      static final InternalLogger logger =
66          InternalLoggerFactory.getInstance(NioClientSocketPipelineSink.class);
67      private static final AtomicInteger nextId = new AtomicInteger();

69      final int id = nextId.incrementAndGet();
70      final Executor bossExecutor;
71      private final Boss boss = new Boss();
72      private final NioWorker[] workers;
73      private final AtomicInteger workerIndex = new AtomicInteger();
74  
75      NioClientSocketPipelineSink(
76              Executor bossExecutor, Executor workerExecutor, int workerCount) {
77          this.bossExecutor = bossExecutor;
78          workers = new NioWorker[workerCount];
79          for (int i = 0; i < workers.length; i ++) {
80              workers[i] = new NioWorker(id, i + 1, workerExecutor);
81          }
82      }

84      public void eventSunk(
85              ChannelPipeline pipeline, ChannelEvent e) throws Exception {
86          if (e instanceof ChannelStateEvent) {
87              ChannelStateEvent event = (ChannelStateEvent) e;
88              NioClientSocketChannel channel =
89                  (NioClientSocketChannel) event.getChannel();
90              ChannelFuture future = event.getFuture();
91              ChannelState state = event.getState();
92              Object value = event.getValue();
93  
94              switch (state) {
95              case OPEN:
96                  if (Boolean.FALSE.equals(value)) {
97                      NioWorker.close(channel, future);
98                  }
99                  break;
100             case BOUND:
101                 if (value != null) {
102                     bind(channel, future, (SocketAddress) value);
103                 } else {
104                     NioWorker.close(channel, future);
105                 }
106                 break;
107             case CONNECTED:
108                 if (value != null) {
109                     connect(channel, future, (SocketAddress) value);
110                 } else {
111                     NioWorker.close(channel, future);
112                 }
113                 break;
114             case INTEREST_OPS:
115                 NioWorker.setInterestOps(channel, future, ((Integer) value).intValue());
116                 break;
117             }
118         } else if (e instanceof MessageEvent) {
119             MessageEvent event = (MessageEvent) e;
120             NioSocketChannel channel = (NioSocketChannel) event.getChannel();
121             boolean offered = channel.writeBuffer.offer(event);
122             assert offered;
123             NioWorker.write(channel, true);
124         }
125     }

127     private void bind(
128             NioClientSocketChannel channel, ChannelFuture future,
129             SocketAddress localAddress) {
130         try {
131             channel.socket.socket().bind(localAddress);
132             channel.boundManually = true;
133             future.setSuccess();
134             fireChannelBound(channel, channel.getLocalAddress());
135         } catch (Throwable t) {
136             future.setFailure(t);
137             fireExceptionCaught(channel, t);
138         }
139     }
140 
141     private void connect(
142             final NioClientSocketChannel channel, ChannelFuture future,
143             SocketAddress remoteAddress) {
144         try {
145             if (channel.socket.connect(remoteAddress)) {
146                 channel.worker.register(channel, future);
147             } else {
148                 future.addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
149                 channel.connectFuture = future;
150                 boss.register(channel);
151             }
152 
153         } catch (Throwable t) {
154             future.setFailure(t);
155             fireExceptionCaught(channel, t);
156         }
157     }

159     NioWorker nextWorker() {
160         return workers[Math.abs(
161                 workerIndex.getAndIncrement() % workers.length)];
162     }

164     private final class Boss implements Runnable {
165 
166         volatile Selector selector;
167         private boolean started;
168         private final AtomicBoolean wakenUp = new AtomicBoolean();
169         private final Object startStopLock = new Object();
170         private final Queue<Runnable> registerTaskQueue = new LinkedTransferQueue<Runnable>();
171 
172         Boss() {
173             super();
174         }
175 
176         void register(NioSocketChannel channel) {
177             Runnable registerTask = new RegisterTask(this, channel);
178             Selector selector;
179 
180             synchronized (startStopLock) {
181                 if (!started) {
182                     // Open a selector if this worker didn't start yet.
183                     try {
184                         this.selector = selector =  Selector.open();
185                     } catch (Throwable t) {
186                         throw new ChannelException(
187                                 "Failed to create a selector.", t);
188                     }
189 
190                     // Start the worker thread with the new Selector.
191                     boolean success = false;
192                     try {
193                         bossExecutor.execute(
194                                 new IoWorkerRunnable(
195                                         new ThreadRenamingRunnable(
196                                                 this, "New I/O client boss #" + id)));
197                         success = true;
198                     } finally {
199                         if (!success) {
200                             // Release the Selector if the execution fails.
201                             try {
202                                 selector.close();
203                             } catch (Throwable t) {
204                                 logger.warn("Failed to close a selector.", t);
205                             }
206                             this.selector = selector = null;
207                             // The method will return to the caller at this point.
208                         }
209                     }
210                 } else {
211                     // Use the existing selector if this worker has been started.
212                     selector = this.selector;
213                 }
214 
215                 assert selector != null && selector.isOpen();
216 
217                 started = true;
218                 boolean offered = registerTaskQueue.offer(registerTask);
219                 assert offered;
220             }
221 
222             if (wakenUp.compareAndSet(false, true)) {
223                 selector.wakeup();
224             }
225         }
226 
227         public void run() {
228             boolean shutdown = false;
229             Selector selector = this.selector;
230             for (;;) {
231                 wakenUp.set(false);
232 
233                 try {
234                     int selectedKeyCount = selector.select(500);
235 
236                     // 'wakenUp.compareAndSet(false, true)' is always evaluated
237                     // before calling 'selector.wakeup()' to reduce the wake-up
238                     // overhead. (Selector.wakeup() is an expensive operation.)
239                     //
240                     // However, there is a race condition in this approach.
241                     // The race condition is triggered when 'wakenUp' is set to
242                     // true too early.
243                     //
244                     // 'wakenUp' is set to true too early if:
245                     // 1) Selector is waken up between 'wakenUp.set(false)' and
246                     //    'selector.select(...)'. (BAD)
247                     // 2) Selector is waken up between 'selector.select(...)' and
248                     //    'if (wakenUp.get()) { ... }'. (OK)
249                     //
250                     // In the first case, 'wakenUp' is set to true and the
251                     // following 'selector.select(...)' will wake up immediately.
252                     // Until 'wakenUp' is set to false again in the next round,
253                     // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
254                     // any attempt to wake up the Selector will fail, too, causing
255                     // the following 'selector.select(...)' call to block
256                     // unnecessarily.
257                     //
258                     // To fix this problem, we wake up the selector again if wakenUp
259                     // is true immediately after selector.select(...).
260                     // It is inefficient in that it wakes up the selector for both
261                     // the first case (BAD - wake-up required) and the second case
262                     // (OK - no wake-up required).
263 
264                     if (wakenUp.get()) {
265                         selector.wakeup();
266                     }
267 
268                     processRegisterTaskQueue();
269 
270                     if (selectedKeyCount > 0) {
271                         processSelectedKeys(selector.selectedKeys());
272                     }
273 
274                     // Exit the loop when there's nothing to handle.
275                     // The shutdown flag is used to delay the shutdown of this
276                     // loop to avoid excessive Selector creation when
277                     // connection attempts are made in a one-by-one manner
278                     // instead of concurrent manner.
279                     if (selector.keys().isEmpty()) {
280                         if (shutdown ||
281                             bossExecutor instanceof ExecutorService && ((ExecutorService) bossExecutor).isShutdown()) {
282 
283                             synchronized (startStopLock) {
284                                 if (registerTaskQueue.isEmpty() && selector.keys().isEmpty()) {
285                                     started = false;
286                                     try {
287                                         selector.close();
288                                     } catch (IOException e) {
289                                         logger.warn(
290                                                 "Failed to close a selector.", e);
291                                     } finally {
292                                         this.selector = null;
293                                     }
294                                     break;
295                                 } else {
296                                     shutdown = false;
297                                 }
298                             }
299                         } else {
300                             // Give one more second.
301                             shutdown = true;
302                         }
303                     } else {
304                         shutdown = false;
305                     }
306                 } catch (Throwable t) {
307                     logger.warn(
308                             "Unexpected exception in the selector loop.", t);
309 
310                     // Prevent possible consecutive immediate failures.
311                     try {
312                         Thread.sleep(1000);
313                     } catch (InterruptedException e) {
314                         // Ignore.
315                     }
316                 }
317             }
318         }
319 
320         private void processRegisterTaskQueue() {
321             for (;;) {
322                 final Runnable task = registerTaskQueue.poll();
323                 if (task == null) {
324                     break;
325                 }
326 
327                 task.run();
328             }
329         }
330 
331         private void processSelectedKeys(Set<SelectionKey> selectedKeys) {
332             for (Iterator<SelectionKey> i = selectedKeys.iterator(); i.hasNext();) {
333                 SelectionKey k = i.next();
334                 i.remove();
335 
336                 if (!k.isValid()) {
337                     close(k);
338                     continue;
339                 }
340 
341                 if (k.isConnectable()) {
342                     connect(k);
343                 }
344             }
345         }
346 
347         private void connect(SelectionKey k) {
348             NioClientSocketChannel ch = (NioClientSocketChannel) k.attachment();
349             try {
350                 if (ch.socket.finishConnect()) {
351                     k.cancel();
352                     ch.worker.register(ch, ch.connectFuture);
353                 }
354             } catch (Throwable t) {
355                 ch.connectFuture.setFailure(t);
356                 fireExceptionCaught(ch, t);
357                 close(k);
358             }
359         }
360 
361         private void close(SelectionKey k) {
362             k.cancel();
363             NioSocketChannel ch = (NioSocketChannel) k.attachment();
364             NioWorker.close(ch, succeededFuture(ch));
365         }
366     }

368     private static final class RegisterTask implements Runnable {
369         private final Boss boss;
370         private final NioSocketChannel channel;
371 
372         RegisterTask(Boss boss, NioSocketChannel channel) {
373             this.boss = boss;
374             this.channel = channel;
375         }
376 
377         public void run() {
378             try {
379                 channel.socket.register(
380                         boss.selector, SelectionKey.OP_CONNECT, channel);
381             } catch (ClosedChannelException e) {
382                 throw new ChannelException(
383                         "Failed to register a socket to the selector.", e);
384             }
385         }
386     }
