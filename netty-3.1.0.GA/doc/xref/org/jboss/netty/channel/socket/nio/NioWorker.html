93      void register(NioSocketChannel channel, ChannelFuture future) {
94  
95          boolean server = !(channel instanceof NioClientSocketChannel);
96          Runnable registerTask = new RegisterTask(channel, future, server);
97          Selector selector;
98  
99          synchronized (startStopLock) {
100             if (!started) {
101                 // Open a selector if this worker didn't start yet.
102                 try {
103                     this.selector = selector = Selector.open();
104                 } catch (Throwable t) {
105                     throw new ChannelException(
106                             "Failed to create a selector.", t);
107                 }
108 
109                 // Start the worker thread with the new Selector.
110                 String threadName =
111                     (server ? "New I/O server worker #"
112                             : "New I/O client worker #") + bossId + '-' + id;
113 
114                 boolean success = false;
115                 try {
116                     executor.execute(
117                             new IoWorkerRunnable(
118                                     new ThreadRenamingRunnable(this, threadName)));
119                     success = true;
120                 } finally {
121                     if (!success) {
122                         // Release the Selector if the execution fails.
123                         try {
124                             selector.close();
125                         } catch (Throwable t) {
126                             logger.warn("Failed to close a selector.", t);
127                         }
128                         this.selector = selector = null;
129                         // The method will return to the caller at this point.
130                     }
131                 }
132             } else {
133                 // Use the existing selector if this worker has been started.
134                 selector = this.selector;
135             }
136 
137             assert selector != null && selector.isOpen();
138 
139             started = true;
140             boolean offered = registerTaskQueue.offer(registerTask);
141             assert offered;
142         }
143 
144         if (wakenUp.compareAndSet(false, true)) {
145             selector.wakeup();
146         }
147     }

149     public void run() {
150         thread = Thread.currentThread();
151 
152         boolean shutdown = false;
153         Selector selector = this.selector;
154         for (;;) {
155             wakenUp.set(false);
156 
157             if (CONSTRAINT_LEVEL != 0) {
158                 selectorGuard.writeLock().lock();
159                     // This empty synchronization block prevents the selector
160                     // from acquiring its lock.
161                 selectorGuard.writeLock().unlock();
162             }
163 
164             try {
165                 int selectedKeyCount = selector.select(500);
166 
167                 // 'wakenUp.compareAndSet(false, true)' is always evaluated
168                 // before calling 'selector.wakeup()' to reduce the wake-up
169                 // overhead. (Selector.wakeup() is an expensive operation.)
170                 //
171                 // However, there is a race condition in this approach.
172                 // The race condition is triggered when 'wakenUp' is set to
173                 // true too early.
174                 //
175                 // 'wakenUp' is set to true too early if:
176                 // 1) Selector is waken up between 'wakenUp.set(false)' and
177                 //    'selector.select(...)'. (BAD)
178                 // 2) Selector is waken up between 'selector.select(...)' and
179                 //    'if (wakenUp.get()) { ... }'. (OK)
180                 //
181                 // In the first case, 'wakenUp' is set to true and the
182                 // following 'selector.select(...)' will wake up immediately.
183                 // Until 'wakenUp' is set to false again in the next round,
184                 // 'wakenUp.compareAndSet(false, true)' will fail, and therefore
185                 // any attempt to wake up the Selector will fail, too, causing
186                 // the following 'selector.select(...)' call to block
187                 // unnecessarily.
188                 //
189                 // To fix this problem, we wake up the selector again if wakenUp
190                 // is true immediately after selector.select(...).
191                 // It is inefficient in that it wakes up the selector for both
192                 // the first case (BAD - wake-up required) and the second case
193                 // (OK - no wake-up required).
194 
195                 if (wakenUp.get()) {
196                     selector.wakeup();
197                 }
198 
199                 processRegisterTaskQueue();
200                 processWriteTaskQueue();
201 
202                 if (selectedKeyCount > 0) {
203                     processSelectedKeys(selector.selectedKeys());
204                 }
205 
206                 // Exit the loop when there's nothing to handle.
207                 // The shutdown flag is used to delay the shutdown of this
208                 // loop to avoid excessive Selector creation when
209                 // connections are registered in a one-by-one manner instead of
210                 // concurrent manner.
211                 if (selector.keys().isEmpty()) {
212                     if (shutdown ||
213                         executor instanceof ExecutorService && ((ExecutorService) executor).isShutdown()) {
214 
215                         synchronized (startStopLock) {
216                             if (registerTaskQueue.isEmpty() && selector.keys().isEmpty()) {
217                                 started = false;
218                                 try {
219                                     selector.close();
220                                 } catch (IOException e) {
221                                     logger.warn(
222                                             "Failed to close a selector.", e);
223                                 } finally {
224                                     this.selector = null;
225                                 }
226                                 break;
227                             } else {
228                                 shutdown = false;
229                             }
230                         }
231                     } else {
232                         // Give one more second.
233                         shutdown = true;
234                     }
235                 } else {
236                     shutdown = false;
237                 }
238             } catch (Throwable t) {
239                 logger.warn(
240                         "Unexpected exception in the selector loop.", t);
241 
242                 // Prevent possible consecutive immediate failures that lead to
243                 // excessive CPU consumption.
244                 try {
245                     Thread.sleep(1000);
246                 } catch (InterruptedException e) {
247                     // Ignore.
248                 }
249             }
250         }
251     }

275     private static void processSelectedKeys(Set<SelectionKey> selectedKeys) {
276         for (Iterator<SelectionKey> i = selectedKeys.iterator(); i.hasNext();) {
277             SelectionKey k = i.next();
278             i.remove();
279             try {
280                 int readyOps = k.readyOps();
281                 if ((readyOps & SelectionKey.OP_READ) != 0) {
282                     if (!read(k)) {
283                         // Connection already closed - no need to handle write.
284                         continue;
285                     }
286                 }
287                 if ((readyOps & SelectionKey.OP_WRITE) != 0) {
288                     write(k);
289                 }
290             } catch (CancelledKeyException e) {
291                 close(k);
292             }
293         }
294     }

296     private static boolean read(SelectionKey k) {
297         ScatteringByteChannel ch = (ScatteringByteChannel) k.channel();
298         NioSocketChannel channel = (NioSocketChannel) k.attachment();
299 
300         ReceiveBufferSizePredictor predictor =
301             channel.getConfig().getReceiveBufferSizePredictor();
302         ChannelBufferFactory bufferFactory =
303             channel.getConfig().getBufferFactory();
304 
305         ChannelBuffer buffer =
306             bufferFactory.getBuffer(predictor.nextReceiveBufferSize());
307 
308         int ret = 0;
309         int readBytes = 0;
310         boolean failure = true;
311         try {
312             while ((ret = buffer.writeBytes(ch, buffer.writableBytes())) > 0) {
313                 readBytes += ret;
314                 if (!buffer.writable()) {
315                     break;
316                 }
317             }
318             failure = false;
319         } catch (AsynchronousCloseException e) {
320             // Can happen, and does not need a user attention.
321         } catch (Throwable t) {
322             fireExceptionCaught(channel, t);
323         }
324 
325         if (readBytes > 0) {
326             // Update the predictor.
327             predictor.previousReceiveBufferSize(readBytes);
328 
329             // Fire the event.
330             fireMessageReceived(channel, buffer);
331         }
332 
333         if (ret < 0 || failure) {
334             close(k);
335             return false;
336         }
337 
338         return true;
339     }

351     static void write(final NioSocketChannel channel, boolean mightNeedWakeup) {
352         if (!channel.isConnected()) {
353             cleanUpWriteBuffer(channel);
354             return;
355         }
356 
357         if (mightNeedWakeup && scheduleWriteIfNecessary(channel)) {
358             return;
359         }
360 
361         if (channel.inWriteNowLoop) {
362             scheduleWriteIfNecessary(channel);
363         } else {
364             writeNow(channel, channel.getConfig().getWriteSpinCount());
365         }
366     }

368     private static boolean scheduleWriteIfNecessary(final NioSocketChannel channel) {
369         final NioWorker worker = channel.worker;
370         final Thread currentThread = Thread.currentThread();
371         final Thread workerThread = worker.thread;
372         if (workerThread == null || currentThread != workerThread) {
373             if (channel.writeTaskInTaskQueue.compareAndSet(false, true)) {
374                 boolean offered = worker.writeTaskQueue.offer(channel.writeTask);
375                 assert offered;
376             }
377 
378             if (!(channel instanceof NioAcceptedSocketChannel) ||
379                 ((NioAcceptedSocketChannel) channel).bossThread != currentThread) {
380                 final Selector workerSelector = worker.selector;
381                 if (workerSelector != null) {
382                     if (worker.wakenUp.compareAndSet(false, true)) {
383                         workerSelector.wakeup();
384                     }
385                 }
386             } else {
387                 // A write request can be made from an acceptor thread (boss)
388                 // when a user attempted to write something in:
389                 //
390                 //   * channelOpen()
391                 //   * channelBound()
392                 //   * channelConnected().
393                 //
394                 // In this case, there's no need to wake up the selector because
395                 // the channel is not even registered yet at this moment.
396             }
397 
398             return true;
399         }
400 
401         return false;
402     }

404     private static void writeNow(NioSocketChannel channel, int writeSpinCount) {
405 
406         boolean open = true;
407         boolean addOpWrite = false;
408         boolean removeOpWrite = false;
409 
410         MessageEvent evt;
411         ChannelBuffer buf;
412         int bufIdx;
413         int writtenBytes = 0;
414 
415         Queue<MessageEvent> writeBuffer = channel.writeBuffer;
416         synchronized (channel.writeLock) {
417             channel.inWriteNowLoop = true;
418             evt = channel.currentWriteEvent;
419             for (;;) {
420                 if (evt == null) {
421                     evt = writeBuffer.poll();
422                     if (evt == null) {
423                         channel.currentWriteEvent = null;
424                         removeOpWrite = true;
425                         break;
426                     }
427 
428                     evt = consolidateComposite(evt);
429                     buf = (ChannelBuffer) evt.getMessage();
430                     bufIdx = buf.readerIndex();
431                 } else {
432                     buf = (ChannelBuffer) evt.getMessage();
433                     bufIdx = channel.currentWriteIndex;
434                 }
435 
436                 try {
437                     for (int i = writeSpinCount; i > 0; i --) {
438                         int localWrittenBytes = buf.getBytes(
439                             bufIdx,
440                             channel.socket,
441                             buf.writerIndex() - bufIdx);
442 
443                         if (localWrittenBytes != 0) {
444                             bufIdx += localWrittenBytes;
445                             writtenBytes += localWrittenBytes;
446                             break;
447                         }
448                     }
449 
450                     if (bufIdx == buf.writerIndex()) {
451                         // Successful write - proceed to the next message.
452                         channel.currentWriteEvent = null;
453                         evt.getFuture().setSuccess();
454                         evt = null;
455                     } else {
456                         // Not written fully - perhaps the kernel buffer is full.
457                         channel.currentWriteEvent = evt;
458                         channel.currentWriteIndex = bufIdx;
459                         addOpWrite = true;
460                         break;
461                     }
462                 } catch (AsynchronousCloseException e) {
463                     // Doesn't need a user attention - ignore.
464                 } catch (Throwable t) {
465                     evt.getFuture().setFailure(t);
466                     evt = null;
467                     fireExceptionCaught(channel, t);
468                     if (t instanceof IOException) {
469                         open = false;
470                         close(channel, succeededFuture(channel));
471                     }
472                 }
473             }
474             channel.inWriteNowLoop = false;
475         }
476 
477         fireWriteComplete(channel, writtenBytes);
478 
479         if (open) {
480             if (addOpWrite) {
481                 setOpWrite(channel);
482             } else if (removeOpWrite) {
483                 clearOpWrite(channel);
484             }
485         }
486     }

488     static MessageEvent consolidateComposite(MessageEvent e) {
489         // Convert a composite buffer into a simple buffer to save memory
490         // bandwidth on full write buffer.
491         // This method should be eliminated once gathering write works.
492         Object m = e.getMessage();
493         if (m instanceof CompositeChannelBuffer) {
494             e = new DownstreamMessageEvent(
495                     e.getChannel(), e.getFuture(),
496                     ((CompositeChannelBuffer) m).copy(),
497                     e.getRemoteAddress());
498         }
499         return e;
500     }

502     private static void setOpWrite(NioSocketChannel channel) {
503         NioWorker worker = channel.worker;
504         Selector selector = worker.selector;
505         SelectionKey key = channel.socket.keyFor(selector);
506         if (key == null) {
507             return;
508         }
509         if (!key.isValid()) {
510             close(key);
511             return;
512         }
513         int interestOps;
514         boolean changed = false;
515 
516         // interestOps can change at any time and at any thread.
517         // Acquire a lock to avoid possible race condition.
518         synchronized (channel.interestOpsLock) {
519             interestOps = channel.getRawInterestOps();
520             if ((interestOps & SelectionKey.OP_WRITE) == 0) {
521                 interestOps |= SelectionKey.OP_WRITE;
522                 key.interestOps(interestOps);
523                 changed = true;
524             }
525         }
526 
527         if (changed) {
528             channel.setRawInterestOpsNow(interestOps);
529         }
530     }

562     static void close(NioSocketChannel channel, ChannelFuture future) {
563         NioWorker worker = channel.worker;
564         Selector selector = worker.selector;
565         SelectionKey key = channel.socket.keyFor(selector);
566         if (key != null) {
567             key.cancel();
568         }
569 
570         boolean connected = channel.isConnected();
571         boolean bound = channel.isBound();
572         try {
573             channel.socket.close();
574             if (channel.setClosed()) {
575                 future.setSuccess();
576                 if (connected) {
577                     fireChannelDisconnected(channel);
578                 }
579                 if (bound) {
580                     fireChannelUnbound(channel);
581                 }
582 
583                 cleanUpWriteBuffer(channel);
584                 fireChannelClosed(channel);
585             } else {
586                 future.setSuccess();
587             }
588         } catch (Throwable t) {
589             future.setFailure(t);
590             fireExceptionCaught(channel, t);
591         }

594     private static void cleanUpWriteBuffer(NioSocketChannel channel) {
595         Exception cause = null;
596         boolean fireExceptionCaught = false;
597 
598         // Clean up the stale messages in the write buffer.
599         synchronized (channel.writeLock) {
600             MessageEvent evt = channel.currentWriteEvent;
601             if (evt != null) {
602                 channel.currentWriteEvent = null;
603                 channel.currentWriteIndex = 0;
604 
605                 // Create the exception only once to avoid the excessive overhead
606                 // caused by fillStackTrace.
607                 if (channel.isOpen()) {
608                     cause = new NotYetConnectedException();
609                 } else {
610                     cause = new ClosedChannelException();
611                 }
612                 evt.getFuture().setFailure(cause);
613                 fireExceptionCaught = true;
614             }
615 
616             Queue<MessageEvent> writeBuffer = channel.writeBuffer;
617             if (!writeBuffer.isEmpty()) {
618                 // Create the exception only once to avoid the excessive overhead
619                 // caused by fillStackTrace.
620                 if (cause == null) {
621                     if (channel.isOpen()) {
622                         cause = new NotYetConnectedException();
623                     } else {
624                         cause = new ClosedChannelException();
625                     }
626                 }
627 
628                 for (;;) {
629                     evt = writeBuffer.poll();
630                     if (evt == null) {
631                         break;
632                     }
633                     evt.getFuture().setFailure(cause);
634                     fireExceptionCaught = true;
635                 }
636             }
637         }
638 
639         if (fireExceptionCaught) {
640             fireExceptionCaught(channel, cause);
641         }
642     }

644     static void setInterestOps(
645             NioSocketChannel channel, ChannelFuture future, int interestOps) {
646         boolean changed = false;
647         try {
648             // interestOps can change at any time and at any thread.
649             // Acquire a lock to avoid possible race condition.
650             synchronized (channel.interestOpsLock) {
651                 NioWorker worker = channel.worker;
652                 Selector selector = worker.selector;
653                 SelectionKey key = channel.socket.keyFor(selector);
654 
655                 if (key == null || selector == null) {
656                     // Not registered to the worker yet.
657                     // Set the rawInterestOps immediately; RegisterTask will pick it up.
658                     channel.setRawInterestOpsNow(interestOps);
659                     return;
660                 }
661 
662                 // Override OP_WRITE flag - a user cannot change this flag.
663                 interestOps &= ~Channel.OP_WRITE;
664                 interestOps |= channel.getRawInterestOps() & Channel.OP_WRITE;
665 
666                 switch (CONSTRAINT_LEVEL) {
667                 case 0:
668                     if (channel.getRawInterestOps() != interestOps) {
669                         key.interestOps(interestOps);
670                         if (Thread.currentThread() != worker.thread &&
671                             worker.wakenUp.compareAndSet(false, true)) {
672                             selector.wakeup();
673                         }
674                         changed = true;
675                     }
676                     break;
677                 case 1:
678                 case 2:
679                     if (channel.getRawInterestOps() != interestOps) {
680                         if (Thread.currentThread() == worker.thread) {
681                             key.interestOps(interestOps);
682                             changed = true;
683                         } else {
684                             worker.selectorGuard.readLock().lock();
685                             try {
686                                 if (worker.wakenUp.compareAndSet(false, true)) {
687                                     selector.wakeup();
688                                 }
689                                 key.interestOps(interestOps);
690                                 changed = true;
691                             } finally {
692                                 worker.selectorGuard.readLock().unlock();
693                             }
694                         }
695                     }
696                     break;
697                 default:
698                     throw new Error();
699                 }
700             }
701 
702             future.setSuccess();
703             if (changed) {
704                 channel.setRawInterestOpsNow(interestOps);
705                 fireChannelInterestChanged(channel);
706             }
707         } catch (Throwable t) {
708             future.setFailure(t);
709             fireExceptionCaught(channel, t);
710         }
711     }

713     private final class RegisterTask implements Runnable {
714         private final NioSocketChannel channel;
715         private final ChannelFuture future;
716         private final boolean server;
717 
718         RegisterTask(
719                 NioSocketChannel channel, ChannelFuture future, boolean server) {
720 
721             this.channel = channel;
722             this.future = future;
723             this.server = server;
724         }
725 
726         public void run() {
727             SocketAddress localAddress = channel.getLocalAddress();
728             SocketAddress remoteAddress = channel.getRemoteAddress();
729             if (localAddress == null || remoteAddress == null) {
730                 if (future != null) {
731                     future.setFailure(new ClosedChannelException());
732                 }
733                 close(channel, succeededFuture(channel));
734                 return;
735             }
736 
737             try {
738                 if (server) {
739                     channel.socket.configureBlocking(false);
740                 }
741 
742                 synchronized (channel.interestOpsLock) {
743                     channel.socket.register(
744                             selector, channel.getRawInterestOps(), channel);
745                 }
746                 if (future != null) {
747                     future.setSuccess();
748                 }
749             } catch (IOException e) {
750                 if (future != null) {
751                     future.setFailure(e);
752                 }
753                 close(channel, succeededFuture(channel));
754                 throw new ChannelException(
755                         "Failed to register a socket to the selector.", e);
756             }
757 
758             if (!server) {
759                 if (!((NioClientSocketChannel) channel).boundManually) {
760                     fireChannelBound(channel, localAddress);
761                 }
762                 fireChannelConnected(channel, remoteAddress);
763             }
764         }
765     }
