35  /**
36   * A {@link DatagramChannelConfig} for a NIO TCP/IP {@link DatagramChannel}.
37   *
38   * <h3>Available options</h3>
39   *
40   * In addition to the options provided by {@link ChannelConfig} and
41   * {@link DatagramChannelConfig}, {@link NioDatagramChannelConfig} allows the
42   * following options in the option map:
43   *
44   * <table border="1" cellspacing="0" cellpadding="6">
45   * <tr>
46   * <th>Name</th><th>Associated setter method</th>
47   * </tr><tr>
48   * <td>{@code "writeBufferHighWaterMark"}</td><td>{@link #setWriteBufferHighWaterMark(int)}</td>
49   * </tr><tr>
50   * <td>{@code "writeBufferLowWaterMark"}</td><td>{@link #setWriteBufferLowWaterMark(int)}</td>
51   * </tr><tr>
52   * <td>{@code "writeSpinCount"}</td><td>{@link #setWriteSpinCount(int)}</td>
53   * </tr><tr>
54   * <td>{@code "receiveBufferSizePredictor"}</td><td>{@link #setReceiveBufferSizePredictor(ReceiveBufferSizePredictor)}</td>
55   * </tr><tr>
56   * </table>
57   *
58   * @author The Netty Project (netty-dev@lists.jboss.org)
59   * @author Trustin Lee (tlee@redhat.com)
60   * @author Daniel Bevenius (dbevenius@jboss.com)
61   *
62   * @version $Rev: 1448 $, $Date: 2009-06-19 17:08:08 +0900 (Fri, 19 Jun 2009) $
63   */
64  public interface NioDatagramChannelConfig extends DatagramChannelConfig {
65  
66      /**
67       * Returns the high water mark of the write buffer.  If the number of bytes
68       * queued in the write buffer exceeds this value, {@link Channel#isWritable()}
69       * will start to return {@code true}.
70       */
71      int getWriteBufferHighWaterMark();
72  
73      /**
74       * Sets the high water mark of the write buffer.  If the number of bytes
75       * queued in the write buffer exceeds this value, {@link Channel#isWritable()}
76       * will start to return {@code true}.
77       */
78      void setWriteBufferHighWaterMark(int writeBufferHighWaterMark);
79  
80      /**
81       * Returns the low water mark of the write buffer.  Once the number of bytes
82       * queued in the write buffer exceeded the
83       * {@linkplain #setWriteBufferHighWaterMark(int) high water mark} and then
84       * dropped down below this value, {@link Channel#isWritable()} will return
85       * {@code false} again.
86       */
87      int getWriteBufferLowWaterMark();
88  
89      /**
90       * Sets the low water mark of the write buffer.  Once the number of bytes
91       * queued in the write buffer exceeded the
92       * {@linkplain #setWriteBufferHighWaterMark(int) high water mark} and then
93       * dropped down below this value, {@link Channel#isWritable()} will return
94       * {@code false} again.
95       */
96      void setWriteBufferLowWaterMark(int writeBufferLowWaterMark);
97  
98      /**
99       * Returns the maximum loop count for a write operation until
100      * {@link WritableByteChannel#write(ByteBuffer)} returns a non-zero value.
101      * It is similar to what a spin lock is used for in concurrency programming.
102      * It improves memory utilization and write throughput depending on
103      * the platform that JVM runs on.  The default value is {@code 16}.
104      */
105     int getWriteSpinCount();
106 
107     /**
108      * Sets the maximum loop count for a write operation until
109      * {@link WritableByteChannel#write(ByteBuffer)} returns a non-zero value.
110      * It is similar to what a spin lock is used for in concurrency programming.
111      * It improves memory utilization and write throughput depending on
112      * the platform that JVM runs on.  The default value is {@code 16}.
113      *
114      * @throws IllegalArgumentException
115      *         if the specified value is {@code 0} or less than {@code 0}
116      */
117     void setWriteSpinCount(int writeSpinCount);
118 
119     /**
120      * Returns the {@link ReceiveBufferSizePredictor} which predicts the
121      * number of readable bytes in the socket receive buffer.  The default
122      * predictor is {@link AdaptiveReceiveBufferSizePredictor}.
123      */
124     ReceiveBufferSizePredictor getReceiveBufferSizePredictor();
125 
126     /**
127      * Sets the {@link ReceiveBufferSizePredictor} which predicts the
128      * number of readable bytes in the socket receive buffer.  The default
129      * predictor is {@link AdaptiveReceiveBufferSizePredictor}.
130      */
131     void setReceiveBufferSizePredictor(ReceiveBufferSizePredictor predictor);
132 }
