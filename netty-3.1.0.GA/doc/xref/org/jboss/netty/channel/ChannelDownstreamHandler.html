23  package org.jboss.netty.channel;

25  import java.net.SocketAddress;
26

28  /**
29   * Handles or intercepts a downstream {@link ChannelEvent}, and sends a
30   * {@link ChannelEvent} to the previous or next handler in a
31   * {@link ChannelPipeline}.
32   *
33   * <h3>Downstream events</h3>
34   * <p>
35   * A downstream event is an event which is supposed to be processed from the
36   * last handler to the first handler in a {@link ChannelPipeline}.
37   * For example, all I/O requests made by a user application are downstream
38   * events.
39   * <p>
40   * The most common use case of this interface is to intercept an I/O request
41   * such as {@link Channel#write(Object)} and {@link Channel#close()}.  The
42   * received {@link ChannelEvent} object is interpreted as described in the
43   * following table:
44   *
45   * <table border="1" cellspacing="0" cellpadding="6">
46   * <tr>
47   * <th>Event name</th><th>Event type and condition</th><th>Meaning</th>
48   * </tr>
49   * <tr>
50   * <td>{@code "write"}</td>
51   * <td>{@link MessageEvent}</td><td>Send a message to the {@link Channel}.</td>
52   * </tr>
53   * <tr>
54   * <td>{@code "bind"}</td>
55   * <td>{@link ChannelStateEvent}<br/>(state = {@link ChannelState#BOUND BOUND}, value = {@link SocketAddress})</td>
56   * <td>Bind the {@link Channel} to the specified local address.</td>
57   * </tr>
58   * <tr>
59   * <td>{@code "unbind"}</td>
60   * <td>{@link ChannelStateEvent}<br/>(state = {@link ChannelState#BOUND BOUND}, value = {@code null})</td>
61   * <td>Unbind the {@link Channel} from the current local address.</td>
62   * </tr>
63   * <tr>
64   * <td>{@code "connect"}</td>
65   * <td>{@link ChannelStateEvent}<br/>(state = {@link ChannelState#CONNECTED CONNECTED}, value = {@link SocketAddress})</td>
66   * <td>Connect the {@link Channel} to the specified remote address.</td>
67   * </tr>
68   * <tr>
69   * <td>{@code "disconnect"}</td>
70   * <td>{@link ChannelStateEvent}<br/>(state = {@link ChannelState#CONNECTED CONNECTED}, value = {@code null})</td>
71   * <td>Disconnect the {@link Channel} from the current remote address.</td>
72   * </tr>
73   * <tr>
74   * <td>{@code "close"}</td>
75   * <td>{@link ChannelStateEvent}<br/>(state = {@link ChannelState#OPEN OPEN}, value = {@code false})</td>
76   * <td>Close the {@link Channel}.</td>
77   * </tr>
78   * </table>
79   * <p>
80   * Other event types and conditions which were not addressed here will be
81   * ignored and discarded.  Please note that there's no {@code "open"} in the
82   * table.  It is because a {@link Channel} is always open when it is created
83   * by a {@link ChannelFactory}.
84   *
85   * <h4>Additional resources worth reading</h4>
86   * <p>
87   * You might want to refer to {@link ChannelUpstreamHandler} to see how a
88   * {@link ChannelEvent} is interpreted when going upstream.  Also, please refer
89   * to the {@link ChannelEvent} and {@link ChannelPipeline} documentation to find
90   * out what an upstream event and a downstream event are, what fundamental
91   * differences they have, and how they flow in a pipeline.
92   *
93   * <h3>Firing an event to the previous or next handler</h3>
94   * <p>
95   * You can forward the received event downstream or upstream.  In most cases,
96   * {@link ChannelDownstreamHandler} will pass the event to the previous
97   * handler (downstream) although it is legal to pass the event to the next
98   * handler (upstream):
99   *
100  * <pre>
101  * // Sending the event forward (downstream)
102  * void handleDownstream({@link ChannelHandlerContext} ctx, {@link ChannelEvent} e) throws Exception {
103  *     ...
104  *     ctx.sendDownstream(e);
105  *     ...
106  * }
107  *
108  * // Sending the event backward (upstream)
109  * void handleDownstream({@link ChannelHandlerContext} ctx, {@link ChannelEvent} e) throws Exception {
110  *     ...
111  *     ctx.sendUpstream(new DefaultChannelStateEvent(...));
112  *     ...
113  * }
114  * </pre>
115  *
116  * <h4>Using the helper class to send an event</h4>
117  * <p>
118  * You will also find various helper methods in {@link Channels} to be useful
119  * to generate and send an artificial or manipulated event.
120  *
121  * <h3>Thread safety</h3>
122  * <p>
123  * {@link #handleDownstream(ChannelHandlerContext, ChannelEvent) handleDownstream}
124  * may be invoked by more than one thread simultaneously.  If the handler
125  * accesses a shared resource or stores stateful information, you might need
126  * proper synchronization in the handler implementation.
127  * <p>
128  * Also, please refer to the {@link ChannelPipelineCoverage} annotation to
129  * understand the relationship between a handler and its stateful properties.
130  *
131  * @author The Netty Project (netty-dev@lists.jboss.org)
132  * @author Trustin Lee (tlee@redhat.com)
133  *
134  * @version $Rev: 1403 $, $Date: 2009-06-17 17:33:54 +0900 (Wed, 17 Jun 2009) $
135  *
136  * @apiviz.exclude ^org\.jboss\.netty\.handler\..*$
137  */
138 public interface ChannelDownstreamHandler extends ChannelHandler {
139 
140     /**
141      * Handles the specified downstream event.
142      *
143      * @param ctx  the context object for this handler
144      * @param e    the downstream event to process or intercept
145      */
146     void handleDownstream(ChannelHandlerContext ctx, ChannelEvent e) throws Exception;
147 }
