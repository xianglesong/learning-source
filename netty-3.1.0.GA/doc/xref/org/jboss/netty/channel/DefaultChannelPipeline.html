78      public void attach(Channel channel, ChannelSink sink) {
79          if (channel == null) {
80              throw new NullPointerException("channel");
81          }
82          if (sink == null) {
83              throw new NullPointerException("sink");
84          }
85          if (this.channel != null || this.sink != null) {
86              throw new IllegalStateException("attached already");
87          }
88          this.channel = channel;
89          this.sink = sink;
90      }

96      public synchronized void addFirst(String name, ChannelHandler handler) {
97          if (name2ctx.isEmpty()) {
98              init(name, handler);
99          } else {
100             checkDuplicateName(name);
101             DefaultChannelHandlerContext oldHead = head;
102             DefaultChannelHandlerContext newHead = new DefaultChannelHandlerContext(null, oldHead, name, handler);
103 
104             callBeforeAdd(newHead);
105 
106             oldHead.prev = newHead;
107             head = newHead;
108             name2ctx.put(name, newHead);
109 
110             callAfterAdd(newHead);
111         }
112     }
113 
114     public synchronized void addLast(String name, ChannelHandler handler) {
115         if (name2ctx.isEmpty()) {
116             init(name, handler);
117         } else {
118             checkDuplicateName(name);
119             DefaultChannelHandlerContext oldTail = tail;
120             DefaultChannelHandlerContext newTail = new DefaultChannelHandlerContext(oldTail, null, name, handler);
121 
122             callBeforeAdd(newTail);
123 
124             oldTail.next = newTail;
125             tail = newTail;
126             name2ctx.put(name, newTail);
127 
128             callAfterAdd(newTail);
129         }
130     }

132     public synchronized void addBefore(String baseName, String name, ChannelHandler handler) {
133         DefaultChannelHandlerContext ctx = getContextOrDie(baseName);
134         if (ctx == head) {
135             addFirst(name, handler);
136         } else {
137             checkDuplicateName(name);
138             DefaultChannelHandlerContext newCtx = new DefaultChannelHandlerContext(ctx.prev, ctx, name, handler);
139 
140             callBeforeAdd(newCtx);
141 
142             ctx.prev.next = newCtx;
143             ctx.prev = newCtx;
144             name2ctx.put(name, newCtx);
145 
146             callAfterAdd(newCtx);
147         }
148     }

150     public synchronized void addAfter(String baseName, String name, ChannelHandler handler) {
151         DefaultChannelHandlerContext ctx = getContextOrDie(baseName);
152         if (ctx == tail) {
153             addLast(name, handler);
154         } else {
155             checkDuplicateName(name);
156             DefaultChannelHandlerContext newCtx = new DefaultChannelHandlerContext(ctx, ctx.next, name, handler);
157 
158             callBeforeAdd(newCtx);
159 
160             ctx.next.prev = newCtx;
161             ctx.next = newCtx;
162             name2ctx.put(name, newCtx);
163 
164             callAfterAdd(newCtx);
165         }

181     private DefaultChannelHandlerContext remove(DefaultChannelHandlerContext ctx) {
182         if (head == tail) {
183             head = tail = null;
184             name2ctx.clear();
185         } else if (ctx == head) {
186             removeFirst();
187         } else if (ctx == tail) {
188             removeLast();
189         } else {
190             callBeforeRemove(ctx);
191 
192             DefaultChannelHandlerContext prev = ctx.prev;
193             DefaultChannelHandlerContext next = ctx.next;
194             prev.next = next;
195             next.prev = prev;
196             name2ctx.remove(ctx.getName());
197 
198             callAfterRemove(ctx);
199         }
200         return ctx;
201     }

203     public synchronized ChannelHandler removeFirst() {
204         if (name2ctx.isEmpty()) {
205             throw new NoSuchElementException();
206         }
207 
208         DefaultChannelHandlerContext oldHead = head;
209         if (oldHead == null) {
210             throw new NoSuchElementException();
211         }
212 
213         callBeforeRemove(oldHead);
214 
215         if (oldHead.next == null) {
216             head = tail = null;
217             name2ctx.clear();
218         } else {
219             oldHead.next.prev = null;
220             head = oldHead.next;
221             name2ctx.remove(oldHead.getName());
222         }
223 
224         callAfterRemove(oldHead);
225 
226         return oldHead.getHandler();
227     }
228 
229     public synchronized ChannelHandler removeLast() {
230         if (name2ctx.isEmpty()) {
231             throw new NoSuchElementException();
232         }
233 
234         DefaultChannelHandlerContext oldTail = tail;
235         if (oldTail == null) {
236             throw new NoSuchElementException();
237         }
238 
239         callBeforeRemove(oldTail);
240 
241         if (oldTail.prev == null) {
242             head = tail = null;
243             name2ctx.clear();
244         } else {
245             oldTail.prev.next = null;
246             tail = oldTail.prev;
247             name2ctx.remove(oldTail.getName());
248         }
249 
250         callBeforeRemove(oldTail);
251 
252         return oldTail.getHandler();
253     }

269     private ChannelHandler replace(DefaultChannelHandlerContext ctx, String newName, ChannelHandler newHandler) {
270         if (ctx == head) {
271             removeFirst();
272             addFirst(newName, newHandler);
273         } else if (ctx == tail) {
274             removeLast();
275             addLast(newName, newHandler);
276         } else {
277             boolean sameName = ctx.getName().equals(newName);
278             if (!sameName) {
279                 checkDuplicateName(newName);
280             }
281 
282             DefaultChannelHandlerContext prev = ctx.prev;
283             DefaultChannelHandlerContext next = ctx.next;
284             DefaultChannelHandlerContext newCtx = new DefaultChannelHandlerContext(prev, next, newName, newHandler);
285 
286             callBeforeRemove(ctx);
287             callBeforeAdd(newCtx);
288 
289             prev.next = newCtx;
290             next.prev = newCtx;
291 
292             if (!sameName) {
293                 name2ctx.remove(ctx.getName());
294                 name2ctx.put(newName, newCtx);
295             }
296 
297             ChannelHandlerLifeCycleException removeException = null;
298             ChannelHandlerLifeCycleException addException = null;
299             boolean removed = false;
300             try {
301                 callAfterRemove(ctx);
302                 removed = true;
303             } catch (ChannelHandlerLifeCycleException e) {
304                 removeException = e;
305             }
306 
307             boolean added = false;
308             try {
309                 callAfterAdd(newCtx);
310                 added = true;
311             } catch (ChannelHandlerLifeCycleException e) {
312                 addException = e;
313             }
314 
315             if (!removed && !added) {
316                 logger.warn(removeException.getMessage(), removeException);
317                 logger.warn(addException.getMessage(), addException);
318                 throw new ChannelHandlerLifeCycleException(
319                         "Both " + ctx.getHandler().getClass().getName() +
320                         ".afterRemove() and " + newCtx.getHandler().getClass().getName() +
321                         ".afterAdd() failed; see logs.");
322             } else if (!removed) {
323                 throw removeException;
324             } else if (!added) {
325                 throw addException;
326             }
327         }
328 
329         return ctx.getHandler();
330     }

332     private void callBeforeAdd(ChannelHandlerContext ctx) {
333         if (!(ctx.getHandler() instanceof LifeCycleAwareChannelHandler)) {
334             return;
335         }
336 
337         if (!isAttached()) {
338             return;
339         }
340 
341         LifeCycleAwareChannelHandler h =
342             (LifeCycleAwareChannelHandler) ctx.getHandler();
343 
344         try {
345             h.beforeAdd(ctx);
346         } catch (Throwable t) {
347             throw new ChannelHandlerLifeCycleException(
348                     h.getClass().getName() +
349                     ".beforeAdd() has thrown an exception; not adding.", t);
350         }
351     }
352 
353     private void callAfterAdd(ChannelHandlerContext ctx) {
354         if (!(ctx.getHandler() instanceof LifeCycleAwareChannelHandler)) {
355             return;
356         }
357 
358         if (!isAttached()) {
359             return;
360         }
361 
362         LifeCycleAwareChannelHandler h =
363             (LifeCycleAwareChannelHandler) ctx.getHandler();
364 
365         try {
366             h.afterAdd(ctx);
367         } catch (Throwable t) {
368             boolean removed = false;
369             try {
370                 remove((DefaultChannelHandlerContext) ctx);
371                 removed = true;
372             } catch (Throwable t2) {
373                 logger.warn("Failed to remove a handler: " + ctx.getName(), t2);
374             }
375 
376             if (removed) {
377                 throw new ChannelHandlerLifeCycleException(
378                         h.getClass().getName() +
379                         ".afterAdd() has thrown an exception; removed.", t);
380             } else {
381                 throw new ChannelHandlerLifeCycleException(
382                         h.getClass().getName() +
383                         ".afterAdd() has thrown an exception; also failed to remove.", t);
384             }
385         }
386     }

472     public synchronized ChannelHandlerContext getContext(ChannelHandler handler) {
473         if (handler == null) {
474             throw new NullPointerException("handler");
475         }
476         if (name2ctx.isEmpty()) {
477             return null;
478         }
479         DefaultChannelHandlerContext ctx = head;
480         for (;;) {
481             if (ctx.getHandler() == handler) {
482                 return ctx;
483             }
484 
485             ctx = ctx.next;
486             if (ctx == null) {
487                 break;
488             }
489         }
490         return null;
491     }
492 
493     public synchronized ChannelHandlerContext getContext(
494             Class<? extends ChannelHandler> handlerType) {
495         if (name2ctx.isEmpty()) {
496             return null;
497         }
498         DefaultChannelHandlerContext ctx = head;
499         for (;;) {
500             if (handlerType.isAssignableFrom(ctx.getHandler().getClass())) {
501                 return ctx;
502             }
503 
504             ctx = ctx.next;
505             if (ctx == null) {
506                 break;
507             }
508         }
509         return null;
510     }

512     public Map<String, ChannelHandler> toMap() {
513         Map<String, ChannelHandler> map = new LinkedHashMap<String, ChannelHandler>();
514         if (name2ctx.isEmpty()) {
515             return map;
516         }
517 
518         DefaultChannelHandlerContext ctx = head;
519         for (;;) {
520             map.put(ctx.getName(), ctx.getHandler());
521             ctx = ctx.next;
522             if (ctx == null) {
523                 break;
524             }
525         }
526         return map;

530      * Returns the {@link String} representation of this pipeline.
531      */
532     @Override
533     public String toString() {
534         StringBuilder buf = new StringBuilder();
535         buf.append(getClass().getSimpleName());
536         buf.append('{');
537         DefaultChannelHandlerContext ctx = head;
538         for (;;) {
539             buf.append('(');
540             buf.append(ctx.getName());
541             buf.append(" = ");
542             buf.append(ctx.getHandler().getClass().getName());
543             buf.append(')');
544             ctx = ctx.next;
545             if (ctx == null) {
546                 break;
547             }
548             buf.append(", ");
549         }
550         buf.append('}');
551         return buf.toString();
552     }

555         DefaultChannelHandlerContext head = getActualUpstreamContext(this.head);
556         if (head == null) {
557             logger.warn(

573     public void sendDownstream(ChannelEvent e) {
574         DefaultChannelHandlerContext tail = getActualDownstreamContext(this.tail);
575         if (tail == null) {
576             try {
577                 getSink().eventSunk(this, e);
578                 return;
579             } catch (Throwable t) {
580                 notifyHandlerException(e, t);
581                 return;
582             }
583         }
584 
585         sendDownstream(tail, e);
586     }

596     DefaultChannelHandlerContext getActualUpstreamContext(DefaultChannelHandlerContext ctx) {
597         if (ctx == null) {
598             return null;
599         }
600 
601         DefaultChannelHandlerContext realCtx = ctx;
602         while (!realCtx.canHandleUpstream()) {
603             realCtx = realCtx.next;
604             if (realCtx == null) {
605                 return null;
606             }
607         }
608 
609         return realCtx;
610     }
611 
612     DefaultChannelHandlerContext getActualDownstreamContext(DefaultChannelHandlerContext ctx) {
613         if (ctx == null) {
614             return null;
615         }
616 
617         DefaultChannelHandlerContext realCtx = ctx;
618         while (!realCtx.canHandleDownstream()) {
619             realCtx = realCtx.prev;
620             if (realCtx == null) {
621                 return null;
622             }
623         }
624 
625         return realCtx;
626     }

628     protected void notifyHandlerException(ChannelEvent e, Throwable t) {
629         if (e instanceof ExceptionEvent) {
630             logger.warn(
631                     "An exception was thrown by a user handler " +
632                     "while handling an exception event (" + e + ")", t);
633             return;
634         }
635 
636         ChannelPipelineException pe;
637         if (t instanceof ChannelPipelineException) {
638             pe = (ChannelPipelineException) t;
639         } else {
640             pe = new ChannelPipelineException(t);
641         }
642 
643         try {
644             sink.exceptionCaught(this, e, pe);
645         } catch (Exception e1) {
646             logger.warn("An exception was thrown by an exception handler.", e1);
647         }
648     }

690     private final class DefaultChannelHandlerContext implements ChannelHandlerContext {
691         volatile DefaultChannelHandlerContext next;
692         volatile DefaultChannelHandlerContext prev;
693         private final String name;
694         private final ChannelHandler handler;
695         private final boolean canHandleUpstream;
696         private final boolean canHandleDownstream;
697         private volatile Object attachment;
698 
699         DefaultChannelHandlerContext(
700                 DefaultChannelHandlerContext prev, DefaultChannelHandlerContext next,
701                 String name, ChannelHandler handler) {
702 
703             if (name == null) {
704                 throw new NullPointerException("name");
705             }
706             if (handler == null) {
707                 throw new NullPointerException("handler");
708             }
709             canHandleUpstream = handler instanceof ChannelUpstreamHandler;
710             canHandleDownstream = handler instanceof ChannelDownstreamHandler;
711 
712 
713             if (!canHandleUpstream && !canHandleDownstream) {
714                 throw new IllegalArgumentException(
715                         "handler must be either " +
716                         ChannelUpstreamHandler.class.getName() + " or " +
717                         ChannelDownstreamHandler.class.getName() + '.');
718             }
719 
720 
721             ChannelPipelineCoverage coverage = handler.getClass().getAnnotation(ChannelPipelineCoverage.class);
722             if (coverage == null) {
723                 logger.warn(
724                         "Handler '" + handler.getClass().getName() +
725                         "' does not have a '" +
726                         ChannelPipelineCoverage.class.getSimpleName() +
727                         "' annotation with its class declaration. " +
728                         "It is strongly recommended to add the annotation " +
729                         "for a documentation purpose to tell if a single " +
730                         "handler instance can handle more than one pipeline " +
731                         "(\"" + ALL + "\") or not (\"" + ONE + "\")");
732             } else {
733                 String coverageValue = coverage.value();
734                 if (coverageValue == null) {
735                     throw new AnnotationFormatError(
736                             ChannelPipelineCoverage.class.getSimpleName() +
737                             " annotation value is undefined for type: " +
738                             handler.getClass().getName());
739                 }
740 
741                 if (!coverageValue.equals(ALL) && !coverageValue.equals(ONE)) {
742                     throw new AnnotationFormatError(
743                             ChannelPipelineCoverage.class.getSimpleName() +
744                             " annotation value: " + coverageValue +
745                             " (must be either \"" + ALL + "\" or \"" + ONE + ")");
746                 }
747             }
748 
749             this.prev = prev;
750             this.next = next;
751             this.name = name;
752             this.handler = handler;
753         }
