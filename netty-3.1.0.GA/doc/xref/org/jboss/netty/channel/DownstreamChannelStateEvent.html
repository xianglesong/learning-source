23  package org.jboss.netty.channel;
24  
25  /**
26   * The default downstream {@link ChannelStateEvent} implementation.
27   *
28   * @author The Netty Project (netty-dev@lists.jboss.org)
29   * @author Trustin Lee (tlee@redhat.com)
30   *
31   * @version $Rev: 1406 $, $Date: 2009-06-17 18:33:20 +0900 (Wed, 17 Jun 2009) $
32   *
33   */
34  public class DownstreamChannelStateEvent implements ChannelStateEvent {
35  
36      private final Channel channel;
37      private final ChannelFuture future;
38      private final ChannelState state;
39      private final Object value;
40  
41      /**
42       * Creates a new instance.
43       */
44      public DownstreamChannelStateEvent(
45              Channel channel, ChannelFuture future,
46              ChannelState state, Object value) {
47  
48          if (channel == null) {
49              throw new NullPointerException("channel");
50          }
51          if (future == null) {
52              throw new NullPointerException("future");
53          }
54          if (state == null) {
55              throw new NullPointerException("state");
56          }
57          this.channel = channel;
58          this.future = future;
59          this.state = state;
60          this.value = value;
61      }
62  
63      public Channel getChannel() {
64          return channel;
65      }
66  
67      public ChannelFuture getFuture() {
68          return future;
69      }
70  
71      public ChannelState getState() {
72          return state;
73      }
74  
75      public Object getValue() {
76          return value;
77      }
78  
79      @Override
80      public String toString() {
81          String channelString = getChannel().toString();
82          StringBuilder buf = new StringBuilder(channelString.length() + 64);
83          buf.append(channelString);
84          switch (getState()) {
85          case OPEN:
86              if (Boolean.TRUE.equals(getValue())) {
87                  buf.append(" OPEN");
88              } else {
89                  buf.append(" CLOSE");
90              }
91              break;
92          case BOUND:
93              if (getValue() != null) {
94                  buf.append(" BIND: ");
95                  buf.append(getValue());
96              } else {
97                  buf.append(" UNBIND");
98              }
99              break;
100         case CONNECTED:
101             if (getValue() != null) {
102                 buf.append(" CONNECT: ");
103                 buf.append(getValue());
104             } else {
105                 buf.append(" DISCONNECT");
106             }
107             break;
108         case INTEREST_OPS:
109             buf.append(" CHANGE_INTEREST: ");
110             buf.append(getValue());
111             break;
112         default:
113             buf.append(' ');
114             buf.append(getState().name());
115             buf.append(": ");
116             buf.append(getValue());
117         }
118         return buf.toString();
119     }
