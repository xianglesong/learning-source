30  /**
31   * A {@link ChannelUpstreamHandler} which provides an individual handler method
32   * for each event type.  This handler down-casts the received upstream event
33   * into more meaningful sub-type event and calls an appropriate handler method
34   * with the down-cast event.  The names of the methods are identical to the
35   * upstream event names, as introduced in the {@link ChannelUpstreamHandler}
36   * documentation.
37   * <p>
38   * Please use {@link SimpleChannelHandler} if you need to implement both
39   * {@link ChannelUpstreamHandler} and {@link ChannelDownstreamHandler}.
40   *
41   * <h3>Overriding the {@link #handleUpstream(ChannelHandlerContext, ChannelEvent) handleUpstream} method</h3>
42   * <p>
43   * You can override the {@link #handleUpstream(ChannelHandlerContext, ChannelEvent) handleUpstream}
44   * method just like overriding an ordinary Java method.  Please make sure to
45   * call {@code super.handleUpstream()} so that other handler methods are invoked
46   * properly:
47   * </p>
48   * <pre>public class MyChannelHandler extends SimpleChannelUpstreamHandler {
49   *
50   *     public void handleUpstream({@link ChannelHandlerContext} ctx, {@link ChannelEvent} e) throws Exception {
51   *
52   *         // Log all channel state changes.
53   *         if (e instanceof ChannelStateEvent) {
54   *             logger.info("Channel state changed: " + e);
55   *         }
56   *
57   *         <strong>super.handleUpstream(ctx, e);</strong>
58   *     }
59   * }</pre>
60   *
61   * @author The Netty Project (netty-dev@lists.jboss.org)
62   * @author Trustin Lee (tlee@redhat.com)
63   *
64   * @version $Rev: 760 $, $Date: 2009-02-05 14:20:37 +0900 (Thu, 05 Feb 2009) $
65   */

68      private static final InternalLogger logger =
69          InternalLoggerFactory.getInstance(SimpleChannelUpstreamHandler.class.getName());

72       * Creates a new instance.
73       */
74      public SimpleChannelUpstreamHandler() {
75          super();
76      }

78      /**
79       * {@inheritDoc}  Down-casts the received upstream event into more
80       * meaningful sub-type event and calls an appropriate handler method with
81       * the down-casted event.
82       */
83      public void handleUpstream(
84              ChannelHandlerContext ctx, ChannelEvent e) throws Exception {
85  
86          if (e instanceof MessageEvent) {
87              messageReceived(ctx, (MessageEvent) e);
88          } else if (e instanceof WriteCompletionEvent) {
89              WriteCompletionEvent evt = (WriteCompletionEvent) e;
90              writeComplete(ctx, evt);
91          } else if (e instanceof ChildChannelStateEvent) {
92              ChildChannelStateEvent evt = (ChildChannelStateEvent) e;
93              if (evt.getChildChannel().isOpen()) {
94                  childChannelOpen(ctx, evt);
95              } else {
96                  childChannelClosed(ctx, evt);
97              }
98          } else if (e instanceof ChannelStateEvent) {
99              ChannelStateEvent evt = (ChannelStateEvent) e;
100             switch (evt.getState()) {
101             case OPEN:
102                 if (Boolean.TRUE.equals(evt.getValue())) {
103                     channelOpen(ctx, evt);
104                 } else {
105                     channelClosed(ctx, evt);
106                 }
107                 break;
108             case BOUND:
109                 if (evt.getValue() != null) {
110                     channelBound(ctx, evt);
111                 } else {
112                     channelUnbound(ctx, evt);
113                 }
114                 break;
115             case CONNECTED:
116                 if (evt.getValue() != null) {
117                     channelConnected(ctx, evt);
118                 } else {
119                     channelDisconnected(ctx, evt);
120                 }
121                 break;
122             case INTEREST_OPS:
123                 channelInterestChanged(ctx, evt);
124                 break;
125             default:
126                 ctx.sendDownstream(e);
127             }
128         } else if (e instanceof ExceptionEvent) {
129             exceptionCaught(ctx, (ExceptionEvent) e);
130         } else {
131             ctx.sendUpstream(e);
132         }
133     }

135     /**
136      * Invoked when a message object (e.g: {@link ChannelBuffer}) was received
137      * from a remote peer.
138      */
139     public void messageReceived(
140             ChannelHandlerContext ctx, MessageEvent e) throws Exception {
141         ctx.sendUpstream(e);
142     }
143 
144     /**
145      * Invoked when an exception was raised by an I/O thread or a
146      * {@link ChannelHandler}.
147      */
148     public void exceptionCaught(
149             ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
150         if (this == ctx.getPipeline().getLast()) {
151             logger.warn(
152                     "EXCEPTION, please implement " + getClass().getName() +
153                     ".exceptionCaught() for proper handling.", e.getCause());
154         }
155         ctx.sendUpstream(e);
156     }
