23  package org.jboss.netty.handler.codec.http;
24  
25  import static org.jboss.netty.handler.codec.http.HttpCodecUtil.*;
26  
27  import java.io.UnsupportedEncodingException;
28  
29  import org.jboss.netty.buffer.ChannelBuffer;
30  
31  /**
32   * Encodes an {@link HttpResponse} or an {@link HttpChunk} into
33   * a {@link ChannelBuffer}.
34   *
35   * @author The Netty Project (netty-dev@lists.jboss.org)
36   * @author Andy Taylor (andy.taylor@jboss.org)
37   * @author Trustin Lee (tlee@redhat.com)
38   * @version $Rev: 1482 $, $Date: 2009-06-20 02:48:17 +0900 (Sat, 20 Jun 2009) $
39   */
40  public class HttpResponseEncoder extends HttpMessageEncoder {
41  
42      /**
43       * Creates a new instance.
44       */
45      public HttpResponseEncoder() {
46          super();
47      }
48  
49      @Override
50      protected void encodeInitialLine(ChannelBuffer buf, HttpMessage message) {
51          HttpResponse response = (HttpResponse) message;
52          try {
53              buf.writeBytes(response.getProtocolVersion().toString().getBytes("ASCII"));
54              buf.writeByte(SP);
55              buf.writeBytes(String.valueOf(response.getStatus().getCode()).getBytes("ASCII"));
56              buf.writeByte(SP);
57              buf.writeBytes(String.valueOf(response.getStatus().getReasonPhrase()).getBytes("ASCII"));
58              buf.writeBytes(CRLF);
59          } catch (UnsupportedEncodingException e) {
60              throw (Error) new Error().initCause(e);
61          }
62      }
63  }
