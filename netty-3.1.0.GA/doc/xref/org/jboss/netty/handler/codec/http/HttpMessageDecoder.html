35  /**
36   * Decodes {@link ChannelBuffer}s into {@link HttpMessage}s and
37   * {@link HttpChunk}s.
38   *
39   * <h3>Parameters that prevents excessive memory consumption</h3>
40   * <table border="1">
41   * <tr>
42   * <th>Name</th><th>Meaning</th>
43   * </tr>
44   * <tr>
45   * <td>{@code maxInitialLineLength}</td>
46   * <td>The maximum length of the initial line
47   *     (e.g. {@code "GET / HTTP/1.0"} or {@code "HTTP/1.0 200 OK"})
48   *     If the length of the initial line exceeds this value, a
49   *     {@link TooLongFrameException} will be raised.</td>
50   * </tr>
51   * <tr>
52   * <td>{@code maxHeaderSize}</td>
53   * <td>The maximum length of all headers.  If the sum of the length of each
54   *     header exceeds this value, a {@link TooLongFrameException} will be raised.</td>
55   * </tr>
56   * <tr>
57   * <td>{@code maxChunkSize}</td>
58   * <td>The maximum length of the content or each chunk.  If the content length
59   *     exceeds this value, the transfer encoding of the decoded message will be
60   *     converted to 'chunked' and the content will be split into multiple
61   *     {@link HttpChunk}s.  If the transfer encoding of the HTTP message is
62   *     'chunked' already, each chunk will be split into smaller chunks if the
63   *     length of the chunk exceeds this value.  If you prefer not to handle
64   *     {@link HttpChunk}s in your handler, insert {@link HttpChunkAggregator}
65   *     after this decoder in the {@link ChannelPipeline}.</td>
66   * </tr>
67   * </table>
68   *
69   * <h3>Extensibility</h3>
70   *
71   * Please note that this decoder is designed to be extended to implement
72   * a protocol derived from HTTP, such as
73   * <a href="http://en.wikipedia.org/wiki/Real_Time_Streaming_Protocol">RTSP</a> and
74   * <a href="http://en.wikipedia.org/wiki/Internet_Content_Adaptation_Protocol">ICAP</a>.
75   * To implement the decoder of such a derived protocol, extend this class and
76   * implement all abstract methods properly.
77   *
78   * @author The Netty Project (netty-dev@lists.jboss.org)
79   * @author Andy Taylor (andy.taylor@jboss.org)
80   * @author Trustin Lee (tlee@redhat.com)
81   * @version $Rev: 1586 $, $Date: 2009-07-20 12:37:35 +0900 (Mon, 20 Jul 2009) $
82   *
83   * @apiviz.landmark
84   */

108         READ_HEADER,

110         READ_VARIABLE_LENGTH_CONTENT_AS_CHUNKS,
111         READ_FIXED_LENGTH_CONTENT,
112         READ_FIXED_LENGTH_CONTENT_AS_CHUNKS,
113         READ_CHUNK_SIZE,
114         READ_CHUNKED_CONTENT,
115         READ_CHUNKED_CONTENT_AS_CHUNKS,
116         READ_CHUNK_DELIMITER,
117         READ_CHUNK_FOOTER;

132     protected HttpMessageDecoder(
133             int maxInitialLineLength, int maxHeaderSize, int maxChunkSize) {
134 
135         super(State.SKIP_CONTROL_CHARS, true);
136 
137         if (maxInitialLineLength <= 0) {
138             throw new IllegalArgumentException(
139                     "maxInitialLineLength must be a positive integer: " +
140                     maxInitialLineLength);
141         }
142         if (maxHeaderSize <= 0) {
143             throw new IllegalArgumentException(
144                     "maxHeaderSize must be a positive integer: " +
145                     maxChunkSize);
146         }
147         if (maxChunkSize < 0) {
148             throw new IllegalArgumentException(
149                     "maxChunkSize must be a positive integer: " +
150                     maxChunkSize);
151         }
152         this.maxInitialLineLength = maxInitialLineLength;
153         this.maxHeaderSize = maxHeaderSize;
154         this.maxChunkSize = maxChunkSize;
155     }

157     @Override
158     protected Object decode(ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer, State state) throws Exception {
159         switch (state) {
160         case SKIP_CONTROL_CHARS: {
161             try {
162                 skipControlCharacters(buffer);
163                 checkpoint(State.READ_INITIAL);
164             } finally {
165                 checkpoint();
166             }
167         }
168         case READ_INITIAL: {
169             String[] initialLine = splitInitialLine(readLine(buffer, maxInitialLineLength));
170             if (initialLine.length < 3) {
171                 // Invalid initial line - ignore.
172                 checkpoint(State.SKIP_CONTROL_CHARS);
173                 return null;
174             }
175 
176             message = createMessage(initialLine);
177             checkpoint(State.READ_HEADER);
178         }
179         case READ_HEADER: {
180             State nextState = readHeaders(buffer);
181             checkpoint(nextState);
182             if (nextState == State.READ_CHUNK_SIZE) {
183                 // Chunked encoding
184                 // Generate HttpMessage first.  HttpChunks will follow.
185                 return message;
186             } else if (nextState == State.SKIP_CONTROL_CHARS) {
187                 // No content is expected.
188                 // Remove the headers which are not supposed to be present not
189                 // to confuse subsequent handlers.
190                 message.removeHeader(HttpHeaders.Names.CONTENT_LENGTH);
191                 message.removeHeader(HttpHeaders.Names.TRANSFER_ENCODING);
192                 return message;
193             } else {
194                 long contentLength = message.getContentLength(-1);
195                 if (contentLength == 0 || contentLength == -1 && isDecodingRequest()) {
196                     content = ChannelBuffers.EMPTY_BUFFER;
197                     return reset();
198                 }
199 
200                 switch (nextState) {
201                 case READ_FIXED_LENGTH_CONTENT:
202                     if (contentLength > maxChunkSize) {
203                         // Generate HttpMessage first.  HttpChunks will follow.
204                         checkpoint(State.READ_FIXED_LENGTH_CONTENT_AS_CHUNKS);
205                         message.addHeader(HttpHeaders.Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED);
206                         // chunkSize will be decreased as the READ_FIXED_LENGTH_CONTENT_AS_CHUNKS
207                         // state reads data chunk by chunk.
208                         chunkSize = message.getContentLength(-1);
209                         return message;
210                     }
211                     break;
212                 case READ_VARIABLE_LENGTH_CONTENT:
213                     if (buffer.readableBytes() > maxChunkSize) {
214                         // Generate HttpMessage first.  HttpChunks will follow.
215                         checkpoint(State.READ_VARIABLE_LENGTH_CONTENT_AS_CHUNKS);
216                         message.addHeader(HttpHeaders.Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED);
217                         return message;
218                     }
219                     break;
220                 }
221             }
222             // We return null here, this forces decode to be called again where we will decode the content
223             return null;
224         }
225         case READ_VARIABLE_LENGTH_CONTENT: {
226             if (content == null) {
227                 content = ChannelBuffers.dynamicBuffer(channel.getConfig().getBufferFactory());
228             }
229             //this will cause a replay error until the channel is closed where this will read what's left in the buffer
230             content.writeBytes(buffer.readBytes(buffer.readableBytes()));
231             return reset();
232         }
233         case READ_VARIABLE_LENGTH_CONTENT_AS_CHUNKS: {
234             // Keep reading data as a chunk until the end of connection is reached.
235             int chunkSize = Math.min(maxChunkSize, buffer.readableBytes());
236             HttpChunk chunk = new DefaultHttpChunk(buffer.readBytes(chunkSize));
237 
238             if (!buffer.readable()) {
239                 // Reached to the end of the connection.
240                 reset();
241                 if (!chunk.isLast()) {
242                     // Append the last chunk.
243                     return new Object[] { chunk, HttpChunk.LAST_CHUNK };
244                 }
245             }
246             return chunk;
247         }
248         case READ_FIXED_LENGTH_CONTENT: {
249             //we have a content-length so we just read the correct number of bytes
250             readFixedLengthContent(buffer);
251             return reset();
252         }
253         case READ_FIXED_LENGTH_CONTENT_AS_CHUNKS: {
254             long chunkSize = this.chunkSize;
255             HttpChunk chunk;
256             if (chunkSize > maxChunkSize) {
257                 chunk = new DefaultHttpChunk(buffer.readBytes(maxChunkSize));
258                 chunkSize -= maxChunkSize;
259             } else {
260                 assert chunkSize <= Integer.MAX_VALUE;
261                 chunk = new DefaultHttpChunk(buffer.readBytes((int) chunkSize));
262                 chunkSize = 0;
263             }
264             this.chunkSize = chunkSize;
265 
266             if (chunkSize == 0) {
267                 // Read all content.
268                 reset();
269                 if (!chunk.isLast()) {
270                     // Append the last chunk.
271                     return new Object[] { chunk, HttpChunk.LAST_CHUNK };
272                 }
273             }
274             return chunk;
275         }
276         /**
277          * everything else after this point takes care of reading chunked content. basically, read chunk size,
278          * read chunk, read and ignore the CRLF and repeat until 0
279          */
280         case READ_CHUNK_SIZE: {
281             String line = readLine(buffer, maxInitialLineLength);
282             int chunkSize = getChunkSize(line);
283             this.chunkSize = chunkSize;
284             if (chunkSize == 0) {
285                 checkpoint(State.READ_CHUNK_FOOTER);
286                 return null;
287             } else if (chunkSize > maxChunkSize) {
288                 // A chunk is too large. Split them into multiple chunks again.
289                 checkpoint(State.READ_CHUNKED_CONTENT_AS_CHUNKS);
290             } else {
291                 checkpoint(State.READ_CHUNKED_CONTENT);
292             }
293         }
294         case READ_CHUNKED_CONTENT: {
295             assert chunkSize <= Integer.MAX_VALUE;
296             HttpChunk chunk = new DefaultHttpChunk(buffer.readBytes((int) chunkSize));
297             checkpoint(State.READ_CHUNK_DELIMITER);
298             return chunk;
299         }
300         case READ_CHUNKED_CONTENT_AS_CHUNKS: {
301             long chunkSize = this.chunkSize;
302             HttpChunk chunk;
303             if (chunkSize > maxChunkSize) {
304                 chunk = new DefaultHttpChunk(buffer.readBytes(maxChunkSize));
305                 chunkSize -= maxChunkSize;
306             } else {
307                 assert chunkSize <= Integer.MAX_VALUE;
308                 chunk = new DefaultHttpChunk(buffer.readBytes((int) chunkSize));
309                 chunkSize = 0;
310             }
311             this.chunkSize = chunkSize;
312 
313             if (chunkSize == 0) {
314                 // Read all content.
315                 checkpoint(State.READ_CHUNK_DELIMITER);
316             }
317 
318             if (!chunk.isLast()) {
319                 return chunk;
320             }
321         }
322         case READ_CHUNK_DELIMITER: {
323             for (;;) {
324                 byte next = buffer.readByte();
325                 if (next == HttpCodecUtil.CR) {
326                     if (buffer.readByte() == HttpCodecUtil.LF) {
327                         checkpoint(State.READ_CHUNK_SIZE);
328                         return null;
329                     }
330                 } else if (next == HttpCodecUtil.LF) {
331                     checkpoint(State.READ_CHUNK_SIZE);
332                     return null;
333                 }
334             }
335         }
336         case READ_CHUNK_FOOTER: {
337             // Skip the footer; does anyone use it?
338             try {
339                 if (!skipLine(buffer)) {
340                     if (maxChunkSize == 0) {
341                         // Chunked encoding disabled.
342                         return reset();
343                     } else {
344                         reset();
345                         // The last chunk, which is empty
346                         return HttpChunk.LAST_CHUNK;
347                     }
348                 }
349             } finally {
350                 checkpoint();
351             }
352             return null;
353         }
354         default: {
355             throw new Error("Shouldn't reach here.");
356         }
357 
358         }
359     }

361     protected boolean isContentAlwaysEmpty(HttpMessage msg) {
362         if (msg instanceof HttpResponse) {
363             HttpResponse res = (HttpResponse) msg;
364             int code = res.getStatus().getCode();
365             if (code < 200) {
366                 return true;
367             }
368             switch (code) {
369             case 204: case 205: case 304:
370                 return true;
371             }
372         }
373         return false;
374     }

376     private Object reset() {
377         HttpMessage message = this.message;
378         ChannelBuffer content = this.content;
379 
380         if (content != null) {
381             message.setContent(content);
382             this.content = null;
383         }
384         this.message = null;
385 
386         checkpoint(State.SKIP_CONTROL_CHARS);
387         return message;
388     }

390     private void skipControlCharacters(ChannelBuffer buffer) {
391         for (;;) {
392             char c = (char) buffer.readUnsignedByte();
393             if (!Character.isISOControl(c) &&
394                 !Character.isWhitespace(c)) {
395                 buffer.readerIndex(buffer.readerIndex() - 1);
396                 break;
397             }
398         }
399     }

401     private void readFixedLengthContent(ChannelBuffer buffer) {
402         long length = message.getContentLength(-1);
403         assert length <= Integer.MAX_VALUE;
404 
405         if (content == null) {
406             content = buffer.readBytes((int) length);
407         } else {
408             content.writeBytes(buffer.readBytes((int) length));
409         }
410     }

412     private State readHeaders(ChannelBuffer buffer) throws TooLongFrameException {
413         headerSize = 0;
414         final HttpMessage message = this.message;
415         String line = readHeader(buffer);
416         String lastHeader = null;
417         if (line.length() != 0) {
418             message.clearHeaders();
419             do {
420                 char firstChar = line.charAt(0);
421                 if (lastHeader != null && (firstChar == ' ' || firstChar == '\t')) {
422                     List<String> current = message.getHeaders(lastHeader);
423                     int lastPos = current.size() - 1;
424                     String newString = current.get(lastPos) + line.trim();
425                     current.set(lastPos, newString);
426                 } else {
427                     String[] header = splitHeader(line);
428                     message.addHeader(header[0], header[1]);
429                     lastHeader = header[0];
430                 }
431 
432                 line = readHeader(buffer);
433             } while (line.length() != 0);
434         }
435 
436         State nextState;
437 
438         if (isContentAlwaysEmpty(message)) {
439             nextState = State.SKIP_CONTROL_CHARS;
440         } else if (message.isChunked()) {
441             nextState = State.READ_CHUNK_SIZE;
442         } else if (message.getContentLength(-1) >= 0) {
443             nextState = State.READ_FIXED_LENGTH_CONTENT;
444         } else {
445             nextState = State.READ_VARIABLE_LENGTH_CONTENT;
446         }
447         return nextState;
448     }

450     private String readHeader(ChannelBuffer buffer) throws TooLongFrameException {
451         StringBuilder sb = new StringBuilder(64);
452         int headerSize = this.headerSize;
453 
454         loop:
455         for (;;) {
456             char nextByte = (char) buffer.readByte();
457             headerSize ++;
458 
459             switch (nextByte) {
460             case HttpCodecUtil.CR:
461                 nextByte = (char) buffer.readByte();
462                 headerSize ++;
463                 if (nextByte == HttpCodecUtil.LF) {
464                     break loop;
465                 }
466                 break;
467             case HttpCodecUtil.LF:
468                 break loop;
469             }
470 
471             // Abort decoding if the header part is too large.
472             if (headerSize >= maxHeaderSize) {
473                 throw new TooLongFrameException(
474                         "HTTP header is larger than " +
475                         maxHeaderSize + " bytes.");
476 
477             }
478 
479             sb.append(nextByte);
480         }
481 
482         this.headerSize = headerSize;
483         return sb.toString();
484     }

489     private int getChunkSize(String hex) {
490         hex = hex.trim();
491         for (int i = 0; i < hex.length(); i ++) {
492             char c = hex.charAt(i);
493             if (c == ';' || Character.isWhitespace(c) || Character.isISOControl(c)) {
494                 hex = hex.substring(0, i);
495                 break;
496             }
497         }
498 
499         return Integer.parseInt(hex, 16);
500     }

502     private String readLine(ChannelBuffer buffer, int maxLineLength) throws TooLongFrameException {
503         StringBuilder sb = new StringBuilder(64);
504         int lineLength = 0;
505         while (true) {
506             byte nextByte = buffer.readByte();
507             if (nextByte == HttpCodecUtil.CR) {
508                 nextByte = buffer.readByte();
509                 if (nextByte == HttpCodecUtil.LF) {
510                     return sb.toString();
511                 }
512             }
513             else if (nextByte == HttpCodecUtil.LF) {
514                 return sb.toString();
515             }
516             else {
517                 if (lineLength >= maxLineLength) {
518                     throw new TooLongFrameException(
519                             "An HTTP line is larger than " + maxLineLength +
520                             " bytes.");
521                 }
522                 lineLength ++;
523                 sb.append((char) nextByte);
524             }
525         }
526     }

528     /**
529      * Returns {@code true} if only if the skipped line was not empty.
530      * Please note that an empty line is also skipped, while {@code} false is
531      * returned.
532      */
533     private boolean skipLine(ChannelBuffer buffer) {
534         int lineLength = 0;
535         while (true) {
536             byte nextByte = buffer.readByte();
537             if (nextByte == HttpCodecUtil.CR) {
538                 nextByte = buffer.readByte();
539                 if (nextByte == HttpCodecUtil.LF) {
540                     return lineLength != 0;
541                 }
542             }
543             else if (nextByte == HttpCodecUtil.LF) {
544                 return lineLength != 0;
545             }
546             else if (!Character.isWhitespace((char) nextByte)) {
547                 lineLength ++;
548             }
549         }
550     }

552     private String[] splitInitialLine(String sb) {
553         int aStart;
554         int aEnd;
555         int bStart;
556         int bEnd;
557         int cStart;
558         int cEnd;
559 
560         aStart = findNonWhitespace(sb, 0);
561         aEnd = findWhitespace(sb, aStart);
562 
563         bStart = findNonWhitespace(sb, aEnd);
564         bEnd = findWhitespace(sb, bStart);
565 
566         cStart = findNonWhitespace(sb, bEnd);
567         cEnd = findEndOfString(sb);
568 
569         return new String[] {
570                 sb.substring(aStart, aEnd),
571                 sb.substring(bStart, bEnd),
572                 sb.substring(cStart, cEnd) };
573     }

575     private String[] splitHeader(String sb) {
576         int nameStart;
577         int nameEnd;
578         int colonEnd;
579         int valueStart;
580         int valueEnd;
581 
582         nameStart = findNonWhitespace(sb, 0);
583         for (nameEnd = nameStart; nameEnd < sb.length(); nameEnd ++) {
584             char ch = sb.charAt(nameEnd);
585             if (ch == ':' || Character.isWhitespace(ch)) {
586                 break;
587             }
588         }
589 
590         for (colonEnd = nameEnd; colonEnd < sb.length(); colonEnd ++) {
591             if (sb.charAt(colonEnd) == ':') {
592                 colonEnd ++;
593                 break;
594             }
595         }
596 
597         valueStart = findNonWhitespace(sb, colonEnd);
598         valueEnd = findEndOfString(sb);
599 
600         return new String[] {
601                 sb.substring(nameStart, nameEnd),
602                 sb.substring(valueStart, valueEnd) };
603     }
604 
605     private int findNonWhitespace(String sb, int offset) {
606         int result;
607         for (result = offset; result < sb.length(); result ++) {
608             if (!Character.isWhitespace(sb.charAt(result))) {
609                 break;
610             }
611         }
612         return result;
613     }
614 
615     private int findWhitespace(String sb, int offset) {
616         int result;
617         for (result = offset; result < sb.length(); result ++) {
618             if (Character.isWhitespace(sb.charAt(result))) {
619                 break;
620             }
621         }
622         return result;
623     }
624 
625     private int findEndOfString(String sb) {
626         int result;
627         for (result = sb.length(); result > 0; result --) {
628             if (!Character.isWhitespace(sb.charAt(result - 1))) {
629                 break;
630             }
631         }
632         return result;
633     }
