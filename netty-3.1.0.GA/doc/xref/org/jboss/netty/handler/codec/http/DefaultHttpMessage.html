38   * The default {@link HttpMessage} implementation.

45  public class DefaultHttpMessage implements HttpMessage {
46  
47      private final HttpVersion version;
48      private final Map<String, List<String>> headers = new TreeMap<String, List<String>>(CaseIgnoringComparator.INSTANCE);
49      private ChannelBuffer content = ChannelBuffers.EMPTY_BUFFER;

51      /**
52       * Creates a new instance.
53       */
54      protected DefaultHttpMessage(final HttpVersion version) {
55          if (version == null) {
56              throw new NullPointerException("version");
57          }
58          this.version = version;
59      }

61      public void addHeader(final String name, final String value) {
62          validateHeaderName(name);
63          validateHeaderValue(value);
64          if (value == null) {
65              throw new NullPointerException("value is null");
66          }
67          if (headers.get(name) == null) {
68              headers.put(name, new ArrayList<String>(1));
69          }
70          headers.get(name).add(value);
71      }
72  
73      public void setHeader(final String name, final String value) {
74          validateHeaderName(name);
75          validateHeaderValue(value);
76          if (value == null) {
77              throw new NullPointerException("value");
78          }
79  
80          List<String> values = new ArrayList<String>(1);
81          values.add(value);
82          headers.put(name, values);
83      }
84  
85      public void setHeader(final String name, final Iterable<String> values) {
86          validateHeaderName(name);
87          if (values == null) {
88              throw new NullPointerException("values");
89          }
90  
91          int nValues = 0;
92          for (String v: values) {
93              validateHeaderValue(v);
94              nValues ++;
95          }
96  
97          if (nValues == 0) {
98              throw new IllegalArgumentException("values is empty.");
99          }
100 
101         if (values instanceof List<?>) {
102             headers.put(name, (List<String>) values);
103         } else {
104             List<String> valueList = new LinkedList<String>();
105             for (String v: values) {
106                 valueList.add(v);
107             }
108             headers.put(name, valueList);
109         }
110     }

112     private static void validateHeaderName(String name) {
113         if (name == null) {
114             throw new NullPointerException("name");
115         }
116         for (int i = 0; i < name.length(); i ++) {
117             char c = name.charAt(i);
118             if (c > 127) {
119                 throw new IllegalArgumentException(
120                         "name contains non-ascii character: " + name);
121             }
122 
123             // Check prohibited characters.
124             switch (c) {
125             case '=':  case ',':  case ';': case ' ': case ':':
126             case '\t': case '\r': case '\n': case '\f':
127             case 0x0b: // Vertical tab
128                 throw new IllegalArgumentException(
129                         "name contains one of the following prohibited characters: " +
130                         "=,;: \\t\\r\\n\\v\\f: " + name);
131             }
132         }
133     }
134 
135     private static void validateHeaderValue(String value) {
136         if (value == null) {
137             throw new NullPointerException("value");
138         }
139         for (int i = 0; i < value.length(); i ++) {
140             char c = value.charAt(i);
141             // Check prohibited characters.
142             switch (c) {
143             case '\r': case '\n': case '\f':
144             case 0x0b: // Vertical tab
145                 throw new IllegalArgumentException(
146                         "value contains one of the following prohibited characters: " +
147                         "\\r\\n\\v\\f: " + value);
148             }
149         }
150     }

157         return getContentLength(0);

160     public long getContentLength(long defaultValue) {
161         List<String> contentLength = headers.get(HttpHeaders.Names.CONTENT_LENGTH);
162         if (contentLength != null && contentLength.size() > 0) {
163             return Long.parseLong(contentLength.get(0));
164         }
165         return defaultValue;
166     }

168     public boolean isChunked() {
169         List<String> chunked = headers.get(HttpHeaders.Names.TRANSFER_ENCODING);
170         if (chunked == null || chunked.isEmpty()) {
171             return false;
172         }
173 
174         for (String v: chunked) {
175             if (v.equalsIgnoreCase(HttpHeaders.Values.CHUNKED)) {
176                 return true;
177             }
178         }
179         return false;
180     }
181 
182     public boolean isKeepAlive() {
183         if (HttpHeaders.Values.CLOSE.equalsIgnoreCase(getHeader(HttpHeaders.Names.CONNECTION))) {
184             return false;
185         }
186 
187         if (getProtocolVersion().equals(HttpVersion.HTTP_1_0) &&
188             !HttpHeaders.Values.KEEP_ALIVE.equalsIgnoreCase(getHeader(HttpHeaders.Names.CONNECTION))) {
189             return false;
190         }
191         return true;
192     }

205     public String getHeader(final String name) {
206         List<String> header = headers.get(name);
207         return header != null && header.size() > 0 ? headers.get(name).get(0) : null;
208     }

210     public List<String> getHeaders(final String name) {
211         List<String> values = headers.get(name);
212         if (values == null) {
213             return Collections.emptyList();
214         } else {

216         }
217     }
