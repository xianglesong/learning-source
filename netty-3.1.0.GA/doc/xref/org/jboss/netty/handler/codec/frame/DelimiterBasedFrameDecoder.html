23  package org.jboss.netty.handler.codec.frame;

25  import org.jboss.netty.buffer.ChannelBuffer;
26  import org.jboss.netty.channel.Channel;
27  import org.jboss.netty.channel.ChannelHandlerContext;

29  /**
30   * A decoder that splits the received {@link ChannelBuffer}s by one or more
31   * delimiters.  It is particularly useful for decoding the frames which ends
32   * with a delimiter such as {@link Delimiters#nulDelimiter() NUL} or
33   * {@linkplain Delimiters#lineDelimiter() newline characters}.
34   *
35   * <h3>Predefined delimiters</h3>
36   * <p>
37   * {@link Delimiters} defines frequently used delimiters for convenience' sake.
38   *
39   * <h3>Specifying more than one delimiter</h3>
40   * <p>
41   * {@link DelimiterBasedFrameDecoder} allows you to specify more than one
42   * delimiter.  If more than one delimiter is found in the buffer, it chooses
43   * the delimiter which produces the shortest frame.  For example, if you have
44   * the following data in the buffer:
45   * <pre>
46   * +--------------+
47   * | ABC\nDEF\r\n |
48   * +--------------+
49   * </pre>
50   * a {@link DelimiterBasedFrameDecoder}{@code (}{@link Delimiters#lineDelimiter() Delimiters.lineDelimiter()}{@code )}
51   * will choose {@code '\n'} as the first delimiter and produce two frames:
52   * <pre>
53   * +-----+-----+
54   * | ABC | DEF |
55   * +-----+-----+
56   * </pre>
57   * rather than incorrectly choosing {@code '\r\n'} as the first delimiter:
58   * <pre>
59   * +----------+
60   * | ABC\nDEF |
61   * +----------+
62   * </pre>
63   *
64   * @author The Netty Project (netty-dev@lists.jboss.org)
65   * @author Trustin Lee (tlee@redhat.com)
66   *
67   * @version $Rev:231 $, $Date:2008-06-12 16:44:50 +0900 (목, 12 6월 2008) $
68   *
69   * @apiviz.uses org.jboss.netty.handler.codec.frame.Delimiters - - useful
70   */
71  public class DelimiterBasedFrameDecoder extends FrameDecoder {
72  
73      private final ChannelBuffer[] delimiters;
74      private final int maxFrameLength;
75      private final boolean stripDelimiter;
76      private volatile boolean discardingTooLongFrame;
77      private volatile long tooLongFrameLength;
78  
79      /**
80       * Creates a new instance.
81       *
82       * @param maxFrameLength  the maximum length of the decoded frame.
83       *                        A {@link TooLongFrameException} is thrown if
84       *                        the length of the frame exceeds this value.
85       * @param delimiter  the delimiter
86       */
87      public DelimiterBasedFrameDecoder(int maxFrameLength, ChannelBuffer delimiter) {
88          this(maxFrameLength, true, delimiter);
89      }

91      /**
92       * Creates a new instance.
93       *
94       * @param maxFrameLength  the maximum length of the decoded frame.
95       *                        A {@link TooLongFrameException} is thrown if
96       *                        the length of the frame exceeds this value.
97       * @param stripDelimiter  whether the decoded frame should strip out the
98       *                        delimiter or not
99       * @param delimiter  the delimiter
100      */
101     public DelimiterBasedFrameDecoder(
102             int maxFrameLength, boolean stripDelimiter, ChannelBuffer delimiter) {
103         validateMaxFrameLength(maxFrameLength);
104         validateDelimiter(delimiter);
105         delimiters = new ChannelBuffer[] {
106                 delimiter.slice(
107                         delimiter.readerIndex(), delimiter.readableBytes())
108         };
109         this.maxFrameLength = maxFrameLength;
110         this.stripDelimiter = stripDelimiter;
111     }

118      *                        the length of the frame exceeds this value.

122         this(maxFrameLength, true, delimiters);
123     }

125     /**
126      * Creates a new instance.
127      *
128      * @param maxFrameLength  the maximum length of the decoded frame.
129      *                        A {@link TooLongFrameException} is thrown if
130      *                        the length of the frame exceeds this value.
131      * @param stripDelimiter  whether the decoded frame should strip out the
132      *                        delimiter or not
133      * @param delimiters  the delimiters
134      */
135     public DelimiterBasedFrameDecoder(
136             int maxFrameLength, boolean stripDelimiter, ChannelBuffer... delimiters) {
137         validateMaxFrameLength(maxFrameLength);
138         if (delimiters == null) {
139             throw new NullPointerException("delimiters");
140         }
141         if (delimiters.length == 0) {
142             throw new IllegalArgumentException("empty delimiters");
143         }
144         this.delimiters = new ChannelBuffer[delimiters.length];
145         for (int i = 0; i < delimiters.length; i ++) {
146             ChannelBuffer d = delimiters[i];
147             validateDelimiter(d);
148             this.delimiters[i] = d.slice(d.readerIndex(), d.readableBytes());
149         }
150         this.maxFrameLength = maxFrameLength;
151         this.stripDelimiter = stripDelimiter;
152     }
153 
154     @Override
155     protected Object decode(
156             ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer) throws Exception {
157         // Try all delimiters and choose the delimiter which yields the shortest frame.
158         int minFrameLength = Integer.MAX_VALUE;
159         ChannelBuffer minDelim = null;
160         for (ChannelBuffer delim: delimiters) {
161             int frameLength = indexOf(buffer, delim);
162             if (frameLength >= 0 && frameLength < minFrameLength) {
163                 minFrameLength = frameLength;
164                 minDelim = delim;
165             }
166         }
167 
168         if (minDelim != null) {
169             int minDelimLength = minDelim.capacity();
170             ChannelBuffer frame;
171 
172             if (discardingTooLongFrame) {
173                 // We've just finished discarding a very large frame.
174                 // Throw an exception and go back to the initial state.
175                 long tooLongFrameLength = this.tooLongFrameLength;
176                 this.tooLongFrameLength = 0L;
177                 discardingTooLongFrame = false;
178                 buffer.skipBytes(minFrameLength + minDelimLength);
179                 fail(tooLongFrameLength + minFrameLength + minDelimLength);
180             }
181 
182             if (minFrameLength > maxFrameLength) {
183                 // Discard read frame.
184                 buffer.skipBytes(minFrameLength + minDelimLength);
185                 fail(minFrameLength);
186             }
187 
188             if (stripDelimiter) {
189                 frame = buffer.readBytes(minFrameLength);
190                 buffer.skipBytes(minDelimLength);
191             } else {
192                 frame = buffer.readBytes(minFrameLength + minDelimLength);
193             }
194 
195             return frame;
196         } else {
197             if (buffer.readableBytes() > maxFrameLength) {
198                 // Discard the content of the buffer until a delimiter is found.
199                 tooLongFrameLength = buffer.readableBytes();
200                 buffer.skipBytes(buffer.readableBytes());
201                 discardingTooLongFrame = true;
202             }
203 
204             return null;
205         }
206     }
207 
208     private void fail(long frameLength) throws TooLongFrameException {
209         throw new TooLongFrameException(
210                 "The frame length exceeds " + maxFrameLength + ": " + frameLength);
211     }
212 
213     /**
214      * Returns the number of bytes between the readerIndex of the haystack and
215      * the first needle found in the haystack.  -1 is returned if no needle is
216      * found in the haystack.
217      */
218     private static int indexOf(ChannelBuffer haystack, ChannelBuffer needle) {
219         for (int i = haystack.readerIndex(); i < haystack.writerIndex(); i ++) {
220             int haystackIndex = i;
221             int needleIndex;
222             for (needleIndex = 0; needleIndex < needle.capacity(); needleIndex ++) {
223                 if (haystack.getByte(haystackIndex) != needle.getByte(needleIndex)) {
224                     break;
225                 } else {
226                     haystackIndex ++;
227                     if (haystackIndex == haystack.writerIndex() &&
228                         needleIndex != needle.capacity() - 1) {
229                         return -1;
230                     }
231                 }
232             }
233 
234             if (needleIndex == needle.capacity()) {
235                 // Found the needle from the haystack!
236                 return i - haystack.readerIndex();
237             }
238         }
239         return -1;
240     }
241 
242     private static void validateDelimiter(ChannelBuffer delimiter) {
243         if (delimiter == null) {
244             throw new NullPointerException("delimiter");
245         }
246         if (!delimiter.readable()) {
247             throw new IllegalArgumentException("empty delimiter");
248         }
249     }
250 
251     private static void validateMaxFrameLength(int maxFrameLength) {
252         if (maxFrameLength <= 0) {
253             throw new IllegalArgumentException(
254                     "maxFrameLength must be a positive integer: " +
255                     maxFrameLength);
256         }
257     }
