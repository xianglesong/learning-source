161     public static ChannelBuffer encode(
162             ChannelBuffer src, int off, int len,
163             boolean breakLines, Base64Dialect dialect, ChannelBufferFactory bufferFactory) {
164 
165         if (src == null) {
166             throw new NullPointerException("src");
167         }
168         if (dialect == null) {
169             throw new NullPointerException("dialect");
170         }
171         if (bufferFactory == null) {
172             throw new NullPointerException("bufferFactory");
173         }
174 
175         int len43 = len * 4 / 3;
176         ChannelBuffer dest = bufferFactory.getBuffer(
177                 len43 +
178                 (len % 3 > 0? 4 : 0) + // Account for padding
179                 (breakLines? len43 / MAX_LINE_LENGTH : 0)); // New lines
180         int d = 0;
181         int e = 0;
182         int len2 = len - 2;
183         int lineLength = 0;
184         for (; d < len2; d += 3, e += 4) {
185             encode3to4(src, d + off, 3, dest, e, dialect);
186 
187             lineLength += 4;
188             if (breakLines && lineLength == MAX_LINE_LENGTH) {
189                 dest.setByte(e + 4, NEW_LINE);
190                 e ++;
191                 lineLength = 0;
192             } // end if: end of line
193         } // end for: each piece of array
194 
195         if (d < len) {
196             encode3to4(src, d + off, len - d, dest, e, dialect);
197             e += 4;
198         } // end if: some padding needed
199 
200         return dest.slice(0, e);
201     }

203     private static void encode3to4(
204             ChannelBuffer src, int srcOffset, int numSigBytes,
205             ChannelBuffer dest, int destOffset, Base64Dialect dialect) {
206 
207         byte[] ALPHABET = alphabet(dialect);
208 
209         //           1         2         3
210         // 01234567890123456789012345678901 Bit position
211         // --------000000001111111122222222 Array position from threeBytes
212         // --------|    ||    ||    ||    | Six bit groups to index ALPHABET
213         //          >>18  >>12  >> 6  >> 0  Right shift necessary
214         //                0x3f  0x3f  0x3f  Additional AND
215 
216         // Create buffer with zero-padding if there are only one or two
217         // significant bytes passed in the array.
218         // We have to shift left 24 in order to flush out the 1's that appear
219         // when Java treats a value as negative that is cast from a byte to an int.
220         int inBuff =
221                 (numSigBytes > 0? src.getByte(srcOffset    ) << 24 >>>  8 : 0) |
222                 (numSigBytes > 1? src.getByte(srcOffset + 1) << 24 >>> 16 : 0) |
223                 (numSigBytes > 2? src.getByte(srcOffset + 2) << 24 >>> 24 : 0);
224 
225         switch (numSigBytes) {
226         case 3:
227             dest.setByte(destOffset    , ALPHABET[inBuff >>> 18       ]);
228             dest.setByte(destOffset + 1, ALPHABET[inBuff >>> 12 & 0x3f]);
229             dest.setByte(destOffset + 2, ALPHABET[inBuff >>>  6 & 0x3f]);
230             dest.setByte(destOffset + 3, ALPHABET[inBuff        & 0x3f]);
231             break;
232         case 2:
233             dest.setByte(destOffset    , ALPHABET[inBuff >>> 18       ]);
234             dest.setByte(destOffset + 1, ALPHABET[inBuff >>> 12 & 0x3f]);
235             dest.setByte(destOffset + 2, ALPHABET[inBuff >>> 6  & 0x3f]);
236             dest.setByte(destOffset + 3, EQUALS_SIGN);
237             break;
238         case 1:
239             dest.setByte(destOffset    , ALPHABET[inBuff >>> 18       ]);
240             dest.setByte(destOffset + 1, ALPHABET[inBuff >>> 12 & 0x3f]);
241             dest.setByte(destOffset + 2, EQUALS_SIGN);
242             dest.setByte(destOffset + 3, EQUALS_SIGN);
243             break;
244         }
245     }

285     public static ChannelBuffer decode(
286             ChannelBuffer src, int off, int len, Base64Dialect dialect,
287             ChannelBufferFactory bufferFactory) {
288 
289         if (src == null) {
290             throw new NullPointerException("src");
291         }
292         if (dialect == null) {
293             throw new NullPointerException("dialect");
294         }
295         if (bufferFactory == null) {
296             throw new NullPointerException("bufferFactory");
297         }
298 
299         byte[] DECODABET = decodabet(dialect);
300 
301         int len34 = len * 3 / 4;
302         ChannelBuffer dest = bufferFactory.getBuffer(len34); // Upper limit on size of output
303         int outBuffPosn = 0;
304 
305         byte[] b4 = new byte[4];
306         int b4Posn = 0;
307         int i = 0;
308         byte sbiCrop = 0;
309         byte sbiDecode = 0;
310         for (i = off; i < off + len; i ++) {
311             sbiCrop = (byte) (src.getByte(i) & 0x7f); // Only the low seven bits
312             sbiDecode = DECODABET[sbiCrop];
313 
314             if (sbiDecode >= WHITE_SPACE_ENC) { // White space, Equals sign or better
315                 if (sbiDecode >= EQUALS_SIGN_ENC) {
316                     b4[b4Posn ++] = sbiCrop;
317                     if (b4Posn > 3) {
318                         outBuffPosn += decode4to3(
319                                 b4, 0, dest, outBuffPosn, dialect);
320                         b4Posn = 0;
321 
322                         // If that was the equals sign, break out of 'for' loop
323                         if (sbiCrop == EQUALS_SIGN) {
324                             break;
325                         }
326                     } // end if: quartet built
327                 } // end if: equals sign or better
328             } // end if: white space, equals sign or better
329             else {
330                 throw new IllegalArgumentException(
331                         "bad Base64 input character at " + i + ": " +
332                         src.getUnsignedByte(i) + " (decimal)");
333             }
334         }
335 
336         return dest.slice(0, outBuffPosn);
337     }

339     private static int decode4to3(
340             byte[] src, int srcOffset,
341             ChannelBuffer dest, int destOffset, Base64Dialect dialect) {
342 
343         byte[] DECODABET = decodabet(dialect);
344 
345         // Example: Dk==
346         if (src[srcOffset + 2] == EQUALS_SIGN) {
347             int outBuff =
348                     (DECODABET[src[srcOffset    ]] & 0xFF) << 18 |
349                     (DECODABET[src[srcOffset + 1]] & 0xFF) << 12;
350 
351             dest.setByte(destOffset, (byte) (outBuff >>> 16));
352             return 1;
353         }
354 
355         // Example: DkL=
356         else if (src[srcOffset + 3] == EQUALS_SIGN) {
357             int outBuff =
358                     (DECODABET[src[srcOffset    ]] & 0xFF) << 18 |
359                     (DECODABET[src[srcOffset + 1]] & 0xFF) << 12 |
360                     (DECODABET[src[srcOffset + 2]] & 0xFF) <<  6;
361 
362             dest.setByte(destOffset    , (byte) (outBuff >>> 16));
363             dest.setByte(destOffset + 1, (byte) (outBuff >>>  8));
364             return 2;
365         }
366 
367         // Example: DkLE
368         else {
369         	int outBuff;
370             try {
371                 outBuff =
372                         (DECODABET[src[srcOffset    ]] & 0xFF) << 18 |
373                         (DECODABET[src[srcOffset + 1]] & 0xFF) << 12 |
374                         (DECODABET[src[srcOffset + 2]] & 0xFF) <<  6 |
375                          DECODABET[src[srcOffset + 3]] & 0xFF;
376             } catch (IndexOutOfBoundsException e) {
377                 throw new IllegalArgumentException("not encoded in Base64");
378             }
379 
380             dest.setByte(destOffset    , (byte) (outBuff >> 16));
381             dest.setByte(destOffset + 1, (byte) (outBuff >>  8));
382             dest.setByte(destOffset + 2, (byte)  outBuff);
383             return 3;
384         }
385     }
