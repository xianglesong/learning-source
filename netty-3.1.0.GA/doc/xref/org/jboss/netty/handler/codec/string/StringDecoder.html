37  /**
38   * Decodes a received {@link ChannelBuffer} into a {@link String}.  Please
39   * note that this decoder must be used with a proper {@link FrameDecoder}
40   * such as {@link DelimiterBasedFrameDecoder} if you are using a stream-based
41   * transport such as TCP/IP.  A typical setup for a text-based line protocol
42   * in a TCP/IP socket would be:
43   * <pre>
44   * {@link ChannelPipeline} pipeline = ...;
45   *
46   * // Decoders
47   * pipeline.addLast("frameDecoder", new {@link DelimiterBasedFrameDecoder}(80, {@link Delimiters#lineDelimiter()}));
48   * pipeline.addLast("stringDecoder", new {@link StringDecoder}("UTF-8"));
49   *
50   * // Encoder
51   * pipeline.addLast("stringEncoder", new {@link StringEncoder}("UTF-8"));
52   * </pre>
53   * and then you can use a {@link String} instead of a {@link ChannelBuffer}
54   * as a message:
55   * <pre>
56   * void messageReceived(ChannelHandlerContext ctx, MessageEvent e) {
57   *     String msg = (String) e.getMessage();
58   *     ch.write("Did you say '" + msg + "'?\n");
59   * }
60   * </pre>
61   *
62   * @author The Netty Project (netty-dev@lists.jboss.org)
63   * @author Trustin Lee (tlee@redhat.com)
64   *
65   * @version $Rev:231 $, $Date:2008-06-12 16:44:50 +0900 (목, 12 6월 2008) $
66   *
67   * @apiviz.landmark
68   */
69  @ChannelPipelineCoverage("all")
70  public class StringDecoder extends OneToOneDecoder {
71  
72      private final String charsetName;
73  
74      /**
75       * Creates a new instance with the current system character set.
76       */
77      public StringDecoder() {
78          this(Charset.defaultCharset());
79      }
80  
81      /**
82       * Creates a new instance.
83       *
84       * @param charsetName  the name of the character set to use for decoding
85       */
86      public StringDecoder(String charsetName) {
87          this(Charset.forName(charsetName));
88      }
89  
90      /**
91       * Creates a new instance.
92       *
93       * @param charset  the character set to use for decoding
94       */
95      public StringDecoder(Charset charset) {
96          if (charset == null) {
97              throw new NullPointerException("charset");
98          }
99          charsetName = charset.name();
100     }
101 
102     @Override
103     protected Object decode(
104             ChannelHandlerContext ctx, Channel channel, Object msg) throws Exception {
105         if (!(msg instanceof ChannelBuffer)) {
106             return msg;
107         }
108         return ((ChannelBuffer) msg).toString(charsetName);
109     }
110 }
