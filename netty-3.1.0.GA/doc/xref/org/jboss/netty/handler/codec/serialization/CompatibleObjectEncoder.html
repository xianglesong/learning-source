23  package org.jboss.netty.handler.codec.serialization;
24  
25  import java.io.ObjectInputStream;
26  import java.io.ObjectOutputStream;
27  import java.io.OutputStream;
28  import java.util.concurrent.atomic.AtomicReference;
29  
30  import org.jboss.netty.buffer.ChannelBuffer;
31  import org.jboss.netty.buffer.ChannelBufferFactory;
32  import org.jboss.netty.buffer.ChannelBufferOutputStream;
33  import org.jboss.netty.buffer.ChannelBuffers;
34  import org.jboss.netty.channel.Channel;
35  import org.jboss.netty.channel.ChannelHandlerContext;
36  import org.jboss.netty.channel.ChannelPipelineCoverage;
37  import org.jboss.netty.handler.codec.oneone.OneToOneEncoder;
38  
39  /**
40   * An encoder which serializes a Java object into a {@link ChannelBuffer}
41   * (interoperability version).
42   * <p>
43   * This encoder is interoperable with the standard Java object streams such as
44   * {@link ObjectInputStream} and {@link ObjectOutputStream}.
45   *
46   * @author The Netty Project (netty-dev@lists.jboss.org)
47   * @author Trustin Lee (tlee@redhat.com)
48   *
49   * @version $Rev:231 $, $Date:2008-06-12 16:44:50 +0900 (목, 12 6월 2008) $
50   */
51  @ChannelPipelineCoverage("one")
52  public class CompatibleObjectEncoder extends OneToOneEncoder {
53  
54      private final AtomicReference<ChannelBuffer> buffer =
55          new AtomicReference<ChannelBuffer>();
56      private final int resetInterval;
57      private volatile ObjectOutputStream oout;
58      private int writtenObjects;
59  
60      /**
61       * Creates a new instance with the reset interval of {@code 16}.
62       */
63      public CompatibleObjectEncoder() {
64          this(16); // Reset at every sixteen writes
65      }
66  
67      /**
68       * Creates a new instance.
69       *
70       * @param resetInterval
71       *        the number of objects between {@link ObjectOutputStream#reset()}.
72       *        {@code 0} will disable resetting the stream, but the remote
73       *        peer will be at the risk of getting {@link OutOfMemoryError} in
74       *        the long term.
75       */
76      public CompatibleObjectEncoder(int resetInterval) {
77          if (resetInterval < 0) {
78              throw new IllegalArgumentException(
79                      "resetInterval: " + resetInterval);
80          }
81          this.resetInterval = resetInterval;
82      }
83  
84      /**
85       * Creates a new {@link ObjectOutputStream} which wraps the specified
86       * {@link OutputStream}.  Override this method to use a subclass of the
87       * {@link ObjectOutputStream}.
88       */
89      protected ObjectOutputStream newObjectOutputStream(OutputStream out) throws Exception {
90          return new ObjectOutputStream(out);
91      }
92  
93      @Override
94      protected Object encode(ChannelHandlerContext context, Channel channel, Object msg) throws Exception {
95          ChannelBuffer buffer = buffer(context);
96          ObjectOutputStream oout = this.oout;
97          if (resetInterval != 0) {
98              // Resetting will prevent OOM on the receiving side.
99              writtenObjects ++;
100             if (writtenObjects % resetInterval == 0) {
101                 oout.reset();
102 
103                 // Also discard the byproduct to avoid OOM on the sending side.
104                 buffer.discardReadBytes();
105             }
106         }
107         oout.writeObject(msg);
108         oout.flush();
109 
110         ChannelBuffer encoded = buffer.readBytes(buffer.readableBytes());
111         return encoded;
112     }
113 
114     private ChannelBuffer buffer(ChannelHandlerContext ctx) throws Exception {
115         ChannelBuffer buf = buffer.get();
116         if (buf == null) {
117             ChannelBufferFactory factory = ctx.getChannel().getConfig().getBufferFactory();
118             buf = ChannelBuffers.dynamicBuffer(factory);
119             if (buffer.compareAndSet(null, buf)) {
120                 boolean success = false;
121                 try {
122                     oout = newObjectOutputStream(new ChannelBufferOutputStream(buf));
123                     success = true;
124                 } finally {
125                     if (!success) {
126                         oout = null;
127                     }
128                 }
129             } else {
130                 buf = buffer.get();
131             }
132         }
133         return buf;
134     }
