27  /**
28   * A large data stream which is consumed by {@link ChunkedWriteHandler}.
29   *
30   * @author The Netty Project (netty-dev@lists.jboss.org)
31   * @author Trustin Lee (tlee@redhat.com)
32   * @version $Rev: 1587 $, $Date: 2009-07-20 13:50:03 +0900 (Mon, 20 Jul 2009) $
33   *
34   * @apiviz.landmark
35   */
36  public interface ChunkedInput {
37  
38      /**
39       * Returns {@code true} if and only if there is any data left in the
40       * stream.
41       */
42      boolean hasNextChunk() throws Exception;
43  
44      /**
45       * Fetches a chunked data from the stream.  The returned chunk is usually
46       * a {@link ChannelBuffer}, but you could extend an existing implementation
47       * to convert the {@link ChannelBuffer} into a different type that your
48       * handler or encoder understands.
49       *
50       * @return the fetched chunk, which is usually {@link ChannelBuffer}.
51       *         {@code null} if there is no data left in the stream.
52       */
53      Object nextChunk() throws Exception;
54  
55      /**
56       * Releases the resources associated with the stream.
57       */
58      void close() throws Exception;
59  }
