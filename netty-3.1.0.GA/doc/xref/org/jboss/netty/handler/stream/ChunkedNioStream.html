23  package org.jboss.netty.handler.stream;
24  
25  import static org.jboss.netty.buffer.ChannelBuffers.*;
26  
27  import java.nio.ByteBuffer;
28  import java.nio.channels.ReadableByteChannel;
29  
30  import org.jboss.netty.buffer.ChannelBuffer;
31  
32  /**
33   * A {@link ChunkedInput} that fetches data from a {@link ReadableByteChannel}
34   * chunk by chunk.  Please note that the {@link ReadableByteChannel} must
35   * operate in blocking mode.  Non-blocking mode channels are not supported.
36   *
37   * @author The Netty Project (netty-dev@lists.jboss.org)
38   * @author Trustin Lee (tlee@redhat.com)
39   * @author Frederic Bregier
40   * @version $Rev: 1496 $, $Date: 2009-06-25 16:45:15 +0900 (Thu, 25 Jun 2009) $
41   */
42  public class ChunkedNioStream implements ChunkedInput {
43  
44      private final ReadableByteChannel in;
45  
46      private final int chunkSize;
47      private volatile long offset;
48  
49      /**
50       * Associated ByteBuffer
51       */
52      private ByteBuffer byteBuffer = null;
53  
54      /**
55       * Creates a new instance that fetches data from the specified channel.
56       */
57      public ChunkedNioStream(ReadableByteChannel in) {
58          this(in, ChunkedStream.DEFAULT_CHUNK_SIZE);
59      }
60  
61      /**
62       * Creates a new instance that fetches data from the specified channel.
63       *
64       * @param chunkSize the number of bytes to fetch on each
65       *                  {@link #nextChunk()} call
66       */
67      public ChunkedNioStream(ReadableByteChannel in, int chunkSize) {
68          if (in == null) {
69              throw new NullPointerException("in");
70          }
71          if (chunkSize <= 0) {
72              throw new IllegalArgumentException("chunkSize: " + chunkSize +
73                      " (expected: a positive integer)");
74          }
75          this.in = in;
76          offset = 0;
77          this.chunkSize = chunkSize;
78          if (byteBuffer != null) {
79              if (byteBuffer.capacity() != chunkSize) {
80                  byteBuffer = null;
81                  byteBuffer = ByteBuffer.allocate(chunkSize);
82              }
83          } else {
84              byteBuffer = ByteBuffer.allocate(chunkSize);
85          }
86      }
87  
88      /**
89       * Returns the number of transferred bytes.
90       */
91      public long getTransferredBytes() {
92          return offset;
93      }
94  
95      public boolean hasNextChunk() throws Exception {
96          if (byteBuffer.position() > 0) {
97              // A previous read was not over, so there is a next chunk in the buffer at least
98              return true;
99          }
100         if (in.isOpen()) {
101             // Try to read a new part, and keep this part (no rewind)
102             int b = in.read(byteBuffer);
103             if (b < 0) {
104                 return false;
105             } else {
106                 offset += b;
107                 return true;
108             }
109         }
110         return false;
111     }
112 
113     public void close() throws Exception {
114         in.close();
115     }
116 
117     public Object nextChunk() throws Exception {
118         if (!hasNextChunk()) {
119             return null;
120         }
121         // buffer cannot be not be empty from there
122         int readBytes = byteBuffer.position();
123         for (;;) {
124             int localReadBytes = in.read(byteBuffer);
125             if (localReadBytes < 0) {
126                 break;
127             }
128             readBytes += localReadBytes;
129             offset += localReadBytes;
130 
131             if (readBytes == chunkSize) {
132                 break;
133             }
134         }
135         byteBuffer.flip();
136         // copy since buffer is keeped for next usage
137         ChannelBuffer buffer = copiedBuffer(byteBuffer);
138         byteBuffer.clear();
139         return buffer;
140     }
