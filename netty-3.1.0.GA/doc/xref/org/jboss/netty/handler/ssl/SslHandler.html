61  /**
62   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
63   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
64   * to the <strong>"SecureChat"</strong> example in the distribution or the web
65   * site for the detailed usage.
66   *
67   * <h3>Beginning the handshake</h3>
68   * <p>
69   * You must make sure not to write a message while the
70   * {@linkplain #handshake(Channel) handshake} is in progress unless you are
71   * renegotiating.  You will be notified by the {@link ChannelFuture} which is
72   * returned by the {@link #handshake(Channel)} method when the handshake
73   * process succeeds or fails.
74   *
75   * <h3>Renegotiation</h3>
76   * <p>
77   * Once the initial handshake is done successfully.  You can always call
78   * {@link #handshake(Channel)} again to renegotiate the SSL session parameters.
79   *
80   * <h3>Closing the session</h3>
81   * <p>
82   * To close the SSL session, the {@link #close(Channel)} method should be
83   * called to send the {@code close_notify} message to the remote peer.  One
84   * exception is when you close the {@link Channel} - {@link SslHandler}
85   * intercepts the close request and send the {@code close_notify} message
86   * before the channel closure automatically.  Once the SSL session is closed,
87   * it is not reusable, and consequently you should create a new
88   * {@link SslHandler} with a new {@link SSLEngine} as explained in the
89   * following section.
90   *
91   * <h3>Restarting the session</h3>
92   * <p>
93   * To restart the SSL session, you must remove the existing closed
94   * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
95   * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
96   * and start the handshake process as described in the first section.
97   *
98   * <h3>Implementing StartTLS</h3>
99   * <p>
100  * Typically, StartTLS is composed of three steps.
101  * <ol>
102  * <li>Client sends a StartTLS request to server.</li>
103  * <li>Server sends a StartTLS response to client.</li>
104  * <li>Client begins SSL handshake.</li>
105  * </ol>
106  * If you implement a server, you need to:
107  * <ol>
108  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
109  *     to {@code true},</li>
110  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
111  * <li>write a StartTLS response.</li>
112  * </ol>
113  * Please note that you must insert {@link SslHandler} <em>before</em> sending
114  * the StartTLS response.  Otherwise the client can send begin SSL handshake
115  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
116  * data corruption.
117  * <p>
118  * The client-side implementation is much simpler.
119  * <ol>
120  * <li>Write a StartTLS request,</li>
121  * <li>wait for the StartTLS response,</li>
122  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
123  *     to {@code false},</li>
124  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
125  * <li>Initiate SSL handshake by calling {@link SslHandler#handshake(Channel)}.</li>
126  * </ol>
127  *
128  * @author The Netty Project (netty-dev@lists.jboss.org)
129  * @author Trustin Lee (tlee@redhat.com)
130  *
131  * @version $Rev: 1590 $, $Date: 2009-07-20 14:35:53 +0900 (Mon, 20 Jul 2009) $
132  *
133  * @apiviz.landmark
134  * @apiviz.uses org.jboss.netty.handler.ssl.SslBufferPool
135  */

305     /**
306      * Starts an SSL / TLS handshake for the specified channel.
307      *
308      * @return a {@link ChannelFuture} which is notified when the handshake
309      *         succeeds or fails.
310      */
311     public ChannelFuture handshake(Channel channel) throws SSLException {
312         ChannelFuture handshakeFuture;
313         synchronized (handshakeLock) {
314             if (handshaking) {
315                 return this.handshakeFuture;
316             } else {
317                 engine.beginHandshake();
318                 runDelegatedTasks();
319                 handshakeFuture = this.handshakeFuture = future(channel);
320                 handshaking = true;
321             }
322         }
323 
324         wrapNonAppData(context(channel), channel);
325         return handshakeFuture;
326     }

342     public void handleDownstream(
343             final ChannelHandlerContext context, final ChannelEvent evt) throws Exception {
344         if (evt instanceof ChannelStateEvent) {
345             ChannelStateEvent e = (ChannelStateEvent) evt;
346             switch (e.getState()) {
347             case OPEN:
348             case CONNECTED:
349             case BOUND:
350                 if (Boolean.FALSE.equals(e.getValue()) || e.getValue() == null) {
351                     closeOutboundAndChannel(context, e);
352                     return;
353                 }
354             }
355         }
356         if (!(evt instanceof MessageEvent)) {
357             context.sendDownstream(evt);
358             return;
359         }
360 
361         MessageEvent e = (MessageEvent) evt;
362         if (!(e.getMessage() instanceof ChannelBuffer)) {
363             context.sendDownstream(evt);
364             return;
365         }
366 
367         // Do not encrypt the first write request if this handler is
368         // created with startTLS flag turned on.
369         if (startTls && sentFirstMessage.compareAndSet(false, true)) {
370             context.sendDownstream(evt);
371             return;
372         }
373 
374         // Otherwise, all messages are encrypted.
375         ChannelBuffer msg = (ChannelBuffer) e.getMessage();
376         PendingWrite pendingWrite;
377 
378         if (msg.readable()) {
379             pendingWrite = new PendingWrite(evt.getFuture(), msg.toByteBuffer(msg.readerIndex(), msg.readableBytes()));
380         } else {
381             pendingWrite = new PendingWrite(evt.getFuture(), null);
382         }
383         synchronized (pendingUnencryptedWrites) {
384             boolean offered = pendingUnencryptedWrites.offer(pendingWrite);
385             assert offered;
386         }
387 
388         wrap(context, evt.getChannel());
389     }

391     @Override
392     public void channelDisconnected(ChannelHandlerContext ctx,
393             ChannelStateEvent e) throws Exception {
394 
395         // Make sure the handshake future is notified when a connection has
396         // been closed during handshake.
397         synchronized (handshakeLock) {
398             if (handshaking) {
399                 handshakeFuture.setFailure(new ClosedChannelException());
400             }
401         }
402 
403         try {
404             super.channelDisconnected(ctx, e);
405         } finally {
406             unwrap(ctx, e.getChannel(), ChannelBuffers.EMPTY_BUFFER, 0, 0);
407             engine.closeOutbound();
408             if (!sentCloseNotify.get() && handshaken) {
409                 try {
410                     engine.closeInbound();
411                 } catch (SSLException ex) {
412                     logger.debug("Failed to clean up SSLEngine.", ex);
413                 }
414             }
415         }
416     }

418     @Override
419     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
420             throws Exception {
421 
422         Throwable cause = e.getCause();
423         if (cause instanceof IOException) {
424             if (cause instanceof ClosedChannelException) {
425                 synchronized (ignoreClosedChannelExceptionLock) {
426                     if (ignoreClosedChannelException > 0) {
427                         ignoreClosedChannelException --;
428                         logger.debug(
429                                 "Swallowing an exception raised while " +
430                                 "writing non-app data", cause);
431                         return;
432                     }
433                 }
434             } else if (engine.isOutboundDone()) {
435                 String message = String.valueOf(cause.getMessage()).toLowerCase();
436                 if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
437                     // It is safe to ignore the 'connection reset by peer' or
438                     // 'broken pipe' error after sending closure_notify.
439                     logger.debug(
440                             "Swallowing a 'connection reset by peer / " +
441                             "broken pipe' error occurred while writing " +
442                             "'closure_notify'", cause);
443 
444                     // Close the connection explicitly just in case the transport
445                     // did not close the connection automatically.
446                     Channels.close(ctx, succeededFuture(e.getChannel()));
447                     return;
448                 }
449             }
450         }
451 
452         ctx.sendUpstream(e);
453     }

455     @Override
456     protected Object decode(
457             final ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer) throws Exception {
458         if (buffer.readableBytes() < 2) {
459             return null;
460         }
461 
462         int packetLength = buffer.getShort(buffer.readerIndex()) & 0xFFFF;
463         if ((packetLength & 0x8000) != 0) {
464             // Detected a SSLv2 packet
465             packetLength &= 0x7FFF;
466             packetLength += 2;
467         } else  if (buffer.readableBytes() < 5) {
468             return null;
469         } else {
470             // Detected a SSLv3 / TLSv1 packet
471             packetLength = (buffer.getShort(buffer.readerIndex() + 3) & 0xFFFF) + 5;
472         }
473 
474         if (buffer.readableBytes() < packetLength) {
475             return null;
476         }
477 
478         ChannelBuffer frame;
479         try {
480             frame = unwrap(ctx, channel, buffer, buffer.readerIndex(), packetLength);
481         } finally {
482             buffer.skipBytes(packetLength);
483         }
484 
485         return frame;
486     }

488     private ChannelFuture wrap(ChannelHandlerContext context, Channel channel)
489             throws SSLException {
490 
491         ChannelFuture future = null;
492         ChannelBuffer msg;
493         ByteBuffer outNetBuf = bufferPool.acquire();
494         boolean success = true;
495         boolean offered = false;
496         boolean needsUnwrap = false;
497         try {
498             loop:
499             for (;;) {
500                 // Acquire a lock to make sure unencrypted data is polled
501                 // in order and their encrypted counterpart is offered in
502                 // order.
503                 synchronized (pendingUnencryptedWrites) {
504                     PendingWrite pendingWrite = pendingUnencryptedWrites.peek();
505                     if (pendingWrite == null) {
506                         break;
507                     }
508 
509                     ByteBuffer outAppBuf = pendingWrite.outAppBuf;
510                     if (outAppBuf == null) {
511                         // A write request with an empty buffer
512                         pendingUnencryptedWrites.remove();
513                         offerEncryptedWriteRequest(
514                                 new DownstreamMessageEvent(
515                                         channel, pendingWrite.future,
516                                         ChannelBuffers.EMPTY_BUFFER,
517                                         channel.getRemoteAddress()));
518                         offered = true;
519                     } else {
520                         SSLEngineResult result = null;
521                         try {
522                             synchronized (handshakeLock) {
523                                 result = engine.wrap(outAppBuf, outNetBuf);
524                             }
525                         } finally {
526                             if (!outAppBuf.hasRemaining()) {
527                                 pendingUnencryptedWrites.remove();
528                             }
529                         }
530 
531                         if (result.bytesProduced() > 0) {
532                             outNetBuf.flip();
533                             msg = ChannelBuffers.buffer(outNetBuf.remaining());
534                             msg.writeBytes(outNetBuf.array(), 0, msg.capacity());
535                             outNetBuf.clear();
536 
537                             if (pendingWrite.outAppBuf.hasRemaining()) {
538                                 // pendingWrite's future shouldn't be notified if
539                                 // only partial data is written.
540                                 future = succeededFuture(channel);
541                             } else {
542                                 future = pendingWrite.future;
543                             }
544 
545                             MessageEvent encryptedWrite = new DownstreamMessageEvent(
546                                     channel, future, msg, channel.getRemoteAddress());
547                             offerEncryptedWriteRequest(encryptedWrite);
548                             offered = true;
549                         } else {
550                             HandshakeStatus handshakeStatus = result.getHandshakeStatus();
551                             switch (handshakeStatus) {
552                             case NEED_WRAP:
553                                 if (outAppBuf.hasRemaining()) {
554                                     break;
555                                 } else {
556                                     break loop;
557                                 }
558                             case NEED_UNWRAP:
559                                 needsUnwrap = true;
560                                 break loop;
561                             case NEED_TASK:
562                                 runDelegatedTasks();
563                                 break;
564                             case FINISHED:
565                             case NOT_HANDSHAKING:
566                                 if (handshakeStatus == HandshakeStatus.FINISHED) {
567                                     setHandshakeSuccess(channel);
568                                 }
569                                 if (result.getStatus() == Status.CLOSED) {
570                                     success = false;
571                                 }
572                                 break loop;
573                             default:
574                                 throw new IllegalStateException(
575                                         "Unknown handshake status: " +
576                                         result.getHandshakeStatus());
577                             }
578                         }
579                     }
580                 }
581             }
582         } catch (SSLException e) {
583             success = false;
584             setHandshakeFailure(channel, e);
585             throw e;
586         } finally {
587             bufferPool.release(outNetBuf);
588 
589             if (offered) {
590                 flushPendingEncryptedWrites(context);
591             }
592 
593             if (!success) {
594                 IllegalStateException cause =
595                     new IllegalStateException("SSLEngine already closed");
596                 // Mark all remaining pending writes as failure if anything
597                 // wrong happened before the write requests are wrapped.
598                 // Please note that we do not call setFailure while a lock is
599                 // acquired, to avoid a potential dead lock.
600                 for (;;) {
601                     PendingWrite pendingWrite;
602                     synchronized (pendingUnencryptedWrites) {
603                         pendingWrite = pendingUnencryptedWrites.poll();
604                         if (pendingWrite == null) {
605                             break;
606                         }
607                     }
608 
609                     pendingWrite.future.setFailure(cause);
610                 }
611             }
612         }
613 
614         if (needsUnwrap) {
615             unwrap(context, channel, ChannelBuffers.EMPTY_BUFFER, 0, 0);
616         }
617 
618         if (future == null) {
619             future = succeededFuture(channel);
620         }
621         return future;
622     }

624     private void offerEncryptedWriteRequest(MessageEvent encryptedWrite) {
625         boolean offered;
626         if (Thread.holdsLock(pendingEncryptedWrites)) {
627             offered = pendingEncryptedWrites.offer(encryptedWrite);
628         } else {
629             synchronized (pendingEncryptedWrites) {
630                 offered = pendingEncryptedWrites.offer(encryptedWrite);
631             }
632         }
633         assert offered;
634     }
635 
636     private void flushPendingEncryptedWrites(ChannelHandlerContext ctx) {
637         // Avoid possible dead lock and data integrity issue
638         // which is caused by cross communication between more than one channel
639         // in the same VM.
640         if (Thread.holdsLock(pendingEncryptedWrites)) {
641             return;
642         }
643 
644         synchronized (pendingEncryptedWrites) {
645             if (pendingEncryptedWrites.isEmpty()) {
646                 return;
647             }
648         }
649 
650         synchronized (pendingEncryptedWrites) {
651             MessageEvent e;
652             while ((e = pendingEncryptedWrites.poll()) != null) {
653                 ctx.sendDownstream(e);
654             }
655         }
656     }

658     private ChannelFuture wrapNonAppData(ChannelHandlerContext ctx, Channel channel) throws SSLException {
659         ChannelFuture future = null;
660         ByteBuffer outNetBuf = bufferPool.acquire();
661 
662         SSLEngineResult result;
663         try {
664             for (;;) {
665                 synchronized (handshakeLock) {
666                     result = engine.wrap(EMPTY_BUFFER, outNetBuf);
667                 }
668 
669                 if (result.bytesProduced() > 0) {
670                     outNetBuf.flip();
671                     ChannelBuffer msg = ChannelBuffers.buffer(outNetBuf.remaining());
672                     msg.writeBytes(outNetBuf.array(), 0, msg.capacity());
673                     outNetBuf.clear();
674 
675                     future = future(channel);
676                     future.addListener(new ChannelFutureListener() {
677                         public void operationComplete(ChannelFuture future)
678                                 throws Exception {
679                             if (future.getCause() instanceof ClosedChannelException) {
680                                 synchronized (ignoreClosedChannelExceptionLock) {
681                                     ignoreClosedChannelException ++;
682                                 }
683                             }
684                         }
685                     });
686 
687                     write(ctx, future, msg);
688                 }
689 
690                 switch (result.getHandshakeStatus()) {
691                 case FINISHED:
692                     setHandshakeSuccess(channel);
693                     runDelegatedTasks();
694                     break;
695                 case NEED_TASK:
696                     runDelegatedTasks();
697                     break;
698                 case NEED_UNWRAP:
699                     if (!Thread.holdsLock(handshakeLock)) {
700                         // unwrap shouldn't be called when this method was
701                         // called by unwrap - unwrap will keep running after
702                         // this method returns.
703                         unwrap(ctx, channel, ChannelBuffers.EMPTY_BUFFER, 0, 0);
704                     }
705                     break;
706                 case NOT_HANDSHAKING:
707                 case NEED_WRAP:
708                     break;
709                 default:
710                     throw new IllegalStateException(
711                             "Unexpected handshake status: " +
712                             result.getHandshakeStatus());
713                 }
714 
715                 if (result.bytesProduced() == 0) {
716                     break;
717                 }
718             }
719         } catch (SSLException e) {
720             setHandshakeFailure(channel, e);
721             throw e;
722         } finally {
723             bufferPool.release(outNetBuf);
724         }
725 
726         if (future == null) {
727             future = succeededFuture(channel);
728         }
729 
730         return future;
731     }

733     private ChannelBuffer unwrap(
734             ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer, int offset, int length) throws SSLException {
735         ByteBuffer inNetBuf = buffer.toByteBuffer(offset, length);
736         ByteBuffer outAppBuf = bufferPool.acquire();
737 
738         try {
739             boolean needsWrap = false;
740             loop:
741             for (;;) {
742                 SSLEngineResult result;
743                 synchronized (handshakeLock) {
744                     if (initialHandshake && !engine.getUseClientMode() &&
745                         !engine.isInboundDone() && !engine.isOutboundDone()) {
746                         handshake(channel);
747                         initialHandshake = false;
748                     }
749 
750                     try {
751                         result = engine.unwrap(inNetBuf, outAppBuf);
752                     } catch (SSLException e) {
753                         throw e;
754                     }
755 
756                     switch (result.getHandshakeStatus()) {
757                     case NEED_UNWRAP:
758                         if (inNetBuf.hasRemaining() && !engine.isInboundDone()) {
759                             break;
760                         } else {
761                             break loop;
762                         }
763                     case NEED_WRAP:
764                         wrapNonAppData(ctx, channel);
765                         break;
766                     case NEED_TASK:
767                         runDelegatedTasks();
768                         break;
769                     case FINISHED:
770                         setHandshakeSuccess(channel);
771                         needsWrap = true;
772                         break loop;
773                     case NOT_HANDSHAKING:
774                         needsWrap = true;
775                         break loop;
776                     default:
777                         throw new IllegalStateException(
778                                 "Unknown handshake status: " +
779                                 result.getHandshakeStatus());
780                     }
781                 }
782             }
783 
784             if (needsWrap) {
785                 // wrap() acquires pendingUnencryptedWrites first and then
786                 // handshakeLock.  If handshakeLock is already hold by the
787                 // current thread, calling wrap() will lead to a dead lock
788                 // i.e. pendingUnencryptedWrites -> handshakeLock vs.
789                 //      handshakeLock -> pendingUnencryptedLock -> handshakeLock
790                 //
791                 // There is also a same issue between pendingEncryptedWrites
792                 // and pendingUnencryptedWrites.
793                 if (!Thread.holdsLock(handshakeLock) &&
794                     !Thread.holdsLock(pendingEncryptedWrites)) {
795                     wrap(ctx, channel);
796                 }
797             }
798 
799             outAppBuf.flip();
800 
801             if (outAppBuf.hasRemaining()) {
802                 ChannelBuffer frame = ChannelBuffers.buffer(outAppBuf.remaining());
803                 frame.writeBytes(outAppBuf.array(), 0, frame.capacity());
804                 return frame;
805             } else {
806                 return null;
807             }
808         } catch (SSLException e) {
809             setHandshakeFailure(channel, e);
810             throw e;
811         } finally {
812             bufferPool.release(outAppBuf);
813         }
814     }

816     private void runDelegatedTasks() {
817         for (;;) {
818             final Runnable task;
819             synchronized (handshakeLock) {
820                 task = engine.getDelegatedTask();
821             }
822 
823             if (task == null) {
824                 break;
825             }
826 
827             delegatedTaskExecutor.execute(new Runnable() {
828                 public void run() {
829                     synchronized (handshakeLock) {
830                         task.run();
831                     }
832                 }
833             });
834         }

865     private void closeOutboundAndChannel(
866             final ChannelHandlerContext context, final ChannelStateEvent e) throws SSLException {
867         if (!e.getChannel().isConnected()) {
868             context.sendDownstream(e);
869             return;
870         }
871 
872         unwrap(context, e.getChannel(), ChannelBuffers.EMPTY_BUFFER, 0, 0);
873         if (!engine.isInboundDone()) {
874             if (sentCloseNotify.compareAndSet(false, true)) {
875                 engine.closeOutbound();
876                 ChannelFuture closeNotifyFuture = wrapNonAppData(context, e.getChannel());
877                 closeNotifyFuture.addListener(
878                         new ClosingChannelFutureListener(context, e));
879                 return;

881         }
