50  /**
51   * A Guice {@link Module} that defines the bindings for all known
52   * {@link ChannelFactory}.
53   *
54   * @author The Netty Project (netty-dev@lists.jboss.org)
55   * @author Trustin Lee (tlee@redhat.com)
56   * @version $Rev: 1580 $, $Date: 2009-07-17 21:49:32 +0900 (Fri, 17 Jul 2009) $
57   *
58   * @apiviz.landmark
59   */
60  public class NettyModule extends AbstractModule {
61  
62      private final ExecutorService executor = Executors.newCachedThreadPool();
63  
64      /**
65       * Releases all resources created by this module.
66       */
67      public void destroy() {
68          ExecutorUtil.terminate(executor);
69      }
70  
71      @Override
72      protected void configure() {
73          if (executor.isShutdown()) {
74              throw new IllegalStateException(
75                      "Executor has been shut down already.");
76          }
77  
78          Executor executor = new UnterminatableExecutor(this.executor);
79  
80          bind(Executor.class).
81              annotatedWith(ChannelFactoryResource.class).
82              toInstance(executor);
83  
84          bind(ClientSocketChannelFactory.class).
85              toProvider(NioClientSocketChannelFactoryProvider.class).
86              in(Scopes.SINGLETON);
87  
88          bind(ServerSocketChannelFactory.class).
89              toProvider(NioServerSocketChannelFactoryProvider.class).
90              in(Scopes.SINGLETON);
91  
92          bind(DatagramChannelFactory.class).
93              toProvider(OioDatagramChannelFactoryProvider.class).
94              in(Scopes.SINGLETON);
95  
96          bind(NioClientSocketChannelFactory.class).
97              toProvider(NioClientSocketChannelFactoryProvider.class).
98              in(Scopes.SINGLETON);
99  
100         bind(NioServerSocketChannelFactory.class).
101             toProvider(NioServerSocketChannelFactoryProvider.class).
102             in(Scopes.SINGLETON);
103 
104         bind(NioDatagramChannelFactory.class).
105             toProvider(NioDatagramChannelFactoryProvider.class).
106             in(Scopes.SINGLETON);
107 
108         bind(OioClientSocketChannelFactory.class).
109             toProvider(OioClientSocketChannelFactoryProvider.class).
110             in(Scopes.SINGLETON);
111 
112         bind(OioServerSocketChannelFactory.class).
113             toProvider(OioServerSocketChannelFactoryProvider.class).
114             in(Scopes.SINGLETON);
115 
116         bind(OioDatagramChannelFactory.class).
117             toProvider(OioDatagramChannelFactoryProvider.class).
118             in(Scopes.SINGLETON);
119 
120         // Local transports
121         bind(LocalClientChannelFactory.class).
122             to(DefaultLocalClientChannelFactory.class).
123             in(Scopes.SINGLETON);
124 
125         bind(LocalServerChannelFactory.class).
126             to(DefaultLocalServerChannelFactory.class).
127             in(Scopes.SINGLETON);
128 
129         bind(DefaultLocalClientChannelFactory.class).
130             to(DefaultLocalClientChannelFactory.class).
131             in(Scopes.SINGLETON);
132 
133         bind(DefaultLocalServerChannelFactory.class).
134             to(DefaultLocalServerChannelFactory.class).
135             in(Scopes.SINGLETON);
136     }
